int find_m(vector<vector<int>>& m) {
    int max = 0;
    vector<int> nums;
    for (int i = 0; i < m.size(); i++) {
        for (int j = 0; j < 3; j++) {
            nums.push_back(m[i][j]);
        }
    }
    for (int i = 0; i < nums.size(); i++) {
        if (max < abs(nums[i])) {
            max = abs(nums[i]);
        }
    }
    return max;
}


vector<bool> StudentSolver :: solve(vector<vector<int>> bulbs) {
    int M = find_m(bulbs);

    vector<bool> wire_out;

    for (int i = 0; i < M; i++) {
        wire_out.push_back(false);
    }

    int N = bulbs.size();
    int sum;
    vector < bool > sum_row;
    for (int i = 0; i < N; i++) {
        sum = 0;
        for (int j = 0; j < 3; j++) {
            if (bulbs[i][j] < 0) {
                sum--;
            }
            else
            {
                sum++;
            }
        }
        if (sum == 3) {
            sum_row.push_back(false);
        }
        else {
            sum_row.push_back(true);
        }
    }
    int count = 0;
    while (count < N) {
        if (sum_row[count] == true) {
            count++;
        }
        else
        {
            int num_set;
            for (int i = 0; i < 3; i++) {
                if (bulbs[count][i] > 0) {
                    num_set = bulbs[count][i];
                    if (wire_out[num_set - 1] == false) {
                        wire_out[num_set - 1] = true;
                        sum_row[count] = true;
                        break;
                    }
                    else
                    {
                        wire_out[num_set - 1] = true;
                        sum_row[count] = true;
                        break;
                    }

                }
            }
            count = 0;
        }
    }

    return wire_out;
}
