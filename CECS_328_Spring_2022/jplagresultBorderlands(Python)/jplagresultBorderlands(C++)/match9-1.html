#pragma once
#ifndef STUDENT_SOLVER_H
#define STUDENT_SOLVER_H
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;


namespace StudentSolver
{
	
	struct wireSets
	{
		int number;
		bool onOrOff;
		inline wireSets(int num)
		{
			number = abs(num);
			onOrOff = false;
		}
		friend bool operator==(const wireSets& lhs, const wireSets& rhs) 
		{
			return (rhs.number == lhs.number);
		}
	};
	
	static bool sortHelper(wireSets lhs, wireSets rhs)
	{
		return (lhs.number < rhs.number);
	}
	
	static vector<bool> solve(vector<vector<int>> bulbs)
	{
		srand(time(NULL));
		vector<wireSets> board;
		
		
		
		
		
		
		
		
		
		
		
		
		
		int biggestNum = 0;
		for (size_t i = 0; i < bulbs.size(); i++)
		{
			for (size_t j = 0; j < bulbs[i].size(); j++)
			{

				if (biggestNum < abs(bulbs[i][j]))
				{
					biggestNum = abs(bulbs[i][j]);
				}
			}

		}
		for (int i = 0; i < biggestNum; i++)
		{
			wireSets temp(i + 1);
			board.push_back(temp);
		}
		
		
		
		

		int count = (rand() + 21);
		for (int i = 0; i < board.size() + 50; i++)
		{
			count = (rand() + 21);
			int index = (rand() % board.size());
			if (count % 2 == 0)
			{
				board[index].onOrOff = false;
			}
			else
			{
				board[index].onOrOff = true;
			}
		}
		
		
		
		
		bool atLeastOneOn = false;
		int faultyWire = 0;
		int rowCount = 0;
		int everythingIsOff = 0;
		while (!atLeastOneOn)
		{
			for (size_t i = 0; i < bulbs.size(); i++)
			{
				if (everythingIsOff == 3)
				{
					everythingIsOff = 0;
					
				}
				for (size_t j = 0; j < bulbs[i].size(); j++)
				{
					int numWire = bulbs[i][j];

					if (numWire < 0)
					{
						bool opposite = board[abs(numWire) - 1].onOrOff;
						if (opposite == true)
						{
							atLeastOneOn = false;
							everythingIsOff++;
							faultyWire = abs(numWire);
							continue;
						}
						else
						{
							atLeastOneOn = true;
							break;
						}
					}
					else
					{
						if (board[numWire - 1].onOrOff)
						{
							atLeastOneOn = true;
							break;
						}
						else
						{
							atLeastOneOn = false;
							everythingIsOff++;
							faultyWire = abs(numWire);
							continue;
						}
					}
				}

				if (atLeastOneOn == true)
				{
					everythingIsOff = 0;
					rowCount++;
				}
				else if (everythingIsOff == 3)
				{
					board[faultyWire - 1].onOrOff = !(board[faultyWire - 1].onOrOff);

				}
				
				
				
				
				
			}
		}
		
		
		
		
		


		
		vector<bool> answer;
		
		
		
		
		for (int i = 0; i < board.size(); i++)
		{
			answer.push_back(board[i].onOrOff);
		}
		
		
		
		


		
		
		
		
		
		return answer;

	}
};
#endif
