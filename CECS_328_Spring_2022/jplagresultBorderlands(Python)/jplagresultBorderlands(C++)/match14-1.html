vector<bool> StudentSolver::solve(vector<vector<int>> bulbs) {
	int M = abs(bulbs[0][0]);

	for (int i = 0; i < bulbs.size(); i++) {
		for (int j = 0; j < 3; j++) {
			if (abs(bulbs[i][j]) > M) {
				M = abs(bulbs[i][j]);
			}
		}
	}
	vector<bool> instructions;
	vector<bool> test = {false, false, false, true, true, false, false, false, false, true, false, true, false, false, false};
	for (int k = 0; k < M; k++) {
		int random = rand() % 2 + 1;
		if (random == 1) {
			instructions.push_back(true);
		}
		if (random == 2) {
			instructions.push_back(false);
		}
		if (random == 3) {
			instructions.push_back(false);
		}
	}
	/*for (int i = 0; i < bulbs.size(); i++) {

			}
			else {
				instructions[abs(bulbs[i][j]) - 1] = true;
			}
		}
	} */ 
	return instructions;
}
