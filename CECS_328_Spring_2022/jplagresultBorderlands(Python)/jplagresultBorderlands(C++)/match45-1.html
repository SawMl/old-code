#include <iostream>
#include <vector>
#include "StudentSolver.h"
#include <algorithm> 
#include <iterator>
using namespace std;

bool contains(vector<int> arr, int num)
{
    for (int i=0; i < arr.size(); i++)
    {
        if (arr[i] == num)
            return true;
    }
    return false;
}

vector<bool> StudentSolver::solve(vector<vector<int>> bulbs)
{

    int rows = bulbs.size(); 
    int largestNum = 0;
    
    for (int i=0; i < rows; i++)
    {
        for (int j=0; j < 3; j++)
        {
            if (abs(bulbs[i][j]) > largestNum)
                largestNum = abs(bulbs[i][j]);  
        }
    }
    
    vector<bool> answer (largestNum, false);

    
    vector<int> on (rows, 0);

    
    for (int i=0; i < largestNum; i++)
    {
        for (int j=0; j < rows; j++)
        {
            
            if (on[j])
                continue;
            int x = ((i+1)*-1);
            int y = (i+1);

            
            if (contains(bulbs[j], y))
            {
                answer[i] = true;
                break;
            }

            
            if (contains(bulbs[j], x))
            {
                on[j] = 1;
                answer[i] = false;
                break;
            }
            
        }
    }

    return answer;
}
