#include "StudentSolver.h"
#include <iostream>
#include <set>
#include <tuple>
using namespace std;

vector<bool> StudentSolver::solve(vector<vector<int>> bulbs)
{
	vector<vector<int>> count;
	for (int i = 0; i < bulbs.size(); i++)
	{
		for (int j = 0; j < bulbs[0].size(); j++)
		{
			if (abs(bulbs[i][j]) > count.size())
			{
				for (int l = count.size(); l < abs(bulbs[i][j]); l++)
				{
					count.push_back({ 0,0 });
				}
			}
			if (bulbs[i][j] > 0)
			{
				count[bulbs[i][j] - 1][0]++;
			}
			else
			{
				count[abs(bulbs[i][j]) - 1][1]++;
			}
		}
	}
	/*for (int i = 0; i < count.size(); i++)
	{
		cout << i << ": pos: " << count[i][0] << "neg: " << count[i][1];
	}*/
	vector<bool> answer(count.size());
	for (int i = 0; i < answer.size(); i++)
	{
		if (count[i][0] > count[i][1])
		{
			answer[i] = true;
		}
	}

	vector<bool> check(bulbs.size());

	for (int i = 0; i < bulbs.size(); i++)
	{
		for (int j = 0; j < bulbs[0].size(); j++)
		{
			if ((bulbs[i][j] < 0 && !answer[abs(bulbs[i][j]) - 1]) || (bulbs[i][j] > 0 && answer[abs(bulbs[i][j]) - 1]))
			{
				check[i] = true;
				break;
			}
		}
	}

	for (int i = 0; i < check.size(); i++)
	{
		if (!check[i])
		{
			if (answer[abs(bulbs[i][0]) - 1])
				answer[abs(bulbs[i][0]) - 1] = false;
			else
				answer[abs(bulbs[i][0]) - 1] = true;
		}
	}

	return answer;
}
