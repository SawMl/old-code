#include "StudentSolver.h"


vector<bool> StudentSolver::solve(vector<vector<int>>bulbs) {
	
	
	int m1 = *max_element(bulbs[0].begin(), bulbs[0].end());;
	int m2 = *max_element(bulbs[1].begin(), bulbs[1].end());;
	int m3 = *max_element(bulbs[2].begin(), bulbs[2].end());;
	int mid;
	if (m1 > m2) {
		if (m1 > m3) {
			mid = m1;
		}
		else {
			mid = m3;
		}
	}
	else {
		if (m2 > m3) {
			mid = m2;
		}
		else {
			mid = m3;
		}
	}
	mid = mid / 2;

	for (int a = 0; a < bulbs.size(); a++) {
		for (int b = 0; b < 3; b++) {
			if (abs(bulbs[a][b]) == mid) {
				if (bulbs[a][b] >= 0) {
					TFList.push_back(true);
					break;
				}
				else {
					TFList.push_back(false);
					break;
				}
			}
		}
		if (bulbs.size() > TFList.size()) {
			TFList.push_back(true);
		}
	}

	int small = mid - 1;
	for (int a = 0; a < bulbs.size(); a++) {
		for (int b = 0; b < 3; b++) {
			if (abs(bulbs[a][b]) == small) {
				if (bulbs[a][b] >= 0) {
					if (TFList[a]) {
						break;
					}
				}
				else {
					if (!TFList[a]) {
						break;
					}
				}
			}
		}
		if (bulbs.size() > TFList.size()) {
			TFList.push_back(true);
		}
		small--;
	}

	int big = mid + 1;
	for (int a = 0; a < bulbs.size(); a++) {
		for (int b = 0; b < 3; b++) {
			if (abs(bulbs[a][b]) == big) {
				if (bulbs[a][b] >= 0) {
					if (TFList[a]) {
						break;
					}
				}
				else {
					if (!TFList[a]) {
						break;
					}
				}
			}
		}
		if (bulbs.size() > TFList.size()) {
			TFList.push_back(true);
		}
		big++;
	}

	return TFList;
}
