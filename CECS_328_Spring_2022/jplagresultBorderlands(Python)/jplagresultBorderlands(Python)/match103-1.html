def is_on(bulb_set):
    on = 0
    off = 0
    for i in range(len(bulb_set)):
        if (bulb_set[i] < 0):
            on += 1
        if (bulb_set[i] > 0):
            off += 1
    if (on > 0):
        return True
    elif (off == 3):
        return False

def find_off(bulb_set):
    off_set = []
    for i in range(len(bulb_set)):
        if (bulb_set[i] > 0):
            off_set.append(bulb_set[i])
    return off_set
