def solve(bulbs):
  #initalize the list we are going to be returning
  answer_list = []
  
  #determining the sixe of the matrix verticle wize
  index_V = 0
  for i in range(len(bulbs)):
    index_V += 1 #length of the matrix we gonna return verticle wise

  #holds the list containing all their (abs) values
  bulbs_list_values = []
  for i in bulbs:
    for j in i:
      bulbs_list_values.append(abs(j))

  #value of the highest int in the matrix
  M = max(bulbs_list_values)

  #holds the list containing all their (true) values
  bulbs_list = []
  for i in bulbs:
    for j in i:
      bulbs_list.append(j)

  # print(bulbs_list) #print
  # print(bulbs_list_values) #print
  # print(index_V) #print
  # print(M) #print
  # print(bulbs[0]) #print

  #make a list that holds how many negatives and positives there are for each int
  negative_list = []
  positive_list = []
  for i in range(1, (M+1)):
    negative_counter = bulbs_list.count(-i)
    negative_list.append(negative_counter)
  
  # print(negative_list) #print

  for i in range(1, (M+1)):
    positive_counter = bulbs_list.count(i)
    positive_list.append(positive_counter)

  # print(positive_list) #print

  boobies = 0
  
  while(boobies < M):
    if((negative_list[boobies] >= 1) and (negative_list[boobies] >= positive_list[boobies])):
      answer_list.append(False)
    elif(positive_list[boobies] > negative_list[boobies] ):
      answer_list.append(True)
    elif((negative_list[boobies] == 1) and (positive_list[boobies] == 0)):
      answer_list.append(False)
    elif((positive_list[boobies] == 1) and (negative_list[boobies] == 0)):
      answer_list.append(True)
    elif((positive_list[boobies] == 1) and (negative_list[boobies] == 0)):
      answer_list.append(False)
    else:
      answer_list.append(False)

    boobies += 1

  return(answer_list)
