def allLight(bulb, config):
    for i in range(len(bulb)-1): 
        if (bulb[i][0] < 0 and config[abs(bulb[i][0])-1] == False) or (bulb[i][0] > 0 and config[abs(bulb[i][0])-1] == True):
            continue
        
        if (bulb[i][1] < 0 and config[abs(bulb[i][1])-1]) == False or (bulb[i][1] > 0 and config[abs(bulb[i][1])-1] == True):
            continue

        if (bulb[i][2] < 0 and config[abs(bulb[i][2])-1]) == False or (bulb[i][2] > 0 and config[abs(bulb[i][2])-1] == True):
            continue
        
        return i
    

    return -1

def solve(bulb):
    max = 0

    for i in range(len(bulb)-1):
        for j in range(3):
            if abs(bulb[i][j]) > max:
                 max = abs(bulb[i][
j])

    configuration = []
    for i in range(max):
        configuration.append(False)

    for i in range(len(bulb)-1):
        for j in range(3):
            if bulb[i][j] > 0:
                configuration[bulb[i][j]-1] = True
            else :
                configuration[abs(bulb[i][j])-1] = False
        

    check = allLight(bulb,configuration)

    while(check != -1):
        i = 0
        double_check = allLight(bulb,configuration)
        while(double_check == check):
            if configuration[abs(bulb[check][i])] == True:
                configuration[abs(bulb[check][i])] = False
            else:
                configuration[abs(bulb[check][i])] = True
            i = i + 1
            double_check = allLight(bulb,configuration)
        
        check = allLight(bulb,configuration)

    return configuration
