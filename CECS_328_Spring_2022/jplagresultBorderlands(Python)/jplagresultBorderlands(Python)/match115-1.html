def checkBoard(arr):

def checkAll(arr):
    for row in arr:    
        if not validRow(row):
            return False

def reduce(arr):
    res = []
    for row in arr:
        removeDupe = list(set(row))
        removeDupe.sort(key=abs)
        if removeDupe not in res:
            res.append(removeDupe)
        
    return res


def generateTotal(arr):
    x = 0
    for i in range(len(arr)): 
        for j in range(len(arr[i])):
            if arr[i][j] < 0:
                test =  arr[i][j] 
            else: 
                test =  arr[i][j]
            if test > x:
                x = test
    return x

def bulbStatus(x):

def validRow(row):
    
    for num in row:
        if bulbStatus(num):
