def solve(bulbs):

    #declare both N and M
    #N is number of rows with 3 bulbs each
    #M is number of wire sets
    N = len(bulbs)
    #print(N)
    M = maxM(bulbs)
    #print(M)

    #print("hello world 2")  
   
    OnBulb = [0] * N
    booleanList = [False] * M
    solution = [0] * N


    #print("hello world 4")  


    #base cases:
    
for a in range(M):
        for b in range(N):

            
            if OnBulb[b]:
                continue

            #else if the current wire is there but the light bulb had red paint (negative)
            if (-1 * a in bulbs[b]):
                booleanList[a] = False
                #print("1")
                #print(booleanList)
                OnBulb[b] = 1
                break
                
            #print("hello world 5")   

            
            #if the current wire is there and the light bulb does not have red paint (positive)
            if(a in bulbs[b]):
                booleanList[a] = True
                #print("2")
                #print(booleanList)
                break


            #print("hello world 6")  

            #print("hello world 7 ")  
    #print("hello world 8")           
    #print(booleanList)

    #check to see if there is a row with a 1
    #not all rows have a bulb that is turned on 

        
    for i in range(N):
        for j in range(3):
            if bulbs[i][j] < 0:
                if booleanList[abs(bulbs[i][j]) - 1] == False:
                    solution[i] = 1
            elif bulbs[i][j] > 0:
                if booleanList[bulbs[i][j] - 1] == True:
                    solution[i] = 1


    return booleanList
