def solve(bulbs):
    # find M in bulbs
    max = 1
    for i in range(len(bulbs)):
        for j in range(len(bulbs[i])):
            if max < bulbs[i][j]:
                max = bulbs[i][j]

    # boolean list
    solution = [False for i in range(max)]
    # Begin at the middle of the M to find a solution
    start = int(max / 2)
    solution[start - 1] = True

    # traverse through list
    count = 1
    while count < max:
        for i in range(len(bulbs)):
            for j in range(len(bulbs[i])):
                if abs(bulbs[i][j]) == start:
                    pass
                elif abs(bulbs[i][j]) == count:
                    found_element = j
                    broken_list = bulbs[i][found_element:] + bulbs[i][:found_element]               # contains the the elements in the row making count the first element of broken_list
                    if count == bulbs[i][j] and solution[abs(broken_list[1]) - 1] is False and solution[abs(broken_list[2]) - 1] is False:
                        solution[count - 1] = True
        count += 1
    return solution
