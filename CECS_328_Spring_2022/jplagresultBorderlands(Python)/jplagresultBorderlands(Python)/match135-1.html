def solve(bulbs):
    #print("bulbs: ",bulbs)
    m = maxVal(bulbs)
    lights = dict()
    weights = weighted(bulbs,m)
    #print("weights: ",weights)
    
    for row in bulbs:
        maxW = [0,0] #store bulb and max weight
        if not rowOn(row,lights):
            #print("rowOff: ",row)
            for i in range(len(row)):
                #print("light= ", row[i])
                if abs(row[i]) in lights:
                    continue
                if weights[row[i]] > maxW[1]:
                    maxW[0],maxW[1] = row[i], weights[row[i]]
             
            if maxW[0] == 0:
                print("failed")
            if row[0]==row[1]==row[2]:
                maxW[0],maxW[1] = row[0], weights[row[0]]
                
            if maxW[0]<0:
                lights[abs(maxW[0])] = False
                #print("F ",maxW[0])
            else:
                lights[abs(maxW[0])] = True
                #print("T ",maxW[0])
        #else:
            #print("rowOn: ",row)
              
    #print(lights)  
    lights = populate(lights,m)
    return lights[1:m+1]

def rowOn(row,diction):
    for i in row:
        if abs(i) in diction:
            if i>0 and diction[abs(i)] or i<0 and not diction[abs(i)]:
                return True
    return False
    
def populate(diction,m):
    array = [False]*(m+1)
    for i in range(len(array)):
        if i in diction:
            array[i] = diction[i
]
    return array
    
def maxVal(array):
    minmax = 0
    for i in array:
        for j in i:
            if abs(j)>minmax:
                minmax = abs(j)
            
    return minmax 
    
def weighted(array,m):
    count= dict()
    count = count.fromkeys(range(-m,m+1),0)
    for i in array:
        for j in i:
            count[j] += 1
    return count
