def solve(bulbs):
    switch=[0 for n in range(12)]
    results=[False for m in range(15)]
    for m in range(14):
        for n in range(10):
            if switch[n]:
                continue
            if -1*(m+1) in bulbs[n]:
                results[m]=False
                switch[n]=1
                break
            if (m+1) in bulbs[n]:
                results[m]=True
                break
    return results

bulbs=[[-4,-4,-1],
    [13,-9,-14],
    [4,-7,-2],
    [-2,6,-3],
    [-6,-8,11],
    [-4,-3,-6],
    [7,11,-10],
    [4,-7,6],
    [5,6,4],
    [10,14,-9],
    [-6,8,11],
    [6,-8,11]]
print("Bulbs: ")
print(bulbs)
results=solve(bulbs)
print("Results: ")
print(results)
