def nums_present(bulbs):
    temp = []
    for x in range(len(bulbs)):
        for y in range(len(bulbs[x])):
            pos = bulbs[x][y]
            pos = abs(pos)
            if pos not in temp:
                temp.append(pos)
    return temp
