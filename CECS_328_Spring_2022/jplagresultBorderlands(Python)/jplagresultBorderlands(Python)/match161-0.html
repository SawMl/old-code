import math

def solve(bulbs):
    
    #holds the return value
    result = []
    
    #Get the N value (number of rows)
    N = len(bulbs)

    #Get the M value (number of wire sets)
    maximum = 0
    minimum = 0

    for i in bulbs:

        lineMax = max(i)

        if lineMax > maximum:
            maximum = lineMax

        lineMin = min(i)

        if lineMin < minimum:
            minimum = lineMin

    M = max(int(math.fabs(maximum)), int(math.fabs(minimum)))

    #print(M)  

    #holds whether the row has a bulb turned on or not
    rowsLeft = set()

    for i in range(1, N + 1):
        rowsLeft.add(i)

    #represents all of the rows
    allRows = set()
    
    for i in range(1, N + 1):
        allRows.add(i)

    #bulbs to be decided
    bulbsLeft = set()

    for i in range(1, M + 1):
        bulbsLeft.add(i)

    #holds the rows that each wireset must be on in to light up
    onRows = {}
    for i in range(1, M + 1):
        onRows[i] = set()

    #holds the rows that each wiresett must be off in to light up
    offRows = {}
    for i in range(1 , M + 1):
        offRows[i] = set()

    #holds whether each wireset is on or off
    onOrOff = {}
    for i in range(1, M + 1):
        onOrOff[i] = None

    #go through each row and hold whether the wireset needs to be on or off in that row
    for i in range(0, N):
        for j in range(3):
            
            wireset = math.fabs(bulbs[i][j])

            if bulbs[i][j] 
                offRows[wireset].add(i + 1)

            else:
                onRows[wireset].add(i + 1)

    print(onRows)
    print(offRows) 

    for i in range(1, M +1):

        #if the wireset only shows up as on or off, set it to that value and remove the rows it covers from the set
        if len(onRows[i]) == 0:

            onOrOff[i] = False
            bulbsLeft -= {i}
            rowsLeft -= offRows[i]

        elif len(offRows[i]) == 0:
            
            onOrOff[i] = True
            bulbsLeft -= {i}
            rowsLeft -= onRows[i]

    #iterate through the remaining bulbs
    removeBulbs = set()

    for i in bulbsLeft:

        if len(rowsLeft) != 0:

            #if it does not have anything in either "on" or "off" mode that can check off a row, set it to the opposite and remove it
            if (len(onRows[i] & rowsLeft) == 0 ):
            
                onOrOff[i] = False
                rowsLeft -= offRows[i]
                removeBulbs.add(i)

            elif(len(offRows[i] & rowsLeft) == 0):

                onOrOff[i] = True
                rowsLeft -= onRows[i]
                removeBulbs.add(i)


    bulbsLeft -= removeBulbs

    #if that satisfies the end, set everything else to false (doesn't matter)
    if len(rowsLeft) == 0:

        print(onOrOff)
        print(bulbsLeft)
        
        for i in bulbsLeft:
            onOrOff[i] = False

        for i in range(1, M + 1):
            result.append(onOrOff[i])

        return result

    #find a way to iterate through remaining values, or use recursion?
    else:

        for i in bulbsLeft:

            if len(onRows[i] & rowsLeft) > len(offRows[i] & rowsLeft):

                onOrOff[i] = True
                rowsLeft -= onRows[i]

            else: 

                onOrOff[i] = False
                rowsLeft -= offRows[i]

        for i in range(1, M + 1):
            
            result.append(onOrOff[i])

        print(onOrOff)
        print(bulbsLeft)

        return result
