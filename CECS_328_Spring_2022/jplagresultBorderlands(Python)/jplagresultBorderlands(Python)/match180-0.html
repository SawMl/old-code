#!/usr/bin/python

import sys

def recursiveSolve(solution, i, bulbSize, truthTable, m):
  for k in range(0, m):
    if(truthTable[i][k] != None):
      if(solution[k] == None or solution[k] == truthTable[i][k]):
        if (i < bulbSize - 1):
          solutionCloned = solution.copy()
          solutionCloned[k] = truthTable[i][k]
          solutionCloned = recursiveSolve(solutionCloned, i+1, bulbSize, truthTable, m)
          if (len(solutionCloned) != 0):
            return solutionCloned
        else:
          solution[k] = truthTable[i][k]
          return solution
  return []

def solve(bulbs):
  m = 0
  for row in bulbs:
    for e in row:
      if (abs(e) > m):
        m = abs(e)


  truthTable = []
  # for i in range(0, len(bulbs)):
  #   truthTable.append([None]*m)


  for i in range(0, len(bulbs)):
    if not (len(bulbs[i]) > 1 and bulbs[i][0] + bulbs[i][1] == 0 or (len(bulbs[i]) > 2 and (bulbs[i][1] + bulbs[i][2] == 0 or bulbs[i][0] + bulbs[i][2] == 0))):  
      truthTable.append([None]*m)
      for e in bulbs[i]:
        absolute = abs(e)
        if absolute == e:
          truthTable[len(truthTable)-1][absolute-1] = True
        else:
          truthTable[len(truthTable)-1][absolute-1] = False

  print(len(bulbs))
  print(len(truthTable))
  solution = [None] * m

  solution = recursiveSolve(solution, 0, len(truthTable), truthTable, m)

  for i in range(0, len(solution)):
    if solution[i] == None:
      solution[i] = True
  
  return solution


entry = sys.argv[1:len(sys.argv)]

bulbs = []
i = 0
while i < len(entry):
  sublist = []
  if i < len(entry):
    sublist.append(int(entry[i]))
  if i+1 < len(entry):
    sublist.append(int(entry[i+1]))
  if i+2 < len(entry):
    sublist.append(int(entry[i+2]))

  min = abs(int(entry[i]))
  for e in sublist:
    if abs(e) < min:
      min = abs(e)
  
  index = 0
  searching = True
  while(searching and index < len(bulbs)):
    e = bulbs[index]
    if(abs(e[0]) > min and abs(e[1]) > min and abs(e[2]) > min):
      searching = False
      break
    if searching:
      index += 1
      
  bulbs.insert(index, sublist)
