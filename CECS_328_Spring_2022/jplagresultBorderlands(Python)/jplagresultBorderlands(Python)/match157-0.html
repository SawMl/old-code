import random

#Function to validate if the current configuration turns on at least one light bulb in every row ----------------
def validate(current_config, bulbs):
    for i in range(len(bulbs)):
        num_of_lights_off = 0
        for j in range(len(bulbs[i])):
            if current_config[abs(bulbs[i][j]) - 1] == True and bulbs[i][j] < 0:
                num_of_lights_off += 1
            elif current_config[abs(bulbs[i][j]) - 1] == False and bulbs[i][j] > 0:
                num_of_lights_off += 1
            if num_of_lights_off == len(bulbs[i]):
                return False
    return True

#Function to determine the number of wiresets in the given sets of light bulbs ----------------------------------
def determine_num_of_wiresets(bulbs):
    max = 0
    for i in range(0,len(bulbs)):
        for j in range(0,len(bulbs[i])):
            if abs(bulbs[i][j]) > max:
                max = abs(bulbs[i][j])
    return max

#Backtracking algorithm to attempt every permutation of True/False to test the output of the light bulbs --------
def create_and_validate_configuration(length, current, i, bulbs):
    #Base return case
    if i == length:
        return 0

    #Set current position to False, attempt validation, then continue
    current[i] = False
    create_and_validate_configuration(length, current, i + 1, bulbs)
    if (validate(current, bulbs)) == True:
        return current

    #Set current position to True, attempt validation, then continue
    current[i] = True
    create_and_validate_configuration(length, current, i + 1, bulbs)
    if (validate(current, bulbs)) == True:
        return current

#Main solve function that uses all other functions --------------------------------------------------------------
def solve(bulbs):
    
    #Determine the number of wiresets and length of final configuration
    num_of_wiresets = determine_num_of_wiresets(bulbs)
    configuration = [False] * num_of_wiresets

    #Create and validate configuration for light bulbs
    return create_and_validate_configuration(num_of_wiresets, configuration, 0, bulbs)
