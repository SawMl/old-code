def validate(wireSetVal, bulbs_Current): 
    curr_state = bulbs_Current[abs(wireSetVal) - 1]

    if(wireSetVal > 0 and curr_state == True):
        return True 
    elif(wireSetVal < 0 and curr_state == False): 
        return True 
    
    return False 

# Sets a valid state if the current wire set has not yet been assigned one
def setToValid(wireSetVal, bulbs_Current): 

    if(wireSetVal < 0):
        
        state_to_set = False
    else :
        state_to_set = True 
    
    bulbs_Current[abs(wireSetVal) - 1] = state_to_set

def getMaxWire(bulbs):

    # For getting the size of the bool list 
    boolSize = bulbs[0][0]
    
    for i in bulbs: 
        for j in i:
            if abs(j) > boolSize: 
                boolSize = abs(j)
    return  boolSize
