def solve(bulbs):
    # m - number of bulbs
    # n - number of rows
    m = 0
    n = 0
    #find the bulb with highest number
    for i in range(len(bulbs)):
        for j in range(3):
            n += 1
            if bulbs[i][j] < 0:
                if (bulbs[i][j]) * -1 > m:
                    m = bulbs[i][j] * -1
            else:
                if bulbs[i][j] > m:
                    m = bulbs[i][j]
    #each row has 3 bulbs

    n = int(n/3)

        pos.append(0)
        neg.append(0)

1
