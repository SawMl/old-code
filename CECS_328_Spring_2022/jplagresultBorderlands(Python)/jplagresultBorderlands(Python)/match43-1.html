def checkIfRowIsSatisfied(arr):
    
for i in range(M):
        for j in range(N):
            if has_current[j]:
                continue
            if -1 * (i + 1) in arr[j]:
                result[i] = False
                has_current[j] = 1
                break

            if (i + 1) in arr[j]:
                result[i] = True
                break


N = 10
M = 14

result = [False for i in range(M)]
has_current = [0 for i in range(N)]

def solve(bulbs):
    checkIfRowIsSatisfied(bulbs)
    print(result)
