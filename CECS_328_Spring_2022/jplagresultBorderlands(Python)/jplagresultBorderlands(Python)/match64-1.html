import random

def solve(bulbs):
  on_list = [False for i in range(len(bulbs))]
  max = 0
  c 
= 0
  for i in range(len(bulbs)):
    for j in range(3):
      if abs(bulbs[i][j]) > max:
        max = abs(bulbs[i][j])
  list = [False for i in range(max)]
  for j in range(max):
    c = c + 1
    max = len(bulbs)
    for k in range(len(bulbs)):
      c = len(bulbs)
      max = c
      if on_list[k]: # if row is already on, move onto next set
        c = c - 1
        continue
      if (j + 1) in bulbs[k]: # for regular bulbs of this set in the row
        list[j] = True
        c = len(bulbs)
        break
      if -abs(j + 1) in bulbs[k]: # for red paint bulbs of this set in the row
        on_list[k] = True
        c = max
        max = len(bulbs)
        list[j] = False
        break
  return list
