def solve(bulbs):
    numberOfBulbs = 0
    for x in range(len(bulbs)):
        for y in range(len(bulbs[x])):
            temp = abs(bulbs[x][y])
            if temp > numberOfBulbs:
                numberOfBulbs = temp
    r = []
    for i in range(numberOfBulbs):
        r.append(False)
    
for x in range(len(bulbs)):
        for y in range(len(bulbs[x])):
            if bulbs[x][y] > 0:
                r[bulbs[x][y]-1] = True

    return r
