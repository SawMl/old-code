import random


def solve(bulbs):

    N = random.randint(1, 100)
    M = random.randint(N + 4, N + 4)
    x = 1
    y = 0

    bulbs = []
    for i in range(N):
        bulb1 = [random.randint(N - 4, N), random.randint(N - 4, N), random.randint(N - 4, N)]
        bulbs.append(bulb1)

    print("These are the bulbs: ", bulbs)
    for bulb in bulbs:
        y += x
        print("Wire #", y, ": ", bulb)

    set_checker = [False for i in range(M)]
    bulb_turned = [0 for i in range(N)]

    for i in range(M):
        for j in range(N):

            if bulb_turned[j]:
                continue

            if (i + 1) * -1 in bulbs[j]:
                set_checker[i] = False
                bulb_turned[j] = 1
                break

            if (i + 1) * 1 in bulbs[j]:
                set_checker[i] = True
                break

    return bulbs


def main():
    bulbs = []
    N = random.randint(1, 100)
    for i in range(N):
        bulb1 = [random.randint(N - 4, N), random.randint(N - 4, N), random.randint(N - 4, N)]
        bulbs.append(bulb1)
    print(
        "===============================================================================================================")
    print(
        "                    W E L C O M E   T O   P R O G R A M M I N G    A S S I G N M E N T   # 2                   ")
    print(
        "===============================================================================================================")
    solve(bulbs)

main()
