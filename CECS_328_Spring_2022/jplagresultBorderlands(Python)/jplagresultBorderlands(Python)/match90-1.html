def solve(bulbs):
    N = 10 
    M = 14
    bulb_array 
= [[-4, -4, -1],[13, -9, -14],[4, -7, -2],[-2, 6, 3],[-6, -8, 11],
       [-4, -3, -6],[7, 11, -10],[4, -7, 6],[5, 6, 4],[10, 14, -9]]
    rows = [False for i in range(M)]
    current_checking = [0 for i in range(N)] 
    while(i==M):   
        for j in range(N):
            if -1 * (i + 1) in bulb_array[j]:
                rows[i] = False
                current_checking[j] = 1
                break                         
    bulbs = [0 for n in range(N)]
    for n in range(N):
        for m in range(3):
            if bulb_array[n][m] < 0 and rows[abs(bulb_array[n][m]) - 1] == False:
                bulbs[n] = 1
            elif bulb_array[n][m] > 0 and rows[bulb_array[n][m] - 1] == True:
                bulbs[n] = 1
print(rows)
print(bulbs)
