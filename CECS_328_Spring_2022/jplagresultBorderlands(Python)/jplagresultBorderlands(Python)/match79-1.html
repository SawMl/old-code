def max(bulbs, row, col):
    max = -1
    for i in range(row):
        for j in range(col):
            if abs(bulbs[i][j]) > max:
                max = bulbs[i][j]
            else:
                continue
    return max

def solve(bulbs):
    N = len(bulbs)
    col = 3
    M = max(bulbs, N, col)
    
    wire_sets = [False for i in range(M)]
    checked_sets = [False for i in range(M)]
    
    for i in range(N):
        for j in range(col):
            x = abs(bulbs[i][j]) - 1
            if checked_sets[x] == True:
                continue;
            else:
                if bulbs[i][j] < 0:
                    
if wire_sets[x] == False:
                        checked_sets[x] = True
                        break;
                else:
                    if wire_sets[x] == False:
                        wire_sets[x] = True
                        checked_sets[x] = True
                        break;
            
    return wire_sets
