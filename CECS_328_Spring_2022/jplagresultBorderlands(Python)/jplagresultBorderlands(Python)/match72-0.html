def solve(bulbs):

    m = get_M(bulbs)

    boolean_m = []
    for i in range(m):
        boolean_m.append(True)

    if if_alltrue(bulbs):
        return boolean_m



    bool_m = []
    for i in range(m):
        bool_m.append(False)



    pos_rows = [] #only pos vals in here
    for i in bulbs:
        if i[0] > 0 and i[1] > 0 and i[2] > 0:
            pos_rows.append(i)

    turn_wireset = []
    for i in pos_rows:
        turn_wireset.append(check_for_negs(i, bulbs))

    for i in turn_wireset:
        bool_m[i - 1] = True

    return bool_m

def if_alltrue(bulbs):

    arr = []
    for i in bulbs:
        if i[0] > 0 or i[1] > 0 or i[2] > 0:
            arr.append(i)

    if len(arr) == len(bulbs):
        return True
    else:
        return False

def get_M(bulbs):
    arr_M = []

    for i in bulbs:
        for j in i: #if choosing all true for false here: count for negs and pos here
            arr_M.append(abs(j))
    arr_M.sort()

    return arr_M[-1]


#takes in a row and bulbs and returns the value that should be on
def check_for_negs(row, bulbs):
    row_index = 0
    while row_index < 3:

        neg = []  # holds neg val of row[x]
        for i in bulbs:
            if -row[row_index] in i:
                neg.append(i)

        if neg is None:
            return row[row_index]

        if checkrow(neg, -row[row_index]):
            return row[row_index]

        row_index += 1


def checkrow(row, num):
    test = []
    for i in row:
        if (i[0] < 0 and i[0] != num) or (i[1] < 0 and i[1] != num) or (i[2] < 0 and i[2] != num):
            test.append(1)

        else:
            test.append(0)
    if test.count(1) == len(test):
        return True
    else:
        return False
