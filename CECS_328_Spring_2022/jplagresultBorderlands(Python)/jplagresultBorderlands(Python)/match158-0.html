import random
def solve(bulbs):
    #Create a hastable with all the values and their boolean pair.
    numBool = {}
    max = 0
    for i in range(len(bulbs)):
        for j in bulbs[i]:
            if abs(j) not in numBool:
                numBool[abs(j)] = True
            if j < 0:
                numBool[abs(j)] = False
            if abs(j) > max:
                max = abs(j)
    #Initialize boolean valid to make sure all number/boolean pairs are valid.
    valid = False
    #Infinite loop until there is a valid number/boolean pair for all numbers.
    while valid is False:
        valid = True
        #Iterate through 'bulbs' and throughout each line.
        for i in range(len(bulbs)):
            #Boolean represents if there is a true value in the line.
            hasTrue = False
            for j in bulbs[i]:
                #If negative number, take the opposite of the boolean value.
                if j < 0:
                    if not(numBool[abs(j)]) is False:
                        continue
                    #When true, hasTrue is True.
                    else:
                        hasTrue = True
                        break
                #If positive number, take it as it is.
                else:
                    if numBool[abs(j)] is False:
                        continue
                    #When True, hasTrue is True.
                    else:
                        hasTrue = True
                        break
            #If the line has no true value.
            if hasTrue is False:
                #Flip the boolean value of a random number and assign False to valid.
                randomIndex = random.randint(0, 2)
                numBool[abs(bulbs[i][randomIndex])] = not(numBool[abs(bulbs[i][randomIndex])])
                valid = False
                print('hi')
    #After breaking from the infinite loop, create the boolean array.
    boolList = []
    for i in range(max):
        if abs(i + 1) not in numBool:
            boolList.append(False)
        else:
            boolList.append(numBool[abs(i + 1)])
    return boolList
