import random


def maxValue(bulbs):
    maxVal = None
    minVal = None
    for lists in bulbs:
        for values in lists:
            if maxVal is None or values > maxVal:
                maxVal = values
    for lists in bulbs:
        for values in lists:
            if minVal is None or values < minVal:
                minVal = values
    if abs(minVal) > maxVal:
        maxVal = abs(minVal)
    return maxVal


def checkIfAllRowsOn(ans, bulbs):
    for lists in bulbs:
        switch = False
        for values in lists:
            if ans[abs(values) - 1] is None:
                switch = True
            if values > 0 and ans[abs(values) - 1]:
                switch = True
            if values < 0 and not ans[abs(values) - 1]:
                switch = True

        if not switch:
            return False

    return True


def checkIfRowOn(ans, lists):
    switch = False
    for values in lists:
        if ans[abs(values) - 1] is None:
            switch = True
        if values > 0 and ans[abs(values) - 1]:
            switch = True
        if values < 0 and not ans[abs(values) - 1]:
            switch = True

    if not switch:
        return False

    return True


def flipSwitch(ans, lists):
    for values in lists:
        if values < 0 and ans[abs(values) - 1] is True:
            ans[abs(values) - 1] = False
        if values > 0 and ans[abs(values) - 1] is False:
            ans[abs(values) - 1] = False


def resetAnsArr(ans, bulbs):
    ans.clear()
    maxVal = maxValue(bulbs)
    for index in range(maxVal):
        ans.append(random.choice([True, False]))


def solve(bulbs):
    ans = []
    maxVal = maxValue(bulbs)
    for index in range(maxVal):
        ans.append(random.choice([True, False]))

    while not checkIfAllRowsOn(ans, bulbs):
        resetAnsArr(ans, bulbs)
        for row in range(len(bulbs)):
            if not checkIfRowOn(ans, bulbs[row]):
                flipSwitch(ans, bulbs[row])
                print(ans)

    if checkIfAllRowsOn(ans, bulbs):
        return ans
