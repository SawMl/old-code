def solve(bulbs):
    # Find N
    N = len(bulbs)
    # Find M
    minValue = abs((min(list(map(min, bulbs)))))
    maxValue = (max(list(map(max, bulbs))))
    M = max(minValue, maxValue)

    # Setting all the values for answer list to False
    ans = [False for i in range(M)]

    # Go through the list and see if the bulb is already on
    bulb_on = [0 for i in range(N)]

    for i in range(M):
        for j in range(N):

            # Base condition
            if bulb_on[j]:
                continue

            # If the bulb in the row is positive
            if i + 1 in bulbs[j]:
                ans[i] = True
                break

            # If the bulb in the row is negative
            if -(i + 1) in bulbs[j]:
                ans[i] = False
                bulb_on[j] = 1
                break

    # List of result
    return ans
