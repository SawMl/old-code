    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
        ArrayList<Boolean> answer_list = new ArrayList<>();

       




        
        Integer[] sorted_array = new Integer[bulbs.size() * 3];

        if (bulbs.size() == 0) {
            return answer_list;
        }
        

        
        for (int i = 0; i < bulbs.size(); i++) {
            ArrayList<Integer> int_subset = bulbs.get(i);
            
            int first = Math.abs(int_subset.get(0)) - Math.abs(int_subset.get(1));
            int second = Math.abs(int_subset.get(1)) - Math.abs(int_subset.get(2));
            int third = Math.abs(int_subset.get(0)) - Math.abs(int_subset.get(2));

            if (Math.abs(first) <= 5 && Math.abs(second) <= 5 && Math.abs(third) <= 5) {
                for (int j = 0; j < 3; j++) {
                    sorted_array[3 * i + j] = int_subset.get(j);
                }
            } else {
                return answer_list;
            }
        }
        int first_element = (int) sorted_array[0];
        int num_element_track = 1;
        for(int i = 1; i < sorted_array.length; i++){
            if(first_element == sorted_array[i]){
                num_element_track++;
            }
        }
        if(num_element_track == sorted_array.length && sorted_array[0] > 0){
            answer_list.add(true);
            return answer_list;
        }
        else if(num_element_track == sorted_array.length && sorted_array[0] < 0){
            answer_list.add(false);
            return answer_list;
        }
        
        for (int i = 0; i < sorted_array.length; i++) {
            if (sorted_array[i] < 0) {
                sorted_array[i] = Math.abs(sorted_array[i]);
            }
        }

        Arrays.sort(sorted_array);

        
        int arrayValue = sorted_array[sorted_array.length - 1] - sorted_array[0];
        System.out.println(arrayValue);
        int[] intArray = new int[arrayValue + 1];

        for(int i = 0; i < intArray.length; i++){
            if(intArray[0] == 0){
                intArray[i] = sorted_array[i];
            }
            else{
                intArray[i] =  intArray[i - 1] + 1;
            }
        }

        for(int i = 0; i < intArray.length; i++){
            System.out.print(intArray[i] + " ");
        }
        System.out.println();
        
        for(int i = 0; i < sorted_array.length; i++)
        {
            System.out.print(sorted_array[i] + " ");
            if((i + 1) % 3 == 0)
            {
                System.out.print("\n");
            }
        }



        
        

        int[] track_boolean = new int[intArray.length];
        int temp = 0;
        for (int i = 0; i < intArray.length; i++) {
            for (int j = 0; j < sorted_array.length; j++) {
                if (intArray[i] == sorted_array[j]) {
                    temp += 1;
                }
                track_boolean[i] = temp;
            }
            temp = 0;
        }
        for (int i = 0; i < track_boolean.length; i++) {
            if (track_boolean[i] >= 2) {
                answer_list.add(i, true);
            } else {
                answer_list.add(i, false);
            }
        }
        return answer_list;
    }
