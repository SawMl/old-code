import java.util.ArrayList;
import java.util.TreeSet;

public class StudentSolver {

    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
        TreeSet<Integer> wires = new TreeSet<>();
        ArrayList<Boolean> switches = new ArrayList<>();
        ArrayList<ArrayList<Integer>> offRows;

        
        for (ArrayList<Integer> row : bulbs)
            for (int bulb : row)
                wires.add(Math.abs(bulb));

        ArrayList<Integer> wiresList = new ArrayList<>(wires);

        printTests(bulbs, wires, true);

        int numberOfTries = 0;
        int resultSize = wiresList.get(wiresList.size() - 1);
        do {
            switches = generateRandomResult(resultSize);
            numberOfTries++;

            if (numberOfTries > 100000)
                break;
        } while (!allIsOn(bulbs, switches));

        
System.out.println("Number of tries: " + numberOfTries);
        return switches;
    }

    public static void printTests(ArrayList<ArrayList<Integer>> bulbs, TreeSet<Integer> wires, boolean printAll) {
        if (printAll) {
            for (ArrayList<Integer> row : bulbs) {
                for (int bulb : row)
                    System.out.print(bulb + " ");
                System.out.println();
            }
        }

        System.out.print("All wires: ");
        for (int bulb : wires)
            System.out.print(bulb + " ");
        System.out.println("\nNumber of wire sets: " + wires.size());
        System.out.println("Size of test: (" + bulbs.size() + ", " + bulbs.get(0).size() + ")");
        System.out.println("End of file.\n");
    }

    public static boolean allIsOn(ArrayList<ArrayList<Integer>> bulbs, ArrayList<Boolean> switches, ArrayList<Integer> wires, ArrayList<ArrayList<Integer>> offRows) {
        for (ArrayList<Integer> row : bulbs) {
            boolean rowIsOff = true;
            for (int bulb : row) {
                boolean switchOn = switches.get(wires.indexOf(Math.abs(bulb)));
                if ((bulb > 0 && switchOn) || (bulb < 0 && !switchOn)) {
                    rowIsOff = false;
                    break;
                }
            }

            if (rowIsOff)
                offRows.add(new ArrayList<>(row));
        }

        return offRows.isEmpty();
    
}

    public static boolean allIsOn(ArrayList<ArrayList<Integer>> bulbs, ArrayList<Boolean> switches) {
        for (ArrayList<Integer> row : bulbs) {
            boolean rowIsOff = true;
            for (int bulb : row) {
                boolean switchOn = switches.get(Math.abs(bulb) - 1);
                if ((bulb > 0 && switchOn) || (bulb < 0 && !switchOn)) {
                    rowIsOff = false;
                    break;
                }
            }
            if (rowIsOff)
                return false
;
        }
        return true;
    }

    public static ArrayList<Boolean> generateRandomResult(int size) {
        ArrayList<Boolean> result = new ArrayList<>();

        for (int i = 0; i < size; i++) {
            if (Math.random() > 0.5)
                result.add(Boolean.TRUE);
            else
                result.add(Boolean.FALSE);
        }
        return result;
    }


}
