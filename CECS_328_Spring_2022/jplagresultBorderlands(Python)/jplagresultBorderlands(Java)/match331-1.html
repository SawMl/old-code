import java.util.*;

public class StudentSolver
{
	 public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs) 
	 {
			
			int N = bulbs.size(); 
			int randomNumber = (int)( Math.random() * 99999 + 1);
			int M = randomNumber; 
			ArrayList<Boolean> currentIndicator = new ArrayList<Boolean>(); 
			ArrayList<Integer> numberBulbOccurencesCollection = new ArrayList<Integer>(); 
			int countingThisNumberSet = 1; 
			int numberOfOccurences = 0; 
			Boolean[] boolArray = {true, false}; 
			boolean numberFoundBoolean; 
			ArrayList<Integer> arrayIndexNumberIsLocated = new ArrayList<Integer>(); 
			int searchNumber; 

			
			for(int rows = N - 1; rows >= 0; rows--) 
			{
				for(int columns = 2; columns >= 0; columns--) 
				{
					int currentNumber =  randomNumber * (bulbs.get(rows).get(columns));
									
					if(currentNumber < 0)
					{
						currentNumber = currentNumber * -1;
					}
							
					if (currentNumber > M)
					{
						M = currentNumber;
					}	
				}
			}
			M = M / randomNumber;
			
			
			for(int rows = 0; rows < N; rows++) {
		        for(int columns = 0; columns < 3; columns++) {
		        	if (bulbs.get(rows).get(columns) == countingThisNumberSet)
		        	{
		        		numberOfOccurences ++;
		        	}
		        	
		        	if(rows == (N - 1) && columns == 2 && countingThisNumberSet > 0 && countingThisNumberSet <= M+1)
		        	{
		        		numberBulbOccurencesCollection.add(numberOfOccurences);
		        		countingThisNumberSet ++;
		        		numberOfOccurences = 0;
		        		rows = 0;
		        		columns= 0;
		        		
		        		if (countingThisNumberSet == M+1)
		        		{
		        			countingThisNumberSet = -1;
		        			numberOfOccurences = 0;
		        			rows = 0;
		        			columns= 0;
		        		}
		        	}
		        	
		        	if(rows == (N - 1) && columns == 2 && countingThisNumberSet < 0 && countingThisNumberSet > (M+1)*(-1))
		        	{
		        		numberBulbOccurencesCollection.add(numberOfOccurences);
		        		countingThisNumberSet --;
		        		numberOfOccurences = 0;
		        		rows = 0;
		        		columns= 0;
		        	}
		        }
		    }
			
			
			
			int firstNumberInBulbsArrayList = bulbs.get(0).get(0);
			if (firstNumberInBulbsArrayList < 0)
			{
				firstNumberInBulbsArrayList = (firstNumberInBulbsArrayList * -1) + M;
			}
			
			int incorrectNumberOfOccurences = numberBulbOccurencesCollection.get(firstNumberInBulbsArrayList - 1);
			incorrectNumberOfOccurences += 1;
			numberBulbOccurencesCollection.set(firstNumberInBulbsArrayList - 1, incorrectNumberOfOccurences);
			
			
			
			for(int l = 0; l < M; l++)
			{
				if(numberBulbOccurencesCollection.get(l) > (numberBulbOccurencesCollection.get(l + M)))
				{
					currentIndicator.add(true);
				}
				else if (numberBulbOccurencesCollection.get(l) < (numberBulbOccurencesCollection.get(l + M)))
				{
					currentIndicator.add(false);
				}
				else if (numberBulbOccurencesCollection.get(l) == (numberBulbOccurencesCollection.get(l + M)))
				{
					if (numberBulbOccurencesCollection.get(l) == 0 && numberBulbOccurencesCollection.get(l + M) == 0)
					{
						currentIndicator.add(false);
					}
					else
					{
						
						int randomBooleanInt = (int) ( Math.random() * 2 + 1);

						if (randomBooleanInt == 1)
						{
							currentIndicator.add(true);
						}
						else if (randomBooleanInt == 2)
						{
							currentIndicator.add(false);
						}
						
						
						for(searchNumber = 1; searchNumber <= M; searchNumber++)
						{
							for(int arrayIndex = 0; arrayIndex < bulbs.size(); arrayIndex++)
							{
								numberFoundBoolean = (bulbs.get(arrayIndex).contains(searchNumber));
								if(numberFoundBoolean == true)
								{
									arrayIndexNumberIsLocated.add(arrayIndex);
								}
							}
							
							arrayIndexNumberIsLocated.clear();
						}
					}
				}
			}
		 return currentIndicator;
	 }
}
