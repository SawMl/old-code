import java.util.ArrayList;

public class StudentSolver {
    
    private static ArrayList<Boolean> finalCircuitPanel;

    
    private static ArrayList<ArrayList<Integer>> listOfBulbs;

    
    private static int numberOfCircuits;

    
    private static boolean done = false;

    
    public static void main(String[] args) {
        
        int n = 25;
        int m = 20;
        ArrayList<ArrayList<Integer>> bulbList = new ArrayList<>();

        
        for (int i = 0; i < n; i++) {
            bulbList.add(new ArrayList<>());
            for (int j = 0; j < 3; j++) {
                int rand = (int) (Math.random() * 2*m - m);

                
                if (rand == 0) {
                    rand = 1;
                }

                
                bulbList.get(i).add(rand);
            }
        }

        
        System.out.println(bulbList);

        
        ArrayList<Boolean> answer = solve(bulbList);

        
        System.out.println(answer);
    }

    
    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
        return borderlands(bulbs);
    }

    
    public static ArrayList<Boolean> borderlands(ArrayList<ArrayList<Integer>> bulbs) {
        
        listOfBulbs = bulbs;

        
        numberOfCircuits = findNumberOfCircuits(listOfBulbs);

        
        ArrayList<Boolean> circuitPanel = new ArrayList<>();

        
        for (int a = 0; a < numberOfCircuits; a++) {
            circuitPanel.add(false);
        }

        
        generateCircuitPanel(circuitPanel, 0);

        
        return finalCircuitPanel;
    }

    
    public static int findNumberOfCircuits(ArrayList<ArrayList<Integer>> bulbs) {
        
        int numberOfCircuits = -1;

        
        for (int i = 0; i < bulbs.size(); i++) {
            for (int j = 0; j < 3; j++) {
                
                if (numberOfCircuits < Math.abs(bulbs.get(i).get(j))) {
                    numberOfCircuits = Math.abs(bulbs.get(i).get(j));
                }
            }
        }
        
        return numberOfCircuits;
    }

    
    public static void generateCircuitPanel(ArrayList<Boolean> circuitPanel, int index) {
        
        if (index < numberOfCircuits) {
            generateCircuitPanel(circuitPanel, index + 1);

            
            if (done) {
                return;
            }

            
            circuitPanel.set(index, true);
            generateCircuitPanel(circuitPanel, index + 1);
        }

        
        if (done) {
            return;
        }

        
        if (validateCircuitPanel(circuitPanel)) {
            finalCircuitPanel = circuitPanel;
            done = true;
        }
    }

    
    public static boolean validateCircuitPanel(ArrayList<Boolean> circuitP) {
        
        boolean validity;

        
        for (int k = 0; k < listOfBulbs.size(); k++) {
            
            validity = false;

            
            for (int j = 0; j < 3; j++) {
                
                if (listOfBulbs.get(k).get(j) < 0 && !circuitP.get(Math.abs(listOfBulbs.get(k).get(j)) - 1)) {
                    validity = true;

                
                } else if (listOfBulbs.get(k).get(j) > 0 && circuitP.get(Math.abs(listOfBulbs.get(k).get(j) - 1))) {
                    validity = true;
                }

                
            }

            
            if (!validity) {
                return false;
            }
        }

        
        return true;
    }

}
