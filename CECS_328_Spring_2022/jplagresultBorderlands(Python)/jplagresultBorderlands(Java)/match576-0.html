import java.util.*;

public class StudentSolver {
    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {



        System.out.println("Question (" + bulbs.size() + ")");
        int size = findSize(bulbs);
        boolean[] solution = new boolean[size];













        ArrayList<Integer> done = new ArrayList<Integer>();
        ArrayList<Integer> doneRows = new ArrayList<Integer>();
        boolean st = true;
        TreeMap<Integer, Integer> data = new TreeMap<Integer, Integer>();
        for (int i = 1; i <= size; i++) {

            data.put(-i, 0);
            data.put(i, 0);
            for (int j = 0; j < bulbs.size(); j++) {
                if (bulbs.get(j).contains(i)) {
                    data.put(i, data.get(i) + 1);
                }
                if (bulbs.get(j).contains(-i)) {
                    data.put(-i, data.get(-i) + 1);
                }
            }

        }

        while (st) {

            System.out.println(data);
            int high = Collections.max(data.values());
            Set<Map.Entry<Integer, Integer>> temp = data.entrySet();
            for (Map.Entry<Integer, Integer> entry : temp) {
                if (entry.getValue().equals(high)) {
                    if (entry.getKey() < 0) {
                        solution[-entry.getKey() - 1] = false;
                    } else if (entry.getKey() > 0) {
                        solution[entry.getKey() - 1] = true;
                    }
                    data.put(entry.getKey(), 0);
                    data.put(-entry.getKey(), 0);
                    done.add(Math.abs(entry.getKey()));

                }
            }
            high = Collections.max(data.values());

            if (high == 0) {
                st = false;
            }
        }




























        



        ArrayList<Boolean> answers = new ArrayList<Boolean>();
        for (int i = 0; i < solution.length; i++) {
            System.out.print(i+1 + "=" + solution[i] + ",  ");
            answers.add(solution[i]);
        }
        return answers;
    }


    public static int findSize(ArrayList<ArrayList<Integer>> bulbs) {
        int max = 0;
        int min = 0;
        for (int i = 0; i < bulbs.size(); i++) {
            int tempMax = Collections.max(bulbs.get(i));
            int tempMin = Collections.min(bulbs.get(i));
            if (tempMax > max) {
                max = tempMax;
            }
            if (tempMin < min) {
                min = tempMin;
            }
        }
        if (min < 0) {
            min = -min;
        }
        return Math.max(max, min);
    }

    public static ArrayList<Integer> aSingleArray(ArrayList<ArrayList<Integer>> bulbs) {
        ArrayList<Integer> list = new ArrayList<Integer>();
        for (int i = 0; i < bulbs.size(); i++) {
            for (int j = 0; j < bulbs.get(i).size(); j++) {
                list.add(bulbs.get(i).get(j));
            }
        }
        return list;
    }

    public static ArrayList<ArrayList<Integer>> sortNumbers(ArrayList<Integer> list) {
        ArrayList<ArrayList<Integer>> array = new ArrayList<ArrayList<Integer>>();
        ArrayList<Integer> onlyPos = new ArrayList<Integer>();
        ArrayList<Integer> onlyNeg = new ArrayList<Integer>();
        ArrayList<Integer> both = new ArrayList<Integer>();
        for (int i = 0; i < list.size(); i++) {
            if (list.get(i) > 0 && !onlyPos.contains(list.get(i)) && !list.contains(-list.get(i))) {
                onlyPos.add(list.get(i));
            } else if (list.get(i) < 0 && !onlyNeg.contains(-list.get(i)) && !list.contains(-list.get(i))) {
                onlyNeg.add(-list.get(i));
            } else if (list.get(i) > 0 && !both.contains(list.get(i)) && list.contains(-list.get(i))) {
                both.add(list.get(i));
            }
        }
        array.add(onlyPos);
        array.add(onlyNeg);
        array.add(both);
        return array;
    }

    public static void setPositiveValues(ArrayList<Integer> list, boolean[] solution) {
        for (int i = 0; i < list.size(); i++) {
            solution[list.get(i) - 1] = true;
        }
    }

    public static void setNegativeValues(ArrayList<Integer> list, boolean[] solution) {
        for (int i = 0; i < list.size(); i++) {
            if (list.get(i) - 1 < 0) {
                list.set(i, -list.get(i));
            }
            solution[list.get(i) - 1] = false;
        }
    }



}
