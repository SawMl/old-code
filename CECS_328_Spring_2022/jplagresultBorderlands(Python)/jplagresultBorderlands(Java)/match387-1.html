	public static ArrayList<Boolean> createDefaultAnswer(ArrayList<ArrayList<Integer>> bulbs) {
		ArrayList<Boolean> answer = new ArrayList<>();
		
		int max = 1;
		int t0,t1,t2;
		for(int i = 0; i < bulbs.size(); i ++) {
			t0 = Math.abs(bulbs.get(i).get(0));
			t1 = Math.abs(bulbs.get(i).get(1));
			t2 = Math.abs(bulbs.get(i).get(2));
			if(t0 > max)
				max = t0;
			if(t1 > max)
				max = t1;
			if(t2 > max)
				max = t2;
		}
		for(int i = 0; i < max; i++) {
			answer.add(false);
		}
		return answer;

	}
	public static ArrayList<ArrayList<Integer>> createBelongs(ArrayList<ArrayList<Integer>> bulbs, int answerSize) {
		ArrayList<ArrayList<Integer>> belongs = new ArrayList<>();
		for(int i = 0; i < answerSize; i++) {
			belongs.add(new ArrayList<Integer>());
		}
		
		for(int i = 0; i < bulbs.size(); i++) {
			int first = Math.abs(bulbs.get(i).get(0));
			int second = Math.abs(bulbs.get(i)
.get(1));
			int third = Math.abs(bulbs.get(i).get(2));
			belongs.get(first-1).add(i); 
			if(first != second)
				belongs.get(second-1).add(i);
			if(third != second && third != first)
				belongs.get(third-1).add(i);
		}

		return belongs;
	}
	public static Boolean bulbDone(ArrayList<ArrayList<Integer>> belongs, ArrayList<Boolean> bulbDone, ArrayList<Integer> guaranteed, int check) {

		for(int i = 0; i < belongs.get(check).size(); i++) {
			if(!guaranteed.contains(belongs.get(check).get(i))) {
				return false;
			}

		}
		return true;
	}
	public static Boolean allBulbsDone(ArrayList<Boolean> bulbDone) {
		for(int i = 0; i < bulbDone.size(); i++) {
			if(bulbDone.get(i) == false) {
				return false;
			}
		}
		return true;
	}
