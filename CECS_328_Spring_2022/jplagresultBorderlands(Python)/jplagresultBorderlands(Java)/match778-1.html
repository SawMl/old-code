import java.util.ArrayList;
import java.util.Random;

class StudentSolver {

    public static void main(String[] args) {

    }

    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs){
        ArrayList<Boolean> ans = new ArrayList<Boolean>();
        ArrayList<Integer> checkDup = new ArrayList<>();

        
        int maxNum = 0;
        int positive = 0;
        maxNum = bulbs.get(0).get(0);
        for (int i = 0; i < bulbs.size(); i++)
        {
            for (int j = 0; j < 
bulbs.get(i).size(); j++)
            {
                if(maxNum < bulbs.get(i).get(j)){
                    maxNum = bulbs.get(i).get(j);
                }
                if(bulbs.get(i).get(j)<0){
                    positive = bulbs.get(i).get(j) * -1;
                    if(maxNum < positive)
{
                        maxNum = positive;
                    }
                }
            }
        }
        
int numM = maxNum;

        int currNum=0;
        iterateAgain:
        for (int i = 0; i < bulbs.size(); i++)
        {
            breakloop:
            for (int j = 0; j < bulbs.get(i).size(); j++)
            {
                currNum = bulbs.get(i).get(j);
                if(i>=1 && i<
bulbs.size()-1){
                    for(int k=0; k < checkDup.size();k++){
                        if(bulbs.get(i).contains(checkDup.get(k))){
                            break breakloop;
                        }
                    }
                }
                if(!checkDup.contains(currNum)) {
                    checkDup.add(currNum);
                    if(checkDup.contains(currNum*-1)){
                        checkDup.remove(checkDup.size()-1);
                        if(!checkDup.contains(bulbs.get(i).get(0))&&!checkDup.contains(bulbs.get(i).get(1))&&!checkDup.contains(bulbs.get(i).get(2))){
                            Random random = new Random();
                            int randNum = random.nextInt((2 - 0) + 1) + 0;
                            int unique = bulbs.get(i).get(randNum);
                            if(checkDup.contains(unique*-1)){
                                int indexDup = checkDup.lastIndexOf(unique*-1);
                                System.out.println(unique);
                                checkDup.remove(indexDup);
                                checkDup.add(unique);
                                break iterateAgain;
                            }
                        }
                    }
                    else{
                        break;
                    }
                }
                else if(checkDup.contains(currNum)){
                    break;
                }
            }
        }

        for(int i = 1; i <= numM;i++){
            if(checkDup.contains(i)){
                ans.add(true);
            }
            else if(checkDup.contains(i*-1)){
                ans.add(false);
            }
            else{
                ans.add(true);
            }
        }

        System.out.println(ans);
        return ans;

    }
}
