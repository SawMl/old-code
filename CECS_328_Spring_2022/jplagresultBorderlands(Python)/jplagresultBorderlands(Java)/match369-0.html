    /**
     * return ArrayList<Boolean> of m values from 0...m
     * **/
    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
        int row = 0; 
        int temp = 0; 
        int absVal = 0; 
        
        int mSize = getMSize(bulbs); 
        int[] m = new int[mSize]; 
        Arrays.fill(m, -1); 
        
        
        HashMap<Integer, Integer> hashInt = new HashMap<>(); 

        for (int i = 0 ; i < bulbs.size(); i++) {
            row = 0; 
            absVal = Math.abs(bulbs.get(i).get(row)); 
            temp = 0; 

            if (bulbs.get(i).get(row) > 0) {
                temp = 1; 
            } else {
                temp = 0; 
            }

            
            if (m[absVal - 1] == -1 || m[absVal - 1] == temp) {
                hashInt.put(i, bulbs.get(i).get(row)); 
                
                m[absVal - 1] = temp; 

            } else {
                
                row = 0; 
                while (row + 1 < 3) {
                    row += 1; 
                    absVal = Math.abs(bulbs.get(i).get(row)); 
                    if (bulbs.get(i).get(row) > 0) {
                        temp = 1; 
                    } else {
                        temp = 0; 
                    }
                    if (m[absVal - 1] == -1 || m[absVal - 1] == temp) {
                        hashInt.put(i, bulbs.get(i).get(row)); 
                        m[absVal - 1] = temp; 
                        break; 
                    } 
                }

                
                int valInv = bulbs.get(i).get(row)*-1; 
                if (hashInt.containsValue(valInv)) {
                    
                    hashInt.put(i, bulbs.get(i).get(row)); 
                    
                    int mInd = 0; 
                    for (Integer key : hashInt.keySet()) {
                        if (key == valInv) {
                            
                            for (int n = 1; n < 3; n++) {
                                absVal = Math.abs(bulbs.get(mInd).get(n)); 
                                if (bulbs.get(mInd).get(n) > 0) {
                                    temp = 1; 
                                } else {
                                    temp = 0; 
                                }
                                if (m[absVal - 1] == -1 || m[absVal - 1] == temp) {
                                    hashInt.put(mInd, bulbs.get(mInd).get(n)); 
                                    m[absVal - 1] = temp; 
                                    break; 
                                } 
                            }
                        }

                    }
                }
            }
        }
        
        
        
        

        
        

        
        

        
        return getBoolArrList(m); 
    }

    public static int getMSize(ArrayList<ArrayList<Integer>> bulbs) {
        int max = 0; 
        int min = 0; 
        for (int i = 0; i < bulbs.size(); i++) {
            int rowMax = Collections.max(bulbs.get(i)); 
            if (rowMax > max) {
                max = rowMax; 
            }
            
        }
        return max; 
    }

    public static ArrayList<Boolean> getBoolArrList(int[] m) {
        ArrayList<Boolean> fin = new ArrayList<>();
        
        for (int val : m) {
            if (val == 1) {
                fin.add(true);
            } else if (val == 0) {
                fin.add(false); 
            } else {
                fin.add(false); 
            }
        }

        System.out.println(fin.toString());

        return fin;
    }
