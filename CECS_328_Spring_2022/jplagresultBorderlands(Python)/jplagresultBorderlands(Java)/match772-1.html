import java.util.ArrayList;
import java.lang.Math;

public class StudentSolver {
  public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs) {
    
    ArrayList<Boolean> solution = new ArrayList<Boolean>();

    
int maxNum = 0;
    
    for (int i = 0; i < bulbs.size(); i++) {
      if (findMax(bulbs.get(i)) > maxNum) {
        maxNum = findMax(bulbs.get(i));
      }
    }
    System.out.println("max of arraylists: " + maxNum);

    boolean on = false;
    
    for (int k = 0; k < maxNum; k++) {
      on = false;
      if (Math.random()*2 > 1) {
        on = true;
      }  
      
solution.add(on);
    }
    return solution;
  }


  public static int findMax(ArrayList<Integer> row) {
    int max = 0;
  
    for (int i = 0; i < 3; i++) {
      if (row.get(i) < 0) {
        if (row.get(i)*-1 > max) {
          max = row.get(i)*-1;
        }
      }
      else if (row.get(i) > max) {
        max = row.get(i);
      }
    }
  
    return max;
  }
}
