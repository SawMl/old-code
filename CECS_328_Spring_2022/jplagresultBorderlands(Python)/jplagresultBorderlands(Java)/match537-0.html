import java.util.*;
import java.lang.Math;

public class StudentSolver {
    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs){
        int rows = bulbs.size(), mid = rows / 2, max = 0;

        
        for(int i = 0; i < rows; i++){
            for(int j = 0; j < 3; j++) {
                if (Math.abs(i) > max)
                    max = Math.abs(i);
            }
        }

        ArrayList<Boolean> switches = new ArrayList<>(max);
        
        boolean[] test = new boolean[max];
        
        int[] check = new int[max];

        while(Arrays.asList(test).contains(false)){
            
            for(int i = 0; i < rows; i++){
                
                ArrayList<Integer> wireset = bulbs.get(i);
                
                for(int j : wireset){
                    int ab = Math.abs(j);
                    
                    if(test[i] = false){
                        if(check[j - 1] != 1){
                            if(j < 0) {
                                switches.set(j - 1, false);
                                check[j - 1] = 1;
                            }
                            else {
                                switches.set(j - 1, true);
                                check[j - 1] = 1;
                            }
                        }
                        test[i] = true;
                    }
                }
            }

        }
        return switches;
    }
}
