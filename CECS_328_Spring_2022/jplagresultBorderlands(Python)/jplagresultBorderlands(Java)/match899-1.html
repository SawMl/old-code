import java.util.ArrayList;
import java.util.*;
public class StudentSolver{
     public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs){
         ArrayList<Boolean> aa = new ArrayList<>();
         ArrayList<Boolean> gg= new ArrayList<>();
        ArrayList<Boolean> ans = new ArrayList<>();
        int k =0;
       int max = bulbs.get(1).get(1);
        int min= bulbs.get(1).get(1);
         for(int i=0;i<bulbs.size();i++){
             if (i==0){
               for(int j=1; j<bulbs.get(i).size();j++){
                
if (max<bulbs.get(i).get(j)){
                  max =  bulbs.get(i).get(j); 
                }
               }
             }
             if (i>0){
              for(int j=0; j<bulbs.get(i).size();j++)
{
                if (max<bulbs.get(i).get(j)){
                  max =  bulbs.get(i).get(j); 
                }
              }
             }
         }
      if(max>0){
          int z=0;
         while(z<max){
             aa.add(false);
             z++;
         }
         while(k<max){
             if(max-min>0){
           for ( int j=0;j<bulbs.size();j++){
               int num=k+1;
               if(bulbs.get(j).contains(-num)){
                   aa.set(k,false);
                
                }
               else if(bulbs.get(j).contains(num)){
                   aa.set(k,true);
                
                }
            }
            k++;
         }
        }
        }         
        return aa; 
}
}
