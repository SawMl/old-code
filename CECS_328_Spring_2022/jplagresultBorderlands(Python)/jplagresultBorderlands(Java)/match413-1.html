import java.util.Scanner;
import java.util.ArrayList;
import java.lang.Math;
import java.io.*;

class StudentSolver{
  public static void main(String[] args)
  {

    int nums = 0;
    int n = 0;

    ArrayList<ArrayList<Integer>> numbers = new ArrayList<ArrayList<Integer>>();

    Scanner input = new Scanner(System.in);
    n = input.nextInt();

    for(int x = 0; x < n; x++){
      ArrayList<Integer> values = new ArrayList<Integer>();
      for(int y = 0; y < 3; y++){
        nums = input.nextInt();
        values.add(nums);
      }
      numbers.add(values);
    }

    
solve(numbers);
  }

  public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs){

    int m = greatestValue(bulbs);

    ArrayList<Boolean> tempWireSet = new ArrayList<Boolean>(m);

    for(int x = 0; x < m; x++){
        tempWireSet.add(false);
    }

    ArrayList<Boolean> wireSet = new ArrayList<Boolean>(m);

    if(bulbs.size() == 1){
      int val0 = -m;
      int val1 = 0;
      while(val0 <= m){
        if(bulbs.get(val1).contains(val0) && val0 > 0){
          tempWireSet.add(val0, true);
        }else if(
bulbs.get(val1).contains(val0) && val0 < 0){
          tempWireSet.add(Math.abs(val0), false);
        }else{
          tempWireSet.add(Math.abs(val0),false);
        }
        val0++;
      }
      for(int x = 0; x < m; x++){ wireSet.add(tempWireSet.get(x)); }

    }


    int val = 1;
    int i = 0;
    int countNeg = 0;
    int countPos = 0;
    while(val <= m){
      for(int x = 0; x < bulbs.size(); x++){
        if(bulbs.get(x).contains(val)){
          System.out.println("Pos: " + val );
          countPos++;
        }
        
if(bulbs.get(x).contains(-val)){
          System.out.println("Neg: " + -val);
          countNeg++;
        }
      }
      if(countPos > countNeg){
        System.out.println("Setting " + val + " true");
        tempWireSet.add(val-1, true);
      }else if(countNeg >= countPos){
        System.out.println("Setting " + -val + " false");
        tempWireSet.add(Math.abs(val+1), false);
      }
      val++;
      countPos = 0;
      countNeg = 0;
    }
    for(int x = 0; x < m; x++){ wireSet.add(tempWireSet.get(x)); }
    System.out.println(wireSet.size());
    
for(int x = 0; x < m; x++){ System.out.println(wireSet.get(x)); }
    return wireSet;
  }



  public static int greatestValue(ArrayList<ArrayList<Integer>> bulbs){
    int mVal = 1;
    for(int i = 0; i < bulbs.size(); i++){ 
      for(int j = 0; j < bulbs.get(i).size(); j++){
        if(mVal <= Math.abs(bulbs.get(i).get(j))){
          mVal = Math.abs(
bulbs.get(i).get(j));
        }
      }
    }
    return mVal;
  }

}
