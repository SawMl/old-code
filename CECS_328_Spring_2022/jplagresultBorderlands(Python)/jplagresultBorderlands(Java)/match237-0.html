import java.util.ArrayList;

    public static ArrayList<Boolean> solve(ArrayList<Integer> bulbs) { 
        ArrayList<Boolean> temp = new ArrayList<>();
        ArrayList<Integer> temp1 = new ArrayList<>();
        Integer mult = 0;
        for(int i = 0; i < bulbs.size(); i++) {
            
            temp1.add(Math.abs(bulbs.get(i)));
        }
        Integer num = 0;
        for(int i = 0; i < temp1.size() - 1;i++) {
            if(temp1.get(i) > temp1.get(i + 1)) {
                num = temp1.get(i);
            }
            else {
                num = temp1.get(i + 1);
            }
        }
        for(int i = 0; i < num; i++) {
            temp.add(false);
        }
        
        for(int i = 0; i < bulbs.size()-2; i = i + 3) {
            mult = 0;
            mult = bulbs.get(i) * bulbs.get(i + 1) * bulbs.get(i + 2);
            System.out.println(bulbs.get(i) + "*" + bulbs.get(i + 1) + "*" + bulbs.get(i + 2) + " = " + mult);














            if(bulbs.get(i) > 0 && bulbs.get(i + 1) > 0 && bulbs.get(i + 2) > 0) {
                
                temp.set((Math.abs(bulbs.get(i)) - 1), true);
                mult = 0;
                
                






            }
            
        }
        return temp;
        
    }
