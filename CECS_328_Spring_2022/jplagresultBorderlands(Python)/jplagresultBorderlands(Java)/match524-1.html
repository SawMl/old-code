import java.util.ArrayList;

public class StudentSolver {

  
public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
    int m = 0;
    for (int i = 0; i < bulbs.size(); i++) {
      for (int j = 0; j < bulbs.get(i).size(); j++) {
        int positive = 0;
        if (bulbs.get(i).get(j) < 0) {
          positive = bulbs.get(i).get(j) * -1;
        } else {
          positive = bulbs.get(i).get(j);
        }
        if (positive > m) {
          m = positive;
        }
      }
    }
    ArrayList<Boolean> bool = new ArrayList<>(m);
    for (int i = 0; i < m; i++) {
      bool.add(false);
    }

    int bulb = 1;
    while (bulb <= m) 
{
      for (int i = 0; i < bulbs.size(); i++) {
        for (int j = 0; j < bulbs.get(i).size(); j++) {
          if (bulbs.get(i).get(j) % bulb == 0) {
            if (bulbs.get(i).get(j) < 0) {
              bool.set(bulb - 1, false);
              break;
            } else {
              bool.set(bulb - 1, true);
              break;
            }
          }
        }
      }
      bulb++;
    }
    return bool;
  }
}
