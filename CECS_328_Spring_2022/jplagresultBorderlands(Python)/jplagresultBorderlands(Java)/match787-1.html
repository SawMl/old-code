    public static boolean updatePossible(ArrayList<ArrayList<Integer>> bulbs,HashMap<Integer, ArrayList<Integer>> wireRows, ArrayList<Boolean> wires,int[] bulbsOn,ArrayList<Integer> offRows, HashMap<Integer, Boolean> offHash, int affected, int goalRow)
    {
        ArrayList<Integer> curHash = wireRows.get(affected);
        
        boolean removed = false;
        for (int i = 0; i < curHash.size(); i++)
        {
            int rowNum = curHash.get(i);
            ArrayList<Integer> curRow = bulbs.get(rowNum);
            int curOn = bulbsOn[i];
            int newOn = 0;
            for (int v = 0; v < 3; v++) {
                int val = curRow.get(v);
                int abval = Math.abs(val);
                if (wires.get(abval-1) == true && val > 0) {
                    newOn += 1;
                } else if (wires.get(abval-1) == false && val < 0) {
                    newOn += 1;
                }
            }
            
            bulbsOn[i] = newOn;
            if (newOn > 0 && curOn == 0) {
                
                if (rowNum==goalRow){
                    removed = true;
                }
                
                if (offHash.containsKey(rowNum)) {
                    
                    offRows.remove(offRows.indexOf(rowNum));
                    offHash.remove(rowNum);
                } else {
                    
                    
                }
            } else if (newOn == 0 && curOn > 0) {
                
                
                if (!offHash.containsKey(rowNum)) {
                    offRows.add(rowNum);
                    offHash.put(rowNum,true);
                    
                }
            }

        }
        return removed;
    }

    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
        ArrayList<Boolean> wires = new ArrayList<Boolean>();
        HashMap<Integer, ArrayList<Integer>> wireRows = new HashMap<Integer, ArrayList<Integer>>();
        ArrayList<Integer> offRows = new ArrayList<Integer>();
        HashMap<Integer, Boolean> offHash = new HashMap<Integer, Boolean>();

        int[] bulbsOn = new int[bulbs.size()];
        int[] curCombo = new int[bulbs.size()];
        int wireRange = 0;
        int removeCount = 0;
        for (int row = bulbs.size()-1; row >= 0; row--) {
            ArrayList<Integer> curRow = bulbs.get(row);
            int v1 = curRow.get(0);
            int v2 = curRow.get(1);
            int v3 = curRow.get(2);
            if (v1 == -v2 || v1 == -v3 || v2 == -v3) {
                bulbs.remove(row);
                removeCount++;
            }
        }
        for (int row = 0; row < bulbs.size(); row++) {
            ArrayList<Integer> curRow = bulbs.get(row);
            for (int column = 0; column < 3; column++) {
                int trueVal = bulbs.get(row).get(column);
                int curVal = Math.abs(trueVal);
                if (trueVal < 0)
                {
                    bulbsOn[row] += 1;
                }
                wireRange = Math.max(wireRange, curVal);
                if (wireRows.containsKey(curVal)) {
                    wireRows.get(curVal).add(row);
                } else {
                    wireRows.put(curVal,new ArrayList<Integer>());
                    wireRows.get(curVal).add(row);
                }
            }
            if (bulbsOn[row] == 0) {
                offRows.add(row);
                offHash.put(row,true);

        }

        {
            wires.add(false);

                    wires.set(abs1 - 1, v1);
                    wires.set(abs2 - 1, v2);
                    wires.set(abs3 - 1, v3);
                    
                    
                    removed1 = updatePossible(bulbs, wireRows, wires, bulbsOn, offRows, offHash, abs1,rowNum);
