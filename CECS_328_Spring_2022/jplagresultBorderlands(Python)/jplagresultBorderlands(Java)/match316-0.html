import java.util.ArrayList;
import java.util.Random;
import java.util.Scanner;
public class StudentSolver {










	public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs){
		Random rand = new Random();



		
		
		int N= 10;
		int M= 14;
		int low = M;
		int up = M+5;
		int randint = rand.nextInt(1);
		bulbs = new ArrayList<ArrayList<Integer>>(N);
		
		
		for (ArrayList<Integer> i:bulbs) {
			if (randint == 1) {
				i.add((int)Math.floor(Math.random()*(low-up+1)+low)*-1);
			}else {
				i.add((int)Math.floor(Math.random()*(low-up+1)+low));
			}
			if (randint == 1) {
				i.add((int)Math.floor(Math.random()*(low-up+1)+low)*-1);
			}else {
				i.add((int)Math.floor(Math.random()*(low-up+1)+low));
			}
			if (randint == 1) {
				i.add((int)Math.floor(Math.random()*(low-up+1)+low)*-1);
			}else {
				i.add((int)Math.floor(Math.random()*(low-up+1)+low));
			}
		}
		
		
		ArrayList<Boolean>ans = new ArrayList<Boolean>();
		for (int i = 0; i<M;i++) {
			int rando = rand.nextInt(1);
			if (rando == 1) {
				ans.add(true);
			}else {
				ans.add(false);
			}
		}
		return ans;
	}
	

}
