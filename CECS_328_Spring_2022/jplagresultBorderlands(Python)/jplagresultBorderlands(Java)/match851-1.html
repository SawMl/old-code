    public static HashMap<Integer, Boolean> randomHash(ArrayList<ArrayList<Integer>> bulbs){
        Random rand = new Random();
        HashMap<Integer, Boolean> hashMap = new HashMap<>();
        for (int i = 1; i <= maxnum(bulbs); i++) {
            hashMap.put(i, rand.nextBoolean());
        }
        hashMap.put(8, false);
        hashMap.put(2, false);
        return hashMap;
    }























    public static int maxnum(ArrayList<ArrayList<Integer>> bulbs){
        int maxim = Integer.MIN_VALUE;
        for (ArrayList<Integer> bulb : bulbs) {
            if (maxim < Math.abs(Collections.max(bulb))) {
                maxim = Math.abs(Collections.max(bulb));
            }
            if (maxim < Math.abs(Collections.min(bulb))) {
                maxim = Math.abs(Collections.min(bulb));
            }
        }
        return Math.abs(maxim);
    }
