import java.util.*;

public class StudentSolver 
{
	public static int findM(ArrayList<ArrayList<Integer>>bulbs)
	{
		int maxMValue = 0;
		for(int row = 0; row < bulbs.size(); 
row++)
		{
			for(int element = 0; element < bulbs.get(row).size(); element++)
			{
				if(bulbs.get(row).get(element) >= maxMValue)
				{
					maxMValue = bulbs.get(row).get(element);

				}
				
			}

		}
		return maxMValue;
	}
	
	
	public static boolean lightsVerify (ArrayList<ArrayList<Integer>>bulbs,ArrayList<Boolean> ans)
	{
		ArrayList<Boolean> container = new ArrayList<Boolean>();
		boolean verify = true;
		
		for(int i = 0; i < ans.size(); i++)
		{
			for(int row = 0; row < bulbs.size(); row++)

			{
				for(int element = 0; element < bulbs.get(row).size(); element++)
				{
					
					if(bulbs.get(row).get(element) == i+1)
					{
						if(ans.get(i) == true)
						{
							bulbs.get(row).set(element, 1);
						}
						else
						{
							bulbs.get(row).set(element, 0);
						}
					}
					
					
					if(bulbs.get(row).get(element) == -(i+1))
					{
						if(ans.get(i) == true)
						{
							bulbs.get(row).set(element, 0);
						}
						else
						{
							bulbs.get(row).set(element, 1);
						}
					}
					
				}
	
			}
		}
		
	
		
		for(int row = 0; row < bulbs.size(); row++)
		{
			verify = bulbs.get(row).contains(1);
			container.add(verify);
		}
		
		if(container.contains(false))
			verify = false;
		
		
		return verify;
	}


    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs)
    {
    	ArrayList<Boolean> answer = new ArrayList<Boolean>();
    	ArrayList<Integer> used = new ArrayList<>();
  
    	
    	int mValue = findM(bulbs);

    		
    		for(int i = 0; i<mValue; i++)
			 {
				 answer.add(false);
			 }
    		
    		
    			for(int row = 0; row < bulbs.size(); row++)
    			{
    				int element = bulbs.get(row).get(0);
    				if(element > 0 && used.contains(element) == false)
    				{
    					answer.set(element -1 , true);
    					used.add(element);
    				}
    				else if(element < 0 && used.contains(element*-1) == false)
    				{
    					used.add(element*-1);
    				}    				
    				else if(element > 0 && used.contains(element) == true)
    				{
    					element = bulbs.get(row).get(1);
    					if(element > 0 && used.contains(element) == false)
    					{
    						answer.set(element -1 , true);
        					used.add(element);
    					}else if(element < 0 && used.contains(element*-1) == false)
    					{
    						used.add(element *-1);
    					}else if(bulbs.get(row).get(2) > 0 && used.contains(element) == false)
    					{
    						answer.set(bulbs.get(row).get(2), true);
        					used.add(element);
    					}else if(bulbs.get(row).get(2) < 0 &&
 used.contains(element*-1) == false)
    					{
    						used.add(bulbs.get(row).get(2) *-1);
    					}
    				}
    			}
    				
    				   	
    	return answer;
    	
    }
}
