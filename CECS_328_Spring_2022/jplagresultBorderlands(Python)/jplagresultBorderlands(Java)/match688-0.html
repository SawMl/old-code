import java.util.*;
import java.lang.*;

public class StudentSolver {
    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
        ArrayList<Boolean> list = new ArrayList<Boolean>();
        int M = 0;
        ArrayList<Integer> occurence = new ArrayList<Integer>();

        for (int i = 0; i < bulbs.size(); i++) {
            for (int j = 0; j < 3; j++) {
                M = (Math.abs(bulbs.get(i).get(j)) > M) ? Math.abs(bulbs.get(i).get(j)) : M;
            }
        }

        for (int i = 0; i < M; i++) {
            list.add(false);
            occurence.add(0);
        }
        System.out.println(M);
        System.out.println(occurence.size());

        for (int i = 0; i < bulbs.size(); i++) {
            for (int j = 0; j < bulbs.get(0).size(); j++) {
                if (bulbs.get(i).get(j) < 0)
                    occurence.set((Math.abs(bulbs.get(i).get(j))-1), (occurence.get(Math.abs(bulbs.get(i).get(j))-1) -1));
                else
                    occurence.set((Math.abs(bulbs.get(i).get(j))-1), (occurence.get(Math.abs(bulbs.get(i).get(j))-1) +1));
            }
        }

        for (int i = 0; i < occurence.size(); i++) {
            if (occurence.get(i) > 0)
                list.set(i, true);
        }
        return list;
    }
}
