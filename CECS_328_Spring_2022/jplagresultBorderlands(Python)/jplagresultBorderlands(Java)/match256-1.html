import java.util.ArrayList;

class StudentSolver {

   public static int size(ArrayList<ArrayList<Integer>> bulbs) {

    int max = 0;
    
ArrayList<ArrayList<Integer>> temp = bulbs;
    for(int i = 0; i < temp.size(); i++) {
      for(int j = 0; j < temp.get(i).size(); j++) {
        if(max < temp.get(i).get(j) || -(max) > temp.get(i).get(j)) 
{
          max = temp.get(i).get(j);
        }
      }
    }
    if(max < 0){
      return -max;
    
}
    return max;
     
  }

  public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {

    ArrayList<Boolean> firHalf = new ArrayList<Boolean>();
    ArrayList<Boolean> secHalf = new ArrayList<Boolean>();
    ArrayList<Boolean> output = new ArrayList<Boolean>();
    for(int i = 0; i < size(bulbs); i++) {
      firHalf.add(true);
      secHalf.add(true);
      output.add(true);
    }
    
for(int i = 0; i < bulbs.size()/2; i++) {
      for(int j = 0; j < bulbs.get(i).size(); j++) {
        if(0 < bulbs.get(i).get(j)) {
          firHalf.set(bulbs.get(i).get(j) - 1, true);
        }
        else if(0 > bulbs.get(i).get(j)) {
          int pos = -(bulbs.get(i).get(j));
          firHalf.set(pos - 1, false);
        }
      }
    }
    for(int i = bulbs.size()/2; i < bulbs.size(); i++) {
      for(int j = 0; j < bulbs.get(i).size(); j++) {
        if(0 < bulbs.get(i).get(j)) {
          secHalf.set(bulbs.get(i).get(j) - 1, true);
        }
        else if(0 > bulbs.get(i).get(j)) {
          int pos = -(bulbs.get(i).get(j));
          secHalf.set(pos - 1, false);
        }
      }
    }
    for(int i = 0; i < size(bulbs); i++) {
      if(firHalf.get(0) == false && secHalf.get(0) == false) {
        output.set(0, false);
      }
      if(firHalf.get(0) == false && secHalf.get(0) == true) {
        output.set(0, false);
      }
    }
    return output;

  }
}
