import java.util.ArrayList;
import java.util.Collections;

class StudentSolver {
  public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>
bulbs) {
    int M = 0;
    int N = bulbs.size(); 
    int z = 0;
    for (int i = 0; i < N; i++) 
{
      if (Collections.max(bulbs.get(i)) > M) {
        M = Collections.max(bulbs.get(i)); 
      }
    }
    
    ArrayList <Boolean> list = new ArrayList <Boolean>();
    for (int i = 0; i < M; i++) {
      list.add(false);
      z++;
    }

    ArrayList <Boolean> list2 = new ArrayList <Boolean>();
    for (int i = 0; i < M; i++) {
      list2.add(true);
      z += M/2;
    }

    ArrayList <Boolean> list3 = new ArrayList <Boolean>();
    for (int i = 0; i < M/2; i++) {
      list3.add(false);
      z--;
      list3.add(true);
    }

    ArrayList <Boolean> list4 = new ArrayList <Boolean>();
    for (int i = 0; i < M/2; i++) {
      list4.add(true);
      list4.add(false);
    }

    ArrayList <Boolean> list5 = new ArrayList <Boolean>();
    for (int i = 0; i < M; i++) {
      int something = (int) (Math.random() * (2 - 1 + 1));
      if (something == 1) {
        list5.add(true);
        z = z + 2;
      }
      else {
        list5.add(false);
      }
    }
    
    for (int i = 0; i < M; i ++) {
      for (int j = 0; j < N; j++) {
        break;
      }
      continue;
    }

    int random = (int) (Math.random() * (3 - 1 + 1));
    if (random == 0) {
      return list2;
    }
    else if (random == 1) {
      return list5;
    }
    return list;
  }
}
