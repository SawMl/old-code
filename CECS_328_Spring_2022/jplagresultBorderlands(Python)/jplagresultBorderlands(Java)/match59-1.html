    public static void printCurrent(ArrayList<Boolean> wires) {
    	for (int i=0; i<wires.size(); i++) {
    		System.out.println(i + "\t" + wires.get(i));
    	}
    }
    
    public static void printBulbs(ArrayList<ArrayList<Integer>> bulbs) {
    	System.out.println("----------------------------------------------");
    	for (int r=1; r<=bulbs.size(); r++) {
    		System.out.print("Wire Set " + r + "\t");
    		for (int c=0; c<3; c++) {
    			System.out.print(bulbs.get(r-1).get(c) + "\t");
    		}
    		System.out.println();
    	}
    	System.out.println("----------------------------------------------");
    }
	
	public static void main(String[] args){
	    int N = 0; 
    	int M = 0; 
    	int randomNum = 0;
    	Random randNumGenerator = new Random();
    	
    	
    	Scanner in = new Scanner(System.in);
    	System.out.print("Enter an integer value for the number of rows (number of wire sets): ");
    	N = in.nextInt();
    	M = N;
    	ArrayList<ArrayList<Integer>> bulbs = new ArrayList<ArrayList<Integer>>();
    	
    	for (int r=1; r<=N; r++) {
    		
    		bulbs.add(new ArrayList<Integer>(3)); 
    		
    		for (int c=0; c<3; c++) { 
    			randomNum = randNumGenerator.nextInt( (M+1) - (-1*M) ) + (-1*M);
    			if (c == 0) {
    				bulbs.get(r-1).add(c, randomNum);
    			}
    			else {
    				
    				int curr = Math.abs(randNumGenerator.nextInt( (M+1) - (-1*M) ) + (-1*M));
    				int prev = Math.abs(bulbs.get(r-1).get(c-1));
    				int diff = Math.abs(curr - prev);

    					bulbs.get(r-1).add(c, randomNum);
    				}
    				else if (diff > 5) {
    					while ( diff > 5 ) {
        					randomNum = Math.abs(randNumGenerator.nextInt( (M+1) - (-1*M) ) + (-1*M));
        					if (Math.abs(randomNum-prev) <= 5) { 
        						bulbs.get(r-1).add(c, randomNum);
        						break;

        				}

    			}
    			
    		}
