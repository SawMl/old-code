import java.util.ArrayList;
import java.util.Scanner;
import java.util.*;
import java.util.Random;

public class StudentSolver
{

    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs)
    {
        Random rand = new Random();

        System.out.println(bulbs);
        ArrayList <Boolean> switches = new ArrayList <Boolean>();

        int numArr1 = bulbs.size();
        int numArr2 = bulbs.get(0).size();

        int compNum = 0;

        for (int i = 0; i < numArr1; i++)
        {
            for (int j = 0; j < numArr2; j++)
            {
                System.out.println("compNum rn is " + compNum);
                if (bulbs.get(i).get(j) > compNum)
                {
                    compNum = bulbs.get(i).get(j);
                } 
                else if (bulbs.get(i).get(j) < 0)
                {
                    int nonNegative = (bulbs.get(i).get(j)) * (-1);
                    if (nonNegative > compNum)
                    {
                        compNum = nonNegative;
                    }
                }
            }
        }

        System.out.println("Final compNum is " + compNum);
        int numSwitches = compNum;

        for (int i = 0; i < numSwitches; i++)
        {
            int randInt = rand.nextInt(2);
            System.out.println(randInt);
            if (randInt == 0)
            {
                switches.add(true);
            }else
            {
                switches.add(false);
            }
        
}



        System.out.println(switches);

        return switches;
    }
