	public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
		
		
		for(ArrayList<Integer> row : bulbs) {
			for(int element : row) {
				if(!hashMap.containsKey(Math.abs(element))) {
					if(element<0) {
						hashMap.put(Math.abs(element), false);
					}
					else {
						hashMap.put(Math.abs(element), true);
					}
				}
				else if(hashMap.containsKey(Math.abs(element))) {
					if(element<0 && hashMap.get(Math.abs(element)) == t
rue) {
						hashMap.put(Math.abs(element), false);
					}
					else if(element>0 && hashMap.get(Math.abs(element)) == false) {
						hashMap.put(Math.abs(element), true);
					}
				}
			}
		}
		
		checkSuccessfulWiring(bulbs);
		
		int highestAbsoluteInteger = findHighestAbsoluteIntegerBulb(bulbs);
		
		
		for(int i=1; i<highestAbsoluteInteger+1; i++) {
			if(hashMap.containsKey(i)) {
				booleans.add(hashMap.get(i));
			}
			else if(!hashMap.containsKey(i)) {
				booleans.add(false);
			}
		}

		
		return booleans;
		
	}
	
	
	
	
	public static int findHighestAbsoluteIntegerBulb(ArrayList<ArrayList<Integer>> bulbs) {
		int highest = 0;
		for(ArrayList<Integer> row : bulbs) {
			for(int element : row) {
				if(Math.abs(element) > highest) {
					highest = Math.abs(element);
				}
			}

		}
		return highest;
	}
	
	
	public static boolean checkSuccessfulWiring(ArrayList<ArrayList<Integer>> bulbs) {
		boolean bulbFound = false;
		for(ArrayList<Integer> row : bulbs) {
			bulbFound = false; 
			for(int element : row) {
				if(element<0 && hashMap.get(Math.abs(element)) == false) {
					
					bulbFound = true;
					break;
				}
				else if(element>0 && hashMap.get(Math.abs(element)) == true) {
					
					bulbFound = true;
					break;
				}
				else {
					
					if(hashMap.get(Math.abs(element))==tru
e) {
						hashMap.put(Math.abs(element), false);
					}
					else if(hashMap.get(Math.abs(element))==false) {
						hashMap.put(Math.abs(element), true);
					}
					bulbFound = true;
				}
			}
			
			
			if(!bulbFound) {
				System.out.println("Row with no true bulb found at ");
				for(int element : row) {
					System.out.print(element + ":" + hashMap.get(Math.abs(element)) + " ");
				}
				System.out.println();
				return false;
			}
		}
		
		return true;
