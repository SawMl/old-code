import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Random;  

public class StudentSolver {
	public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs){
		
		
		ArrayList<Boolean> answers = new ArrayList<Boolean>();
		
		int M = 0;
		
		
		for(ArrayList<Integer> row : bulbs) {
			for(int index = 0; index < row.size(); index++) {
				int n = Math.abs(row.get(index));

				if(n >= M) {
					M = n;
				}
			}
		}
		
		System.out.println(bulbs +"\n");
				
		for(int i = 1; i <= M; i++) {  
			 
			for(int row = 0; i < bulbs.size(); row++) {  
				
				ArrayList<Integer> x = bulbs.get(row);
				
				for(int j = 0; j < x.size(); j++) { 
					
					
					System.out.println(x +" == "+ i );
				
					
					if (x.get(j)==i) {
						answers.add(true);
						System.out.println("true because "+i+" is positive");
						
						x.clear();
						
						row=0;  
						
						System.out.println("\n"+ bulbs +"\n"); 
						i++;
					} 
					
					else if (x.get(j)==(i*-1)) {
						
						answers.add(false);
						System.out.println("false because "+i+" is negative");
						
						x.clear();
						
						row=0;
						
						System.out.println("\n"+ bulbs +"\n"); 
						i++;
					}
					
					
					else if(row+1 == bulbs.size()) {
						System.out.println("no "+ i + " found \n" + "\n"+ bulbs +"\n");
						
						answers.add(false);
						
						i++;
						
						row=0;
						System.out.println("row: "+row);
					}
					
				} 
				
				System.out.println("row: "+row);
			}
			
			System.out.println("error looking for "+ i);
			
			
			answers.add(false);
		}
		
		System.out.println(bulbs+"\n");
		System.out.println(answers);
		System.out.println("M = "+ M + " & Answers: "+ answers.size());
		return answers;
	}	
}
