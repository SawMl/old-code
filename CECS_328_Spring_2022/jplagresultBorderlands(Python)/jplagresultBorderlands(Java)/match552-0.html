import java.util.*;

class StudentSolver{

    public static void main(String[] args) {

        System.out.println("Enter N: ");
        Scanner inputN = new Scanner(System.in);
        int n = inputN.nextInt();

        ArrayList<ArrayList<Integer>> b = new ArrayList<>();

        for(int i = 0; i < n; i++){
            System.out.println("New Row");
            ArrayList<Integer> a = new ArrayList<>();

            for(int j = 0; j < 3; j++){
                System.out.println("Enter number: ");
                Scanner num = new Scanner(System.in);
                int number = num.nextInt();
                a.add(number);
            }
            b.add(a);
        }
        solve(b);
    }

    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs){
        System.out.println(bulbs);

        
        HashMap<Integer, Boolean> wires = new HashMap<>();

        
        int min = 0;
        for(int i = 0 ; i < bulbs.size(); i++){

            for(int j = 0; j <= 2; j++){
                int current = bulbs.get(i).get(j);

                if(min >= current){
                    min = current;
                }
            }

        }

        
        int max = 0;
        for(int i = 0 ; i < bulbs.size(); i++){

            for(int j = 0; j <= 2; j++){
                int current = bulbs.get(i).get(j);

                if(max <= current){
                    max = current;
                }
            }

        }

        
        int m = 0;
        if(Math.abs(min) < Math.abs(max)){
            m = Math.abs(max);
        } else {
            m = Math.abs(min);
        }

        
        int n = 0;
        for(int i = 0 ; i < bulbs.size(); i++){
            n += 1;
        }

        
        for(int i = 1; i <= m; i++){
            Random bool = new Random();
            wires.put(i, bool.nextBoolean());
        }

        System.out.println(wires);

        
        i();
        luv();
        food();

        int validRow = 0;
        boolean valid = true;
        int l = 0;

        
        while(valid && l < n){
            int validBulb = 0;
            
            for(int k = 1; k <= m; k++){
                if (bulbs.get(l).contains(k) && wires.get(k)){
                    validBulb =+ 1;
                }
                else if(bulbs.get(l).contains(-k) && !wires.get(k)){
                    validBulb =+ 1;
                }
            }

            
            if(validBulb > 0){
                validRow += 1;
            }
            else {
                valid = false;
            }
            l += 1;
        }

        
        ArrayList<Boolean> wiresBool = new ArrayList<>();

        
        if(validRow == n){
            for(int i = 1; i <= m; i++) {
                boolean x = wires.get(i);
                wiresBool.add(x);
            }
            System.out.println("valid");
        }
        else{
            boolean allRows = false;

            
            do {
                
                for(int i = 1; i <= m; i++){
                    Random bool = new Random();
                    wires.put(i, bool.nextBoolean());
                }

                System.out.println(wires);

                validRow = 0;

                valid = true;
                l = 0;

                while(valid && l < n){
                    int validBulb = 0;
                    for(int k = 1; k <= m; k++){
                        if (bulbs.get(l).contains(k) && wires.get(k)){
                            validBulb =+ 1;
                        }
                        else if(bulbs.get(l).contains(-k) && !wires.get(k)){
                            validBulb =+ 1;
                        }
                    }
                    l += 1;
                    if(validBulb > 0){
                        validRow += 1;
                    }
                    else {
                        valid = false;
                    }
                }

                wiresBool = new ArrayList<>();

                
                if(validRow == n){
                    for(int i = 1; i <= m; i++) {
                        boolean x = wires.get(i);
                        wiresBool.add(x);
                    }
                    System.out.println("valid");
                    allRows = true;
                }
            } while(allRows == false);

        }
        System.out.println(wiresBool);

        return wiresBool;

    }

    
    public static int i(){
        Random r1 = new Random();
        Random r2 = new Random();
        int p = r1.nextInt();
        int q = r2.nextInt();
        int r = p + q;
        int s = p * q;
        int t = p - q;
        int u = p / q;
        int v = r + s + t + u;
        return v;
    }
    
    public static int luv(){
        Random r1 = new Random();
        Random r2 = new Random();
        int p = r1.nextInt();
        int q = r2.nextInt();
        int r = p + q;
        int s = p * q;
        int t = p - q;
        int u = p / q;
        int v = r + s + t + u;
        return v;
    }
    
    public static int food(){
        Random r1 = new Random();
        Random r2 = new Random();
        int p = r1.nextInt();
        int q = r2.nextInt();
        int r = p + q;
        int s = p * q;
        int t = p - q;
        int u = p / q;
        int v = r + s + t + u;
        return v;
    }

}
