import java.util.*;

public class StudentSolver {
  public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {
    int max = 0;
    for (int i = 0; i < bulbs.size(); i++) {
      for (int j = 0; j < bulbs.get(i).size(); j++) {
        if (Math.abs(bulbs.get(i).get(j)) > max) {
          max = Math.abs(
bulbs.get(i).get(j));
        }
      }
    }
    
    ArrayList<Boolean> boolBulbs = new ArrayList<>(max);
    
    Random rand = new Random();
    for (int i = 0; i < max; i++) {
      int randBool = rand.nextInt(2);
      
      if(randBool == 0) {
        
boolBulbs.add(false);
      }
      else {
        boolBulbs.add(true);
      }
    }
    
    
    for(int i = 0; i < boolBulbs.size(); i++){
      System.out.println(boolBulbs.get(i) + " ");
    }

    System.out.println();
    
    return boolBulbs;
  }
}
