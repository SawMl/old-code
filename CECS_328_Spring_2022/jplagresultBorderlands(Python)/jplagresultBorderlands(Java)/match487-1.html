public class StudentSolver {

	 public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs)
	 {
		 ArrayList<Integer> answer = new ArrayList<Integer>(); 
		 int M = 0;
		 for(int x = 0; x<bulbs.size();x++) {
			 for(int col =0 ;col<3;col++) {
				 if (Math.abs(bulbs.get(x).get(col)) > M)
					 
					 M = Math.abs(bulbs.get(x).get(col));
			 }
		 }
		 
		 ArrayList<Integer> Non = new ArrayList<Integer>(); 
		 for(int i = 0 ; i<bulbs.size();i++) {
			 Non.add(i);
		 }
		 recursion(answer,bulbs,Non,M/2,M,0);
		 System.out.println(answer);
		 System.out.println(Non);
		 
		 return convert(answer);
	 }

	 public static ArrayList<Boolean> convert(ArrayList<Integer>answer){
		 ArrayList<Boolean> solution = new ArrayList<Boolean>();
		 for (int i : answer) {
			 solution.add(false);
		 }
		 for(int i : answer) {
			 if( i > 0) {
				 solution.set(i-1, true);
			 }

		 }
		 return solution;
		 
	 }
	 public static void recursion(ArrayList<Integer> answer, ArrayList<ArrayList<Integer>> bulbs, ArrayList<Integer> Non, int number, int M,int counter) {
		 
		 ArrayList<Integer> positive = contains(bulbs, Non, number);
		 ArrayList<Integer> negative = contains(bulbs, Non, -number);
		 if(positive.size()>negative.size()) {
			 if(number !=0) {
				answer.add(number);
				for(int i = 0; i<positive.size();i++) {
					int row = positive.get(i);
					if(Non.contains(row)){
						Non.remove(Non.indexOf(row));
					}
				}
			 }
			counter++;
			if(counter <=M) {
			if (counter%2 != 0)
				recursion(answer,bulbs,Non,number+counter,M,counter);
			else
				recursion(answer,bulbs,Non,number-counter,M,counter);
			}
		 }
		 else {
			 if(number != 0) {
				 answer.add(-number);
					for(int i = 0; i<negative.size();i++) {
						int row = negative.get(i);
						if(Non.contains(row)){
							Non.remove(Non.indexOf(row));
						}
					}
			 }
				counter++;
				if(counter <=M) {
					if (counter%2 != 0)
						recursion(answer,bulbs,Non,number+counter,M,counter);
					else
						recursion(answer,bulbs,Non,number-counter,M,counter);
					}
		 }
	 }
	 
	 
	 public static ArrayList<Integer> contains(ArrayList<ArrayList<Integer>>bulbs, ArrayList<Integer>on, int number){
		 ArrayList<Integer> lists = new ArrayList<Integer>();
		 for(int row = 0; row<on.size();row++) {
			 for(int col = 0; col<3;col++) {
				 int check = bulbs.get(on.get(row)).get(col);
				 if(check == number ) {
					 lists.add(on.get(row));
				 }
			 }
		 }
		 return lists;
		 
	 }
	 
	 public static ArrayList<Integer> findNegative(ArrayList<ArrayList<Integer>>bulbs,ArrayList<Integer>contain, int number) {
		 ArrayList<Integer> negatives = new ArrayList<Integer>();
		 for(int row = 0;row<contain.size();row++) {
			 for(int col = 0 ; col<3;col++) {
				 if(bulbs.get(contain.get(row)).get(col) == -number ) {
					 if(!negatives.contains(contain.get(row)))
						 negatives.add(contain.get(row));
				 }
			 }
		 }
		 return negatives;
	 }
