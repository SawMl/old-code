import java.util.ArrayList;
class StudentSolver {
  public static void main(String[] args) {
    ArrayList<ArrayList<Integer>>bulbs=new ArrayList<ArrayList<Integer>>();
    ArrayList<Boolean> boo=new ArrayList<Boolean>();
    boo=solve(bulbs);
  }
  public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs){
    ArrayList<Boolean> boo=new ArrayList<Boolean>();
    ArrayList<ArrayList<Integer>>bulb=bulbs;
    
for(int i=0;i<bulb.size();i++){
      for(int j=0;j<bulb.get(i).size();j++){
        int peak=bulb.get(i).get(j);
        if(peak<0){
          boo.add(false);
        }
        else if(peak>0){
          boo.add(true);
        
}
      }
    }

    
    return boo;
  }
}
