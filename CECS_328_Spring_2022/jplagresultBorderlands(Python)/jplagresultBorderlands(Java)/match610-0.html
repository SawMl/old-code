import java.util.ArrayList;
import java.util.Arrays;
import java.util.Random;
import java.util.Collections;
import java.lang.Math;

public class StudentSolver{
	@SuppressWarnings({ "unchecked", "unlikely-arg-type" })
	public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs){
		ArrayList<Boolean> solve = new ArrayList<Boolean> ();
		
		int maxValue = bulbs.get(0).get(0);
		for(int i = 0; i < bulbs.size(); i++ ) {
			for(int j = 0; j < bulbs.get(i).size(); j++) {
				if(Math.abs(bulbs.get(i).get(j)) > Math.abs(maxValue)) {
					maxValue = Math.abs(bulbs.get(i).get(j));
				}
			}
		}
		
		ArrayList<ArrayList<Boolean>> possibleConfigs = new ArrayList<ArrayList<Boolean>> ();
		
		for(int i = 0; i < maxValue; i++) {
			ArrayList<Boolean> possibleConfig = new ArrayList<Boolean> ();
			for(int j = 0; j < bulbs.size(); j++) {
				for(int k = 0; k < bulbs.get(j).size(); k ++) {
					if(bulbs.get(j).get(k) == (i+1) || bulbs.get(j).get(k) == ((i+1) * (-1))){
						if(bulbs.get(j).get(k) == (i+1)) {
							if(possibleConfig.contains(true)) {}
							else {
								possibleConfig.add(true);
							}
						}
						else if(bulbs.get(j).get(k) == ((i+1) * (-1))) {
							if(possibleConfig.contains(false)) {}
							else {
								possibleConfig.add(false);
							}
						}
					}
				}
			}
			if(possibleConfig.isEmpty()) {
				possibleConfig.add(true);
				possibleConfig.add(false);
			}
			possibleConfigs.add(possibleConfig);
		}
		
		ArrayList<Boolean> lineValues = new ArrayList<Boolean> (bulbs.size());
		for(int i = 0; i < bulbs.size(); i++) {
			lineValues.add(false);
		}
		
		for(int i = 0; i < maxValue; i++) {
			for(int j = 0; j < bulbs.size(); j++) {
				if (lineValues.get(j) == true) {
					continue;
				}
				else {
						if(bulbs.get(j).contains(i+1) || bulbs.get(j).contains((i+1) * (-1))) {
							if(bulbs.get(j).contains(i+1)) {
								if(possibleConfigs.get(i).contains(true)) {
									if(possibleConfigs.get(i).size() > 1) {
										possibleConfigs.remove(false); 
										lineValues.set(j, true); 
										
									}
									else {
										lineValues.set(j, true); 
									}
								}
							}
							else if(bulbs.get(j).contains((i+1) * (-1))) {
								if(possibleConfigs.get(i).contains(false)) {
									if(possibleConfigs.get(i).size() > 1) {
										possibleConfigs.remove(true); 
										lineValues.set(j, true); 
										
									}
									else {
										lineValues.set(j, true); 
									}
								}
							}
						}
				}
			}
		}
		
		ArrayList<Boolean> finalConfig = new ArrayList<Boolean> ();
		for(int i = 0; i < maxValue; i++) {
			finalConfig.add(possibleConfigs.get(i).get(0));
		}
		
		for(int i = 0; i < lineValues.size(); i++) {
			if(lineValues.get(i) == false) {
				for(int j = 0; j < bulbs.get(i).size(); j ++) {
					int value = Math.abs(bulbs.get(i).get(j));
					if(possibleConfigs.get(value).size() > 1) {
						finalConfig.set(value-1, !finalConfig.get(value));
						lineValues.set(i, true);
						break;
					}
				}
			}
		}
		
		solve = finalConfig;
		
		return solve;

	}

}
