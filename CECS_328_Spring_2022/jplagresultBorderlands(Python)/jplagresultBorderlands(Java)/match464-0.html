import java.util.*;
import java.lang.Math;

public class StudentSolver
{
  
  public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>
bulbs)
  {
  int max = 0;
  for(int i=0;i < bulbs.size();i++)
  {
      for(int j=0;j < bulbs.get(i).size();j++)
      {
        if(Math.abs(bulbs.get(i).get(j)) > max)
          max = Math.abs(bulbs.get(i).get(j));
      }
  }
  
  int M = max;
  ArrayList<Boolean> sol = new ArrayList<Boolean>();
  int r = (int)(Math.random()*2) + 1;
  System.out.println(M);
  for(int j = 0; j < M; j++)
  {
  
      if(r == 1)
        sol.add(j,true);
      if(r == 2)
        sol.add(j,false);
    r = (int)(Math.random()*2) + 1;
     
  }
  System.out.println(sol.size());
  return sol;
  }
 
}
