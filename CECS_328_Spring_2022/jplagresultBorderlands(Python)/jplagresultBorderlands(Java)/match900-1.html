import java.util.*;

class StudentSolver 
{
    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs) 
  {

    ArrayList<Boolean> onOrOffValues = new ArrayList<Boolean>();

    
    for(int i = 0; i < bulbs.size(); i++)
      {
        int j = 0;
        while(j < 3)
          {
            if(bulbs.get(i).get(j) < 0)  
            {
              
System.out.println(bulbs.get(i).get(j));
              j++;
              if(bulbs.get(i).get(j) < 0)  
             {
                System.out.println(bulbs.get(i).get(j));
                j++;
                if(bulbs.get(i).get(j) < 0)  
                {
                  System.out.println(bulbs.get(i).get(j));
                  onOrOffValues.add(false);  
                  break;
                }
             }
           }
                else
                {
                  onOrOffValues.add(true);  
                  break;
                }
          }
      }
    return onOrOffValues;
  }
}
