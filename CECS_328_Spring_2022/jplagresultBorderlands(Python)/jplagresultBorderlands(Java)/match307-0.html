import java.util.ArrayList;
import java.util.Collections;
import java.util.Map;
import java.util.Random;

public class StudentSolver {
    public static ArrayList<Boolean> solve (ArrayList<ArrayList<Integer>> bulbs){
        ArrayList<Boolean> validConfigurationForGame = new ArrayList<>();

        ArrayList<Integer> firstCol = new ArrayList<>();
        ArrayList<Integer> secondCol = new ArrayList<>();
        ArrayList<Integer> thirdCol = new ArrayList<>();

        int maxNumInBulbs = 0;

        


        
        for (int i = 0; i < bulbs.size(); i++){
            ArrayList<Integer> currentRow = bulbs.get(i);
            for (int j = 0; j < currentRow.size(); j++) {
                if (j == 0) firstCol.add(currentRow.get(j));
                if (j == 1) secondCol.add(currentRow.get(j));
                if (j == 2) thirdCol.add(currentRow.get(j));
            }
        }

        


        
        ArrayList<Integer> firstColAbs = new ArrayList<>();
        ArrayList<Integer> secondColAbs = new ArrayList<>();
        ArrayList<Integer> thirdColAbs = new ArrayList<>();
        for (int i = 0; i < 3; i++){
            ArrayList<Integer> currentRow = new ArrayList<>();
            if (i == 0) currentRow = firstCol;
            if (i == 1) currentRow = secondCol;
            if (i == 2) currentRow = thirdCol;
            for (int j = 0; j < currentRow.size(); j++){
                int tempAbs = Math.abs(currentRow.get(j));
                if (i == 0) firstColAbs.add(tempAbs);
                if (i == 1) secondColAbs.add(tempAbs);
                if (i == 2) thirdColAbs.add(tempAbs);
            }
        }

        int highestNum = Collections.max(firstColAbs);
        for (int i = 1; i < 3; i++) {
            int currentNum = (i == 1) ? Collections.max(secondColAbs) : Collections.max(thirdColAbs);
            if(highestNum < currentNum){
                highestNum = currentNum;
            }
        }
        maxNumInBulbs = highestNum;


        for(int i = 0; i < maxNumInBulbs; i++){ 
            validConfigurationForGame.add(null);
        }
        
        














        
        for (int i = 0; i < bulbs.size(); i++) {
            ArrayList<Integer> currentArr = bulbs.get(i);
            for (int j = 0; j < currentArr.size(); j++) {
                boolean sign = false;
                int currentNumber = currentArr.get(j);
                if (currentNumber < 0) sign = false;
                if (currentNumber >= 0) sign = true;
                currentNumber = Math.abs(currentNumber);
                if (validConfigurationForGame.get(currentNumber - 1) == null){
                    validConfigurationForGame.set(currentNumber - 1, sign);
                } else if (validConfigurationForGame.get(currentNumber - 1) != sign){
                    int count = 0;
                    for (int k = 0; k < currentArr.size(); k++) {
                        int index = currentArr.get(k);
                        sign = false;
                        if (index < 0) sign = false;
                        if (index >= 0) sign = true;
                        index = Math.abs(index);
                        
                        if (validConfigurationForGame.get(index - 1) != null){
                            if (index != currentNumber && validConfigurationForGame.get(index - 1) != sign){
                                count++;
                            }
                        } else{
                            validConfigurationForGame.set(index - 1, sign);
                        }
                        if (count >= 2){
                            validConfigurationForGame.set(currentNumber - 1, !validConfigurationForGame.get(currentNumber - 1));
                        }
                    }
                }
            }
            
            
        }

        
        for (int i = 0; i < validConfigurationForGame.size(); i++) {
            if (validConfigurationForGame.get(i) == null){
                validConfigurationForGame.set(i, false);
            }
        }


        
        



























            



        return validConfigurationForGame;
    }







































































}
