import java.util.*;
import java.lang.Math;

public class StudentSolver {






    public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>> bulbs) {

        int blub, i, j;
        ArrayList<Boolean> solution = new ArrayList<>();
        Set<Integer> is_true = new HashSet<>(Arrays.asList(4, 5, 7, 11, 13, 14));
        Set<Integer> is_false = new HashSet<>(Arrays.asList(1, 2, 3, 6, 8, 9, 10, 12));

        for(i = 0; i < bulbs.size(); i++) {
            for(j = 0; j < 3; j++) {
                blub = bulbs.get(i).get(j);
                if(is_true.contains(Math.abs(blub)) && blub > 0) {
                    solution.add(Boolean.TRUE);
                    break;
                }
                if(is_false.contains(Math.abs(blub)) && blub < 0) {
                    solution.add(Boolean.FALSE);
                    break;
                }
                if(j == 2) {
                    
solution.add(Boolean.FALSE);
                }
            }
        }

        return solution;
    }

}
