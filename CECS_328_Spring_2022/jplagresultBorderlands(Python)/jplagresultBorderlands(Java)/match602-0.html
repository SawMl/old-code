import java.util.ArrayList;
import java.util.HashSet;
import java.lang.Math;

public class StudentSolver {
		
		static int n = 1;
		static ArrayList <Boolean> key = new ArrayList <Boolean>();
		static ArrayList <Integer> nums = new ArrayList <Integer>();
		
		public static ArrayList<Boolean> solve(ArrayList<ArrayList<Integer>>bulbs){
			
			findN(bulbs);
			defTrue();
			majorityf(0, bulbs);

				
	
			
			System.out.println("N: "+ n);
			System.out.println("Answer: "+key.size());
			return key;
			
		}
		
		public static void majorityf( int x, ArrayList<ArrayList<Integer>>bulbs) {
			
			if(x > n-1) {
				return;
			}
			
			int pos = 0;
			int neg = 0;
			
			for(int i = 0; i < bulbs.size();i ++) {
				for (int k = 0; k < 3; k ++) {
					
					if (bulbs.get(i).get(k) == nums.get(x) ) {
						pos = pos + 1;
					}
					if(bulbs.get(i).get(k) == nums.get(x)*-1) {
						neg = neg + 1;
					}
					
				}
			}
			if(pos < neg) {
				key.set(x, false);
			}
			pos = 0;
			neg = 0;
			majorityf(x +1, bulbs);
			
			
		}
		
		
		
		public static void findN(ArrayList<ArrayList<Integer>>bulbs) {
			
			int i = Integer.MIN_VALUE;
			
			
			for(int k = 0; k<bulbs.size(); k++) {
				for(int j = 0; j < 3; j++) {
					if (Math.abs(bulbs.get(k).get(j))> i) {
						
						i = bulbs.get(k).get(j);
						
					}
				}
			}
			
			n = i;
			
		}
		
		public static void defTrue() {
			
			for (int i = 0; i < n;i++) {
				key.add(true);
				nums.add(i+1);
			}
		}

	}
