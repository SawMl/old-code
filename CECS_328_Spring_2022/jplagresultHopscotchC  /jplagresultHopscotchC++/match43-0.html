#include "StudentSolver.h"
using namespace std;

vector<int> StudentSolver::solve(vector<vector<int>> grid)
{
    vector<int> solution;
    int m = grid.size();
    int n = grid[0].size();
    int startCol = 0;
    int startRow = 0;
    int r = 0;
    for(int i = 0; i < 2; i++)
    {
        
        for(int j = 0; j < n; j++)
        {
            if(grid[startRow][startCol] < grid[r][j])
            {
                startRow = r;
                startCol = j;
            }
        }
        r = m-1;
    }
    if(startRow == 0)
    {
        solution.push_back(1);
    }
    else
    {
        solution.push_back(0);
    }

    solution.push_back(startCol);

    int exitCol = startCol - 1;
    if(exitCol < 0)
    {
        exitCol = n-1;
    }
    int exitRow;
    if(grid[0][exitCol] > grid[m-1][exitCol])
    {
        exitRow = 0;
    }
    else
    {
        exitRow = m-1;
    }

    cout << "Exit Row: " << exitRow << endl;


    
    int tempCol = startCol;
    int tempRow = startRow;
    for(int i = 0 ; i < n-1; i++)
    {   
        int highestCol = tempCol+1;
        int highestRow = tempRow-1;
        if(highestRow < 0)
        {
            highestRow = m-1;
        }
        if(highestCol > n-1)
        {
            highestCol = 0;
        }
        
        int move = 0;
        for(int j = 1 ; j < 3 ; j++)
        {
            int overHop; 
            if(highestRow+1 > m-1)
            {
                overHop = 0;
            }
            else
            {
                overHop = highestRow + 1;
            }
            if(grid[highestRow][highestCol] <= grid[overHop][highestCol])
            {
                
                highestRow = overHop;
                move = j;
            }
        }
        if(move == 0)
        {
            solution.push_back(1);
        }
        else if(move == 1)
        {
            solution.push_back(0);
        }
        else
        {
            solution.push_back(-1);
        }

        tempRow = highestRow;
        tempCol = highestCol;
    }





    
    cout <<"["<< startRow << "]" << "["<< startCol << "]" << endl;
    return solution;







}
