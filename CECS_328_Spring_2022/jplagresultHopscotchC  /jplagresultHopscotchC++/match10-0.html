#include <iostream>
#include <vector>
#include "StudentSolver.h"

using namespace std;

vector<int> StudentSolver::solve(vector<vector<int>> grid) {
	int row = grid.size();
	int col = grid[0].size();
	int temp = 0;
	int startcol = 0;
	int topbot;
	vector<int> path;

	for (int i = 0; i < col; i++) {
		if (grid[0][i] > temp) {
			temp = grid[0][i];
			startcol = i;
			topbot = 1;
		}
		if (grid[row-1][i] > temp) {
			temp = grid[row-1][i];
			startcol = i;
			topbot = 0;
		}
	}

	path.push_back(topbot);
	path.push_back(startcol);


	int startposrow;
	int startposcol = startcol;
	if (topbot == 1) {
		startposrow = 0;
	}
	else {
		startposrow = row - 1;
	}

	int tempz = 0;
	int uppos;
	int downpos;
	int rightpos = grid[startposrow][startposcol + 1];
	if (topbot == 1) {
		uppos = grid[row - 1][startposcol + 1];
		downpos = grid[startposrow + 1][startposcol + 1];
	}
	else if (topbot == 0) {
		uppos = grid[startposrow - 1][startposcol + 1];
		downpos = grid[tempz][startposcol + 1];
	}
	

	for (int j = 0; j < col; j++) {
		if (j == col - 1) {
			if (startposrow == row - 2) {
				path.push_back(-1);
				break;
			}
			else if (startposrow == 1) {
				path.push_back(1);
				break;
			}
			else {
				continue; 
			}
		}

		if (rightpos > uppos && rightpos > downpos) {
			path.push_back(0);
			startposcol++;
		}
		else if (uppos > rightpos && uppos > downpos) {
			path.push_back(1);
			startposcol++;
			if (startposrow == 0) {
				startposrow = row - 1;
			}
			else {
				startposrow++;
			}
		}
		else if (downpos > rightpos && downpos > uppos) {
			path.push_back(-1);
			startposcol++;
			if (startposrow == row - 1) {
				startposrow = 0;
			}
			else {
				startposrow++;
			}
		}
	}
	return path;
}
