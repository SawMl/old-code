#include "StudentSolver.h"

vector<int> StudentSolver::solve(vector<vector<int>> grid)
{

	vector<int> path;    
	vector<int> optimalPath; 
	int numRows;		 
	int numCol;	         
	int pathTotal;	     
	int maxPath;	     
	bool notDone;        

	int searchCol;    
	int searchRow;    
	int travelLength; 

	numRows = grid.size();
	numCol  = grid[0].size();
	pathTotal = 0;
	maxPath = 0;

	notDone = true;

	
	for (int i = 0; i < numRows; i += (numRows - 1))
	{
		for (int j = 0; j < numCol; j++)
		{
			path.clear();
			pathTotal = 0;

			
			if (i == 0)
			{
				path.push_back(1);
			}
			else
			{
				path.push_back(0);
			}

			path.push_back(j);

			pathTotal += grid[i][j];

			
			
			
			travelLength = 0;	
			searchRow = i;		
			searchCol = j;		
			notDone = true;		

			while (notDone)
			{
				
				if (travelLength != numCol - 1) 
				{
					
					if (searchCol + 1 == numCol)
					{
						searchCol = -1;
					}

					
					if (searchRow == 0)
					{
						
						if (grid[numRows - 1][searchCol + 1] > grid[searchRow][searchCol + 1] && 
							grid[numRows - 1][searchCol + 1] > grid[searchRow + 1][searchCol + 1])
						{
							path.push_back(1);
							pathTotal += grid[numRows - 1][searchCol + 1];
							searchRow = numRows - 1;
						}
						
						else if (grid[searchRow][searchCol + 1] > grid[numRows - 1][searchCol + 1] &&
							     grid[searchRow][searchCol + 1] > grid[searchRow + 1][searchCol + 1])
						{
							path.push_back(0);
							pathTotal += grid[searchRow][searchCol + 1];
						}
						
						else
						{
							path.push_back(-1);
							pathTotal += grid[searchRow + 1][searchCol + 1];
							searchRow++;
						}
					}
					
					else if (searchRow == numRows - 1)
					{
						
						if (grid[searchRow - 1][searchCol + 1] > grid[searchRow][searchCol + 1] &&
							grid[searchRow - 1][searchCol + 1] > grid[0][searchCol + 1])
						{
							path.push_back(1);
							pathTotal += grid[searchRow - 1][searchCol + 1];
							searchRow--;
						}
						
						else if (grid[searchRow][searchCol + 1] > grid[searchRow - 1][searchCol + 1] &&
							grid[searchRow][searchCol + 1] > grid[0][searchCol + 1])
						{
							path.push_back(0);
							pathTotal += grid[searchRow][searchCol + 1];
						}
						
						else
						{
							path.push_back(-1);
							pathTotal += grid[0][searchCol + 1];
							searchRow = 0;
						}
					}
					
					else
					{
						
						if (grid[searchRow - 1][searchCol + 1] > grid[searchRow][searchCol + 1] &&
							grid[searchRow - 1][searchCol + 1] > grid[searchRow + 1][searchCol + 1])
						{
							path.push_back(1);
							pathTotal += grid[searchRow - 1][searchCol + 1];
							searchRow--;
						}
						
						else if (grid[searchRow][searchCol + 1] > grid[searchRow - 1][searchCol + 1] &&
							grid[searchRow][searchCol + 1] > grid[searchRow + 1][searchCol + 1])
						{
							path.push_back(0);
							pathTotal += grid[searchRow][searchCol + 1];
						}
						
						else
						{
							path.push_back(-1);
							pathTotal += grid[searchRow + 1][searchCol + 1];
							searchRow++;
						}
					}
					searchCol++;
					travelLength++;
				}
				else
				{
					notDone = false;
				}
			}

			
			if (pathTotal > maxPath)
			{
				optimalPath = path;
				maxPath = pathTotal;
			}
		}
	}

	return optimalPath;
}
