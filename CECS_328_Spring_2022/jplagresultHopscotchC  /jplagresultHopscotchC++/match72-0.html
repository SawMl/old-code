#include "StudentSolver.h"

#include<vector>
using namespace std;






vector<int> StudentSolver:: solve(vector<vector<int>> grid)
{

  
  for (int i = 0; i < grid.size(); i++)
    {
      for (int j = 0; j < grid[i].size(); j++)
        {
          cout << grid[i][j];
        }
      cout << endl;
    }
    
  vector<int> tempVec = {1, 2, 3, 4};
  return(tempVec);
  }
