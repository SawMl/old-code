def solve(grid):
#case mx1
  m = len(grid) #rows
  n = len(grid[0]) #collumns
  mylist =[]
  
  # print(grid[0][0])
  # print(grid[-1][0])
  if (n == 1):  #collums = 1
    if (grid[0][0] > grid[-1][0]):
      mylist.append(1) 
    else:
      mylist.append(0)
    mylist.append(0)
    return mylist
  
  if (m==1):  #rows = 1
    curr_max = grid[0][0]
    idx = 0
    mylist.append(1)
    for i in range(1,n):
      if grid[0][i] > curr_max:
        curr_max = grid[0][i]
        idx = i
    mylist.append(idx)  
    for i in range(1,n-1):
      mylist.append(1)
    
    return mylist

  
  if (m==2 and n==2):
    if (grid[0][0] >= grid[0][1] and grid[0][0]>=grid[1][0] and grid[0][0]>=grid[1][1]):
      mylist.append(0)
      mylist.append(0)
    elif (grid[0][1] > grid[1][1] and grid[0][1]>grid[1][0] and grid[0][1]>grid[0][0]):
      mylist.append(0)
      mylist.append(1)
    elif (grid[1][0] > grid[1][1] and grid[1][0]>grid[0][0] and grid[1][0]>grid[0][1]):
      mylist.append(1)
      mylist.append(0)
    else:
      mylist.append(1)
      mylist.append(1)

    if (mylist == [0,0]):
      if (grid[0][1] >= grid[1][1]):
        mylist.append(0)
      else:
        mylist.append(-1)
    elif (mylist == [0,1]):
      if (grid[0][0] >= grid[0][1]):
        mylist.append(0)
      else:
        mylist.append(1)
    elif (mylist == [1,0]):
      if grid[0][1] >= grid[1][1]:
        mylist.append(1)
      else:
        mylist.append(0)
    elif (mylist == [1,1]):
      if grid[0][0] >= grid[1][0]:
        mylist.append(-1)
      else:
        mylist.append(0)
    return mylist
