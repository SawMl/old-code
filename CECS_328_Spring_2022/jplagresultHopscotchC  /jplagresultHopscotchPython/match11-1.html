bulbs 
= [[-4, -4, -1],
       [13, -9, -14],
       [4, -7, -2],
       [-2, 6, 3],
       [-6, -8, 11],
       [-4, -3, -6],
       [7, 11, -10],
       [4, -7, 6],
       [5, 6, 4],
       [10, 14, -9]]
def solve(bulbs):
# finding the maximum number in array by using max() function.
# https:
    m = max(max(m) for m in bulbs )
    # n equals to number of rows
    n = len(bulbs)
    '''
    If each of any N row contains negative integer, then the row contains red bulb.
    Meaning that turning off same number of wire guarantees the row with at least
    one light bulb on.

    check arry's elements if exists.
    '''
    lst = []
    replace = [0,0,0]
    for i in range(m):
        lst.append(True)
    for i in range(-1,-m-1,-1):  
        for j in range(n):
            if(i) in bulbs[j]:
                bulbs[j] = replace
                lst[-1*i-1] = False
    return lst
print(solve)
