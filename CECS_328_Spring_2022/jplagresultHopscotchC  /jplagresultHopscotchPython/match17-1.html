def solve(grid):
    m = len(grid)
    n = len(grid[0])
    highestStart = 0
    finalAnswer = [0, 0]
    for i in range(1, n):
        if grid
[0][i] + grid[0][i-1] > highestStart:
            highestStart = grid[0][i] + grid[0][i-1]
            finalAnswer[0] = 1
            finalAnswer[1] = i
        if grid[0][i] + grid[m-1][i-1] > highestStart:
            highestStart = grid[0][i] + grid[m-1][i-1]
            finalAnswer[0] = 1
            finalAnswer[1] = i
    for i in range(1, n):
        if grid[m-1][i] + grid[0][i-1] > highestStart:
            highestStart = grid[m-1][i] + grid[0][i-1]
            finalAnswer[0] = 1
            finalAnswer[1] = i
        if grid[m-1][i] + grid[m-1][i-1] > highestStart:
            highestStart = grid[m-1][i] + grid[m-1][i-1]
            finalAnswer[0] = 1
            finalAnswer[1] = i
    steps = 0
    columnTrack = finalAnswer[1]
    if finalAnswer[0] == 1:
        startingPoint = 0
    else:
        startingPoint = m-1
    while steps < n-1:
        highestValue = 0
        stepToTake = 0
        columnTrack += 1
        if columnTrack == n:
            columnTrack = 0
        if startingPoint - 1 < 0:
            if grid[m-1][columnTrack] >= highestValue:
                highestValue = grid[m-1][columnTrack]
                stepToTake = 1
            if grid[startingPoint][columnTrack] >= highestValue:
                highestValue = grid[startingPoint][columnTrack]
                stepToTake = 0
            if grid[startingPoint+1][columnTrack] >= highestValue:
                highestValue = grid[startingPoint+1][columnTrack]
                stepToTake = -1
        elif startingPoint + 1 >= m:
            if grid[0][columnTrack] >= highestValue:
                highestValue = grid[0][columnTrack]
                stepToTake = -1
            if grid[startingPoint][columnTrack] >= highestValue:
                highestValue = grid[startingPoint][columnTrack]
                stepToTake = 0
            if grid[startingPoint-1][columnTrack] >= highestValue:
                highestValue = grid[startingPoint-1][columnTrack]
                stepToTake = 1
        else:
            if grid[startingPoint-1][columnTrack] >= highestValue:
                highestValue = grid[startingPoint-1][columnTrack]
                stepToTake = 1
            if grid[startingPoint][columnTrack] >= highestValue:
                highestValue = grid[startingPoint][columnTrack]
                stepToTake = 0
            if grid[startingPoint+1][columnTrack] >= highestValue:
                highestValue = grid[startingPoint+1][columnTrack]
                stepToTake = -1
        finalAnswer.append(stepToTake)
        startingPoint += -(stepToTake)
        if startingPoint >= m:
            startingPoint = 0
        elif startingPoint < 0:
            startingPoint = m-1
        steps += 1
    return finalAnswer
