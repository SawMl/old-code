def starterPokemon():
    startC = 0
    startR = 0
    
    theLoot = 0 #comparitive number for sart finder
    for h in range(COLS):
        if cheemer(h):
            if cheemer(h) + BOARD[0][(h+1) % COLS] > theLoot:
                theLoot = cheemer(h) + BOARD[0][(h+1) % COLS]
                startC = (h + 1)  % COLS
                startR = 0
            elif cheemer(h) + BOARD[-1][(h+1) % COLS] > theLoot:
                theLoot = cheemer(h) + BOARD[-1][(h+1) % COLS]
                startC = (h + 1)  % COLS
                startR = -1
    return startC,startR

def cheemer(x): #COL validator
    if BOARD[0][x] == STINKY[x]:
        return BOARD[0][x]
    elif BOARD[-1][x] == STINKY[x]:
        return BOARD[-1][x]

    return None

def moldTheCheese():
    maxMemes = []
    for m in range(COLS):
        cheese = set()
        for k in range(ROWS):
            cheese.add(BOARD[k][m])
        maxMemes.append(max(cheese))
    return maxMemes

def magic(a, b, instructions, sum = 0, memes = []):
    
    if memes:
        print(max(memes))
    if len(instructions) == COLS + 1:
        if a != 0 and a != ROWS - 1:
            return []
        return [[sum]+instructions]
    
    if not instructions:
        if a == 0:
            instructions.append(1)
        else:
            instructions.append(0)
        instructions.append(b)
    
    pos = Space(a, b)
    
    moves = [
pos.getNextDown().getValue(), pos.getNextRight().getValue(), pos.getNextUp().getValue()]
    
    
    memes =  magic(pos.getNextDown().getRow(), pos.getNextDown().getCol(), instructions + [-1], sum + moves[0], memes) + magic(pos.getNextRight().getRow(), pos.getNextRight().getCol(), instructions + [0], sum + moves[1], memes) + magic(pos.getNextUp().getRow(), pos.getNextUp().getCol(), instructions + [1], sum + moves[2], memes)

    return memes


class Space:
    def __init__(self, a = 0, b = 0):
        self.a = a % ROWS
        self.b = b % COLS
       
        
         
    def getRow(self):
        return self.a
    
    def getCol(self):
        return self.b
    
    def getValue(self):
        return BOARD[self.a][self.b]
    

    def getNextUp(self):
        return Space(self.a - 1 , self.b + 1)
    
    def getNextRight(self):
        return Space(self.a, self.b + 1)
    
    def getNextDown(self):
        return Space(self.a + 1 , self.b + 1)
