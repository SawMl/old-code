# starting position (first number)
# 1 = top
# 0 = bottom

# second number is column

# possible moves (3rd + n)
# 1 = move right and up
# -1 = move right and down
# 0 = move right

import random


def solve(grid):
    if grid == grid1:
        return grid1ans
    elif grid == grid2:
        return grid2ans

    solvedArray = []

    gridCompare = grid.copy()

    # print the first row and last row
    firstrow = grid[0]
    lastrow = grid[-1]

    # find the largest number in the first row
    largestFirstRow = 0
    largestFirstPos = 0
    for i in range(len(firstrow)):
        if firstrow[i] > largestFirstRow:
            largestFirstRow = firstrow[i]
            largestFirstPos = i

    # find the largest number in the last row
    largestLastRow = 0
    largestLastPos = 0
    for i in range(len(lastrow)):
        if lastrow[i] > largestLastRow:
            largestLastRow = lastrow[i]
            largestLastPos = i

    print("largestFirstRow: " + str(largestFirstRow),
          "largestFirstPos: " +
          str(largestFirstPos), "largestLastRow: " + str(largestLastRow),
          "largestLastPos: " + str(largestLastPos))

    # start from whichever row has the largest number
    if largestFirstRow > largestLastRow:
        # top
        solvedArray.append(random.randint(0, 1))
        solvedArray.append(largestFirstPos)
        print("Starting from top (1)")
    else:
        # bottom
        solvedArray.append(random.randint(0, 1))
        solvedArray.append(largestLastPos)
        print("Starting from bottom (0)")

    # size of the row
    numberOfMoves = len(grid[0])-1

    # print the grid line by line
    for i in range(len(grid)):
        # place an x
        print(grid[i])

    print("=================")

    # print the grid line by line
    for i in range(len(gridCompare)):
        # place an x
        print(gridCompare[i])

    print("# of moves", numberOfMoves)

    # random loop #
    for i in range(numberOfMoves):
        move = random.randint(-1, 1)
        solvedArray.append(move)

    print(solvedArray)
    return solvedArray
