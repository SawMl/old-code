def solve(grid):
    answer = []
    max_top = 0
    max_bot = 0

    # get the largest number in the first list
    for i in grid[0]:
        if max_top < i:
            max_top = i
    # get the largest number in the last list
    for i in grid[len(grid) - 1]:
        if max_bot < i:
            max_bot = i

    if grid[0].index(max_top) < grid[len(grid) - 1].index(max_bot):
        answer.append(len(grid))
        answer.append(grid[len(grid) - 1].index(max_bot))
        row = len(grid) - 1
        col = grid[len(grid) - 1].index(max_bot)
    else:
        answer.append(1)
        answer.append(grid[0].index(max_top))
        row = 0
        col = grid[0].index(max_top)

    for i in range(1, len(grid[0])):
        # pointer (row) to keep track of where the path is going
        # case where you reach the end of the column, move your check to the beginning of the list
        if col + 1 == len(grid[0]):
            col = 0
            if grid[row - 1][col] > grid[row][col] and grid[row - 1][col] > grid[row + 1][col]:
                answer.append(1)
                row -= 1
            elif grid[row][col] > grid[row - 1]
[col] and grid[row][col] > grid[row + 1][col]:
                answer.append(0)
            elif grid[row + 1][col] > grid[row - 1][col] and grid[row + 1][col] > grid[row][col]:
                answer.append(-1)
                row = 0
        # case where you reach the bottom of the row to, move your check to the top of the list
        elif row + 1 == len(grid):
            if grid[row - 1][col + 1] > grid[row][col + 1] and grid[row - 1][col + 1] > grid[0][col + 1]:
                answer.append(1)
                row -= 1
            elif grid[row][col + 1] > grid[row - 1]
[col + 1] and grid[row][col + 1] > grid[0][col + 1]:
                answer.append(0)
            elif grid[0][col + 1] > grid[row - 1][col + 1] and grid[0][col + 1] > grid[row][col + 1]:
                answer.append(-1)
                row = 0
            col += 1
        else:
            if grid[row - 1][col + 1] > grid[row][col + 1] and grid[row - 1][col + 1] > grid[row + 1][col + 1]:
                answer.append(1)
                row -= 1
            elif grid[row][col + 1] > grid[row - 1][col + 1] and grid[row][col + 1] > grid[row + 1][col + 1]:
                answer.append(0)
            elif grid[row + 1][col + 1] > grid[row - 1][col + 1] and grid[row + 1][col + 1] > grid[row][col + 1]:
                answer.append(-1)
                row += 1
            # condition where two or more are the same value/also to make the the program reach the end of the cylinder
            elif grid[row - 1][col + 1] == grid[row][col + 1] or grid[row - 1][col + 1] == grid[row + 1][col + 1] or \
                    grid[row - 1][col + 1] == grid[row + 1][col + 1]:
                if row > len(grid[0]) / 2:
                    answer.append(1)
                    row -= 1
                elif row < len(grid[0]) / 2:
                    answer.append(-1)
                    row += 1
                else:
                    answer.append(0)
            col += 1
    return answer
