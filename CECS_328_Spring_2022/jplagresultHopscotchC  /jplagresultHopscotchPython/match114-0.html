def solve(grid):
    row = len(grid)
    col = len(grid[0])
    topAr = []
    botAr = []
    topValue = 0
    botValue = 0
    
    if col == 1:
        if grid[0][0] > grid[row-1][0]:
            topAr =[1, 0]
        else:
            topAr =[0,0]
        return topAr

    for x in range(0,col-1):
        topRow = 0
        botRow = row-1
        topCol = x
        botCol = x
        # print(x)
        top = [1, x]
        bot = [0, x]
        # print("row", topRow)
        # print("col ",topCol)
        newTop = grid[topRow][topCol]
        newBot = grid[botRow][topCol]

        for i in range(col-1):
            topRow, topCol, topMovement, topNum = numCheck(grid, topRow, topCol, i)
            # print(topRow, " ", topCol," ", topMovement," ", value)
            newTop += topNum
            top.append(topMovement)
            botRow, botCol, botMovement, botNum = numCheck(grid, botRow, botCol, i)
            newBot += botNum
            bot.append(botMovement)
            # print("i, ",i)
            # print(topNum," ,", topRow, ", ", topCol)
        # if x ==1:
        #     print(newTop)
        #     print(top)
        if newTop > topValue:
            topValue = newTop
            # print("before",topAr)
            # print("topvalue,",topValue)
            topAr = top
            # print("after",top)
        if newBot > botValue:
            botValue = newBot
            botAr = bot
    #print(top)
    # print(topValue, " and ", botValue)
    if topValue > botValue:
        return topAr
    else:
        return botAr
