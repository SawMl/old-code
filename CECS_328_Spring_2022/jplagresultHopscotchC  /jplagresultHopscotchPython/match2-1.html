def getScore(r, c, pth, grd):
    score = grd[r][c]
    for p in pth:
        if p == 0:
            
if c == len(grd[0])-1:
                c = 0
            else:
                c += 1
            score += grd[r][c]
        elif p == 1:
            if c == len(grd[0])-1:
                c = 0
            else:
                c += 1

            if r == 0:
                r = len(grd)-1
            else: 
                r -= 1
            score += grd[r][c]
        elif p == -1:
            
if c == len(grd[0])-1:
                c = 0
            else:
                c += 1

            if r == len(grd)-1:
                r = 0
            else: 
                r += 1
            score += grd[r][c]
    return score


def getPaths(valid, pref, n, l , all_paths): 
    if l == 0:
        lst = pref.strip().split(" ")
        lst = list(map(int, lst))
        sum_lst = sum(lst) 
        if(sum_lst == 1):
            all_paths
.append(lst)
        return
        
    for i in range(n):
        nPref = pref + valid[i] + " "
        getPaths(valid, nPref, n, l-1, all_paths)

    
def solve(grid):
    r = 0
    c = 0
    max = 0
    opt_path = []
    all_paths = []
    valid = ['1', '0', '-1']
    l = len(grid[r]) - 1
    getPaths(valid, "", len(valid), l, all_paths)

    for i in range(l):
        for path in all_paths:
            score = getScore(r, i, path, grid)
            if score > max:
                max = score
                opt_path = path
                c = i
    opt_path = list(map(str, opt_path))
    path_str = ", ".join(opt_path)
    fin_str = "1, " + str(c) + ", " + path_str
    res = fin_str.split(", ")
    res = list(map(int, res))
    return res
