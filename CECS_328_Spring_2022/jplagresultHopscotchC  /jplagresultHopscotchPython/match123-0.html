def solve(grid):
    rows = len(grid)
    col = len(grid[0])
    traverse = []
    compare = []
    compareOne = []
    additionList = []
    for i in range(col):
        if i == 0:
            for j in range(col - 1):
                checker = grid[0][j] + max(grid[0][j + 1], grid[1][j + 1])
                checkerTwo = grid[rows - 1][j] + max(grid[rows - 1][j + 1], grid[rows - 2][j + 1])
                compare.append(checker)
                compareOne.append(checkerTwo)
    
            if(max(compare) > max(compareOne)):
                traverse.append(1)
                start_row = 0
                index_max = max(range(len(compare)), key = compare.__getitem__)
                traverse.append(index_max)
            elif(max(compare) < max(compareOne)):
                traverse.append(0)
                start_row = rows - 1
                index_max = max(range(len(compareOne)), key = compareOne.__getitem__)
                traverse.append(index_max)
            elif(max(compare) == max(compareOne)):
                traverse.append(0)
                start_row = rows - 1
                index_max = max(range(len(compareOne)), key = compareOne.__getitem__)
                traverse.append(index_max)
    
        else:
            #bottom right corner OOB
            if(start_row + 1 >= rows and index_max + 1 >= col):
                grid[start_row][index_max] += max(grid[start_row - 1][index_max - (col - 1)], max(grid[start_row][index_max - (col - 1)], grid[start_row - (rows - 1)][index_max - (col - 1)]))

                
                if(grid[start_row - 1][index_max - (col - 1)] > max(grid[start_row][index_max -(col - 1)], grid[start_row - (rows - 1)][index_max - (col - 1)])):
                    traverse.append(1)
                    start_row -= 1
                    index_max = index_max - (col - 1)
                elif(grid[start_row][index_max - (col - 1)] > max(grid[start_row - 1][index_max - (col - 1)], grid[start_row - (rows - 1)][index_max - (col - 1)])):
                    traverse.append(0)
                    index_max = index_max - (col - 1)
                elif(grid[start_row - (rows - 1)][index_max - (col - 1)] > max(grid[start_row - 1][index_max - (col - 1)], grid[start_row][index_max - (col - 1)])):
                    traverse.append(-1)
                    start_row = start_row - (rows - 1)
                    index_max = index_max - (col - 1)
                else:
                    if(rows - (start_row + 1) < (start_row - 1)):
                        traverse.append(-1)
                        start_row += 1
                        index_max += 1
                    elif(start_row - 1 < rows - (start_row + 1)):
                        traverse.append(1)
                        start_row -= 1
                        index_max -= 1
                
            #top right corner OOB
            elif(start_row - 1 < 0 and index_max + 1 >= col):
                grid[start_row][index_max] += max(grid[start_row + (rows - 1)][index_max - (col - 1)], max(grid[start_row][index_max - (col - 1)], grid[start_row + 1][index_max - (col - 1)]))

                

                if(grid[start_row + (rows - 1)][index_max - (col - 1)] > max(grid[start_row][index_max -(col - 1)], grid[start_row + 1][index_max - (col - 1)])):
                    traverse.append(1)
                    start_row = start_row + (rows - 1)
                    index_max = index_max - (col - 1)
                elif(grid[start_row][index_max - (col - 1)] > max(grid[start_row + (rows - 1)][index_max - (col - 1)], grid[start_row + 1][index_max - (col - 1)])):
                    traverse.append(0)
                    index_max = index_max - (col - 1)
                elif(grid[start_row + 1][index_max - (col - 1)] > max(grid[start_row + (rows - 1)][index_max - (col - 1)], grid[start_row][index_max - (col - 1)])):
                    traverse.append(-1)
                    start_row += 1
                    index_max = index_max - (col - 1)
                else:
                    if(rows - (start_row + 1) < (start_row - 1)):
                        traverse.append(-1)
                        start_row += 1
                        index_max += 1
                    elif(start_row - 1 < rows - (start_row + 1)):
                        traverse.append(1)
                        start_row -= 1
                        index_max -= 1

            else:
                #bottom OOB
                if(start_row + 1 >= rows):
                    grid[start_row][index_max] += max(grid[start_row - 1][index_max + 1], max(grid[start_row][index_max + 1], grid[start_row - (rows - 1)][index_max + 1]))

                    
                    
                    if(grid[start_row - 1][index_max + 1] > max(grid[start_row][index_max + 1], grid[start_row - (rows - 1)][index_max + 1])):
                        traverse.append(1)
                        start_row -= 1
                        index_max += 1
                    elif(grid[start_row][index_max + 1] > max(grid[start_row - 1][index_max + 1], grid[start_row - (rows - 1)][index_max + 1])):
                        traverse.append(0)
                        index_max += 1
                    elif(grid[start_row - (rows - 1)][index_max + 1] > max(grid[start_row - 1][index_max + 1], grid[start_row][index_max + 1])):
                        traverse.append(-1)
                        start_row = start_row - (rows - 1)
                        index_max += 1
                    else:
                        if(rows - (start_row + 1) < (start_row - 1)):
                            traverse.append(-1)
                            start_row += 1
                            index_max += 1
                        elif(start_row - 1 < rows - (start_row + 1)):
                            traverse.append(1)
                            start_row -= 1
                            index_max -= 1
                #top OOB
                elif(start_row - 1 < 0):
                    grid[start_row][index_max] += max(grid[start_row + (rows - 1)][index_max + 1], max(grid[start_row][index_max + 1], grid[start_row + 1][index_max + 1]))

                    

                    if(grid[start_row + (rows - 1)][index_max + 1] > max(grid[start_row][index_max + 1], grid[start_row + 1][index_max + 1])):
                        traverse.append(1)
                        start_row = start_row + (rows - 1)
                        index_max += 1
                    elif(grid[start_row][index_max + 1] > max(grid[start_row + (rows - 1)][index_max + 1], grid[start_row + 1][index_max + 1])):
                        traverse.append(0)
                        index_max += 1
                    elif(grid[start_row + 1][index_max + 1] > max(grid[start_row + (rows - 1)][index_max + 1], grid[start_row][index_max + 1])):
                        traverse.append(-1)
                        start_row += 1

                        if(rows - (start_row + 1) < (start_row - 1)):
                            traverse.append(-1)
                            start_row += 1
                            index_max += 1
                        elif(start_row - 1 < rows - (start_row + 1)):
                            traverse.append(1)
                            start_row -= 1
                            index_max -= 1
                #right OOB
                elif(index_max + 1 >= col):
                    grid[start_row][index_max] += max(grid[start_row - 1][index_max - (col - 1)], max(grid[start_row][index_max - (col - 1)], grid[start_row + 1][index_max - (col - 1)]))

                    if(grid[start_row - 1][index_max - (col - 1)] > max(grid[start_row][index_max - (col - 1)], grid[start_row + 1][index_max - (col - 1)])):
                        traverse.append(1)
                        start_row -= 1
                        index_max = index_max - (col - 1)
                    elif(grid[start_row][index_max - (col - 1)] > max(grid[start_row - 1][index_max - (col - 1)], grid[start_row + 1][index_max - (col - 1)])):
                        traverse.append(0)
                        index_max = index_max - (col - 1)
                    elif(grid[start_row + 1][index_max - (col - 1)] > max(grid[start_row - 1][index_max - (col - 1)], grid[start_row][index_max - (col - 1)])):
                        traverse.append(-1)

                #no OOB

                    grid[start_row][index_max] += max(grid[start_row - 1][index_max + 1], max(grid[start_row][index_max + 1], grid[start_row + 1][index_max + 1]))


                    if(grid[start_row - 1][index_max + 1] > max(grid[start_row][index_max + 1], grid[start_row + 1][index_max + 1])):
                        traverse.append(1)

                        traverse.append(0)

    print(traverse)
