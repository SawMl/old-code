def solve(grid):
    m = len(grid)
    n = len(grid[0])

    # to store the move at the position
    # if visited[i][j] = 2 mean it was not visited
    visited = [ [2]*n for i in range(m)]
    visited[0][0]=0
    visited[m-1][0]=0

    ans = 0
    
    # for each element in grid we will check from previous element we will get max
    for col in range(1, n):
        for row in range(0, m):
            a = isVisited(row-1, col-1, visited)
            b = isVisited(row, col-1, visited)
            c = isVisited(row+1, col-1, visited)
            
            if(a == -math.inf and b == -math.inf and c == -math.inf):
                visited[row][col] = 2
            elif(a>b and a>c):
                visited[row][col] = -1
                grid[row][col] = grid[row][col] + a
            elif (b>a and b>c):
                visited[row][col] = 0
                grid[row][col] = grid[row][col] + b
            else:
                visited[row][col] = 1
                grid[row][col] = grid[row][col] + c
                
            if row == 0 or row == m-1:
                ans = max(ans, grid[row][col])
        
    # path will be max at top right or bottom right because if we have to exit from top or bottom
    print("Paths:")
    if ans == grid[0][n-1]:
        find_path(0, n-1, visited)
    if ans == grid[m-1][n-1]:
        find_path(m-1, n-1, visited)
    
    print("Total Score : "+str(ans))  
  

m = int(input("Enter number of rows : "))
n = int(input("Enter number of columns : "))

grid = [ [0]*n for i in range(m)]

for i in range(m):
    for j in range(n):
        grid[i][j] = int(input())


solve(grid)
