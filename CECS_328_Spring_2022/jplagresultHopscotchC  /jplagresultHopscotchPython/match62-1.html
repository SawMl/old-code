def solve(grid):
    steps = []
    stepsIndex = 0
    maxSum = 0
    rowSize = len(grid)
    columnSize = len(grid[0])
    maxMoves = columnSize - 1
    while stepsIndex < 2:
        for l in range(columnSize):
            tempSteps = []
            j = l
            tempSum 
= 0
            # starts from top
            if stepsIndex == 0:
                tempSteps.append(1)
                tempSteps.append(j)
                i = rowSize
            # starts from bottom
            else:
                tempSteps.append(0)
                tempSteps.append(j)
                i = rowSize - 1

            tempSum += grid[i % rowSize][j % columnSize]
            for moves in range(maxMoves):
                # 0 = right
                if grid[(rowSize + i) % rowSize][(j + 1) % columnSize] >= grid[(rowSize + i + 1) % rowSize][
                    (j + 1) % columnSize] and grid[(rowSize + i) % rowSize][(j + 1) % columnSize] >= \
                        grid[(rowSize + i + -1) % rowSize][(j + 1) % columnSize]:
                    j += 1
                    tempSteps.append(0)
                # -1 = right down
                elif grid[(rowSize + i + 1) % rowSize][(j + 1) % columnSize] >= grid[(rowSize + i) % rowSize][
                    (j + 1) % columnSize] and grid[(rowSize + i + 1) % rowSize][(j + 1) % columnSize] >= \
                        grid[(rowSize + i + -1) % rowSize][(j + 1) % columnSize]:
                    i += 1
                    j += 1
                    tempSteps.append(-1)
                # 1 = right up
                elif grid[(rowSize + i - 1) % rowSize][(j + 1) % columnSize] >= grid[(rowSize + i) % rowSize][
                    (j + 1) % columnSize] and grid[(rowSize + i + -1) % rowSize][(j + 1) % columnSize] >= \
                        grid[(rowSize + i + 1) % rowSize][(j + 1) % columnSize]:
                    i -= 1
                    j += 1
                    tempSteps.append(1)

                tempSum += grid[i % rowSize][j % columnSize]
            # check if it ends up at top or bottom row
            if i % rowSize == 0 or i % rowSize == rowSize - 1:
                if tempSum > maxSum:
                    maxSum = tempSum
                    steps = tempSteps

        stepsIndex += 1

    print("Total Score:", maxSum)
    print(steps)
    return steps
