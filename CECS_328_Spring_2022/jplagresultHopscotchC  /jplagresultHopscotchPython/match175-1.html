def solve(grid):
    # print(grid)
    direction = True
    output = []
    inverse_grid = []
    final_output = []
    Sum = []
    for i in range(len(grid)-1, -1,-1):
        inverse_grid.append(grid[i])

    for q in range(2):
        if q == 1:
            grid = inverse_grid
        for y in range(len(grid[0])):
            steps_remaining = len(grid[0]) - 1
            x = 0
            elements = []

            inner = []
            if(q == 0):
                inner.append(1)
            else:
                inner.append(0)
            inner.append(y)
            
            current_element = grid[x][y]
            elements.append(current_element)
            # Sum = current_element

            while(steps_remaining != 0):
                if(x < 0):
                    if(-x < int(len(grid)/2)+1):
                        steps_for_fr = -x
                        steps_for_lr = -x - 1
                    elif(-x > int(len(grid)/2)+1):
                        steps_for_fr = len(grid) - (-x)
                        steps_for_lr = len(grid) - (-x) + 1
                    elif(-x == int(len(grid) / 2)+1):    
                        if(len(grid) % 2 == 1):
                            steps_for_fr = -x -1
                            steps_for_lr = -x -1
                        elif(len(grid) % 2 == 0):
                            steps_for_fr = -x + 1
                            steps_for_lr = -x - 1
                else:
                    if(x < int(len(grid)/2)):
                        steps_for_fr = x
                        steps_for_lr = x + 1
                    elif(x > int(len(grid)/2)):
                        steps_for_fr = len(grid) - x
                        steps_for_lr = len(grid) - x - 1
                    elif(x == int(len(grid) / 2)):    
                        if(len(grid) % 2 == 1):
                            steps_for_fr = x
                            steps_for_lr = x
                        elif(len(grid) % 2 == 0):
                            steps_for_fr = x
                            steps_for_lr = x - 1
                
                if(x + 1 == len(grid)):
                    x = 0
                if(y+1 == len(grid[0])):
                    y = -1
                if(x-1 == len(grid)*-1):
                    x = -1

                if(steps_for_fr > steps_remaining and steps_for_lr < steps_remaining) or ((x < 0 and steps_for_fr == steps_remaining and steps_for_lr < steps_remaining)):
                    right = grid[x][y+1]
                    right_down = grid[x+1][y+1]
                
                    M = max(right,right_down)

                    if(M == right):
                        current_element = right
                        elements.append(current_element)
                        y += 1
                        steps_remaining -= 1
                        inner.append(0)
                    else:
                        x += 1
                        y += 1
                        current_element = right_down
                        elements.append(current_element)
                        steps_remaining -= 1
                        inner.append(-1)
                        direction =True

                elif(steps_for_fr > steps_remaining and steps_for_lr == steps_remaining):
                    right_down = grid[x+1][y+1]
                    x += 1
                    y += 1
                    current_element = right_down
                    elements.append(current_element)
                    steps_remaining -= 1
                    inner.append(-1)
                    direction = True
                
                elif(steps_for_fr < steps_remaining and steps_for_lr > steps_remaining) or (((x >= 0 and steps_for_fr < steps_remaining and steps_for_lr == steps_remaining))):
                    right = grid[x][y+1]
                    right_up = grid[x-1][y+1]
                
                    M = max(right,right_up)

                    if(M == right):
                        current_element = right
                        elements.append(current_element)
                        y += 1
                        steps_remaining -= 1
                        inner.append(0)
                    else:
                        x -= 1
                        y += 1
                        current_element = right_up
                        elements.append(current_element)
                        steps_remaining -= 1
                        inner.append(1)
                        direction = False

                elif(steps_for_fr == steps_remaining and steps_for_lr > steps_remaining):
                    right_up = grid[x-1][y+1]
                    x -= 1
                    y += 1
                    current_element = right_up
                    elements.append(current_element)
                    steps_remaining -= 1
                    inner.append(1)
                    direction = False

                elif(steps_for_fr == steps_for_lr and steps_for_fr == steps_remaining):
                    right_down = grid[x+1][y+1]
                    right_up = grid[x-1][y+1]
                
                    M = max(right_down,right_up)

                    if(M == right_down):
                        current_element = right_down
                        elements.append(current_element)
                        x += 1
                        y += 1
                        steps_remaining -= 1
                        inner.append(-1)
                        direction = True
                    else:
                        x -= 1
                        y += 1
                        current_element = right_up
                        elements.append(current_element)
                        steps_remaining -= 1
                        inner.append(1)
                        direction 
= False

                else:
                    right = grid[x][y+1]
                    right_down = grid[x+1][y+1]
                    right_up = grid[x-1][y+1]

                    M = max(right, right_down, right_up)

                    if(M == right):
                        right = grid[x][y+1]                                                                   
                        y = y +1 
                        current_element = right
                        elements.append(current_element)
                        steps_remaining -= 1
                        inner.append(0)
                    elif(right_up == right_down):
                        if(x < 0):
                            c = x - 1
                            d = x + 1
                            if(c > d):
                                right_up = grid[x-1][y+1]
                                x = x - 1
                                y = y + 1
                                current_element = right_up
                                elements.append(current_element)
                                steps_remaining -= 1
                                inner.append(1)
                                direction =True
                                
                            else:
                                right_down = grid[x+1][y+1]
                                x = x + 1
                                y  = y + 1
                                current_element = right_down
                                elements.append(current_element)
                                steps_remaining -= 1
                                inner.append(-1)
                                direction = False
                        else:
                            c = x - 1
                            d = x + 1
                            if(c < d):
                                x = x - 1
                                y = y + 1
                                current_element = right_up
                                elements.append(current_element)
                                steps_remaining -= 1
                                inner.append(1)
                                direction = False
                            else:
                                x = x + 1
                                y  = y + 1
                                current_element = right_down
                                elements.append(current_element)
                                steps_remaining -= 1
                                inner.append(-1)
                                direction = True
                    else:
                        if(M == right_up):
                            x = x - 1
                            y = y + 1
                            current_element = right_up
                            elements.append(current_element)
                            steps_remaining -= 1
                            inner.append(1)
                            direction = False
                        elif(M == right_down):
                            x = x +1
                            y = y +1
                            current_element = right_down
                            elements.append(current_element)
                            steps_remaining -= 1
                            inner.append(-1)
                            direction = True

                # Sum += current_element
            inner_sum = sum(elements)
            
            if(inner_sum < 0):
                inner_sum = inner_sum * (-1)
            # inner.append(inner_sum)
            # inner.append(x)
            Sum.append(inner_sum)

            # output.append(inner)
            if(q == 1):
                for r in range(2,len(inner)):
                    inner[r] = inner[r]*(-1)
            if((x == -1) or (x == 0) or (x == len(grid)-1) or (x == len(grid) * -1)):
                output.append(inner)
    max_sum = max(Sum)
    max_index = 0
    for p in range(len(output)):
        if(Sum[p] == max_sum):
            max_index = p
            final_output.append(output[max_index])
    # print(Sum)
    # print(output)
    return final_output[0]
