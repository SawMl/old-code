def biggest_number_on_the_right(array, index_row, index_col, i):
    num = array[index_row][index_col]

    right = 0
    up_rigth = 0
    down_rigth = 0


    if index_col == len(array[0]) - 1:
        index_col = 0
    else:
        index_col = index_col + 1
    traverse = False
    traverse1 = False


    right = array[index_row][index_col]
    if index_row == 0:
        up_right = array[len(array) - 1][index_col]
        traverse = True
    else:
        up_right = array[index_row - 1][index_col]

    if index_row == len(array) - 1:
        traverse1 = True
        down_rigth = array[0][index_col]
    else:
        down_rigth = array[index_row + 1][index_col]

    if (index_row - (i - 1) != 0 and index_row - (i - 1) != -1)  and (index_row + ( i - 1) != len(array)-1 and index_row + ( i - 1) != len(array)):
        if right > up_right and right > down_rigth:
            return [right, 0, False, index_row, index_col]
        elif up_right >= right and up_right > down_rigth:
            if traverse is True:
                return [up_right, 1, True, len(array) - 1, index_col]
            else:
                return [up_right, 1, False, index_row - 1, index_col]
        elif down_rigth >= right and down_rigth > up_right:

            if traverse1 is True:
                return [down_rigth, -1, True, 0, index_col]
            else:
                return [down_rigth, -1, False, index_row + 1, index_col]
        elif up_right == down_rigth:
            if index_row > len(array) / 2:
                if traverse1 is True:
                    return [down_rigth, -1, True, 0, index_col]
                else:
                    return [down_rigth, -1, False, index_row + 1, index_col ]
            else:
                if traverse is True:
                    return [up_right, 1, True, len(array) - 1, index_col]
                else:
                    return [up_right, 1, False, index_row - 1, index_col]
    elif index_row - (i - 1) == 0 or index_row - (i - 1) == -1:
        if index_row == 0:
            if right > up_right:
                return [right, 0, False, index_row, index_col]
        if traverse is True:
            return [up_right, 1, True, len(array) - 1, index_col]
        else:
            return [up_right, 1, False, index_row - 1, index_col]
    elif index_row + ( i - 1) != len(array)-1 or index_row + ( i - 1) != len(array):
        if index_row == len(array) - 1:
            if right > down_rigth:
                return [right, 0, False, index_row, index_col]
        if traverse1 is True:
            return [down_rigth, -1, True, 0, index_col]
        else:
            return [down_rigth, -1, False, index_row + 1, index_col]
