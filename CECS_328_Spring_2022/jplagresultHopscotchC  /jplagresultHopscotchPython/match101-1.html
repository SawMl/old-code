def solve(grid):
  max = 0
  column_pos = 0
  path = []
  
for i in range(len(grid[0])):
    if (grid[0][i] > max):
      max = grid[0][i]
      column_pos = i
  start_pos = 1
  path.append(start_pos)
  path.append(column_pos)
  if start_pos == 0:
    row_pos = len(grid) - 1
  else:
    row_pos = 0
  next_row_pos = row_pos
  next_column_pos = column_pos
  next_nums = []
  for i in range(0, len(grid[0]) - 1):
    next_nums = []
    next_row_pos = row_pos
    next_column_pos += 1
    if next_column_pos == len(grid[0]): #ADDING RIGHT
      next_column_pos = 0
    next_nums.append(grid[next_row_pos][next_column_pos])
    next_row_pos = row_pos - 1
    if next_row_pos < 0: #ADDING TOP RIGHT
      next_row_pos = len(grid) - 1
    if next_column_pos == len(grid[0]):
      next_column_pos = 0
    next_nums.append(grid[next_row_pos][next_column_pos])
    next_row_pos = row_pos + 1
    if next_row_pos == len(grid): #ADDING BOTTOM RIGHT
      next_row_pos = 0
    if next_column_pos == len(grid[0]):
      next_column_pos = 0
    next_nums.append(grid[next_row_pos][next_column_pos])
    max = 0
    next_spot = 0
    for i in range(len(next_nums)):
      if (next_nums[i] >= max):
        max = next_nums[i]
        next_spot = i
    if i == len(grid[0]) - 2:
      if (row_pos == 1):
        next_spot = 1
      if (row_pos == 3):
        next_spot = 2
    if next_spot == 0:
      path.append(0)
    if next_spot == 1:
      row_pos -= 1
      if row_pos < 0:
        row_pos = len(grid) - 1
      path.append(1)
    if next_spot == 2:
      row_pos += 1
      if row_pos == len(grid):
        row_pos = 0
      path.append(-1)
  return path
