def solve(grid):
    columns = len(grid[0])
    rows = len(grid)
    firstMax = max(grid[0])
    posFirstMax = grid[0].index(firstMax)
    sampleAnswer = [1, len(grid) - 1]
    for i in range(columns-1):
        sampleAnswer.append(1)
    event = 0
    
    #FIRST Iteration
    a = 0
    b = posFirstMax
    one = grid[-1][posFirstMax + 1] 
    zero = grid[0][posFirstMax + 1] 
    negOne = grid[1][posFirstMax + 1] 
    options = [one, zero, negOne]
    optionsDict = {one: 1, zero: 0, negOne: -1}
    newMax = max(options)
    finalAnswer = [1]
    finalAnswer.append(grid[0].index(firstMax))
    result = optionsDict[newMax]
    finalAnswer.append(result)
    
    #SECOND Iteration
    if result == -1:
        a += 1
    elif result == 1:
        a -=1
    b += 1
    #if hits ceiling
    if (b+1 == columns):
        one = grid[a-1][0] 
        zero = grid[a][0] 
        negOne 
= grid[a+1][0] 
    else:
        one = grid[a-1][b+1] 
        zero = grid[a][b+1] 
        negOne = grid[a+1][b+1] 
    options = [one, zero, negOne]
    optionsDict = {one: 1, zero: 0, negOne: -1}
    newMax = max(options)
    finalAnswer.append(optionsDict[newMax])
    
    #THIRD Iteration
    if result == -1:
        a += 1
    elif result == 1:
        a -=1
    b += 1
    if (b >= columns):
        a = rows -1
        b = 0
        one = grid[a-1][b+1] 
        zero = grid[a][b+1] 
        negOne = grid[0][b+1] 
        event = 1
    elif (b+1 == columns):
        one = grid[a-1][0] 
        zero = grid[a][0] 
        negOne = grid[a+1][0] 
    else:
        one = grid[a-1][b+1] 
        zero = grid[a][b+1] 
        negOne = grid[a+1][b+1] 
    options = [one, zero, negOne]
    optionsDict = {one: 1, zero: 0, negOne: -1}
    newMax = max(options)
    finalAnswer.append(optionsDict[newMax])
    
    #FOURTH Iteration
    if (len(finalAnswer) < columns + 1 ):
        if result == -1:
            a += 1
        elif result == 1:
            a -=1
        if event:
            b = 0
        else:
            b += 1
        #if hits Right Top/Bottom corner
        if (b >= columns or event):
            a = rows -1
            b = 0
            one = grid[a-1][b+1] 
            zero = grid[a][b+1] 
            negOne = grid[0][b+1] 
        elif (b+1 == columns):
            one = grid[a-1][0] 
            zero = grid[a][0] 
            negOne = grid[a+1][0] 
        else:
            one = grid[a-1][b+1] 
            zero = grid[a][b+1] 
            negOne = grid[a+1][b+1] 
        options = [one, zero, negOne]
        optionsDict = {one: 1, zero: 0, negOne: -1}
        newMax = max(options)
        finalAnswer.append(optionsDict[newMax])
    return finalAnswer
