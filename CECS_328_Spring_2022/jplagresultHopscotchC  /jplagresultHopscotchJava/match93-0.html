import java.util.ArrayList;

public class StudentSolver {
  public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {
    ArrayList<Integer> solution = new ArrayList<Integer>();
    int gridRows = grid.size();
    int gridCols = grid.get(0).size();
    
    double rand = Math.random();
    if (rand > 0.5) {
      solution.add(0);
    }
    else {
      solution.add(gridCols-1);
    }

    int col = (int)(Math.random()*gridCols);
    solution.add(col);

    for (int i = 0; i < gridCols-1; i++) {
      rand = Math.random();
      if (rand > 0.66) {
        solution.add(0);
      }
      else if (rand > 0.33) {
        solution.add(1);
      }
      else {
        solution.add(-1);
      }
    }
    
    return solution;
  }
}
