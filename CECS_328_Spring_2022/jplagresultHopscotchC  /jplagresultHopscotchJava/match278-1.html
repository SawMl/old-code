import java.util.ArrayList;
import java.util.Random;

class StudentSolver {

    public static void main(String[] args) {
    
    }

    public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {

        ArrayList<Integer> ansTop = new ArrayList<Integer>(grid.get(0).size() + 1);
        ArrayList<Integer> ansBot = new ArrayList<Integer>(
grid.get(0).size() + 1);


        
        int maxTop = grid.get(0).get(0);

        for (int i = 0; i < 1; i++) {
            for (int j = 0; j < grid.get(i).size(); j++) {
                if (maxTop < grid.get(i).get(j)) {
                    maxTop = grid.get(i).get(j);
                }
            }
        }
        int maxTopIndex = grid.get(0).indexOf(maxTop);
        ansTop.add(1);
        ansTop.add(maxTopIndex);

        int row = 0;
        int col = maxTopIndex;





            for (int i = 0; i < grid.get(0).size() - 1; i++) {

                Random random = new Random();
                int rand1 = random.nextInt(2 - 1) + 1;
                int rand2 = random.nextInt(3 - 1) + 1;

                System.out.println("check row: " + row);

                int up = 0;
                int right = 0;
                int down = 0;
                
                if (col == grid.get(0).size() - 1) {
                    col = -1;
                }

                if (row == 0) {
                    up = grid.get(grid.size() - 1).get(col + 1);

                } else if (row != 0) {
                    up = grid.get(row - 1).get(col + 1);
                }

                if (row == grid.size() - 1) {
                    down = grid.get(-1 + 1).get(col + 1);
                } else if (row != grid.size() - 1) {
                    down = grid.get(row + 1).get(col + 1);
                }
                right = grid.get(row).get(col + 1);

                
                int checkMax = 0;
















                if (up == down && up > right) {
                    if (rand1 == 1) {
                        ansTop.add(-1);
                        checkMax = down;
                    } else {
                        ansTop.add(1);
                        checkMax = up;
                    }
                } else if (up == right && up > down) {
                    if (rand1 == 1) {
                        ansTop.add(1);
                        checkMax = up;
                    } else if (rand1 == 2) {
                        ansTop.add(0);
                        checkMax = right;
                    }
                } else if (down == right && down > up) {
                    if (rand1 == 1) {
                        ansTop.add(-1);
                        checkMax = down;
                    } else {
                        ansTop.add(0);
                        checkMax = right;
                    }
                } else if (down == right && down == up) {
                    if (rand2 == 1) {
                        ansTop.add(-1);
                        checkMax = down;
                    } else if (rand2 == 2) {
                        ansTop.add(0);
                        checkMax = right;
                    } else if (rand2 == 3) {
                        ansTop.add(1);
                        checkMax = up;
                    }
                } else if (right > up && right > down) {
                    ansTop.add(0);
                    checkMax = right;
                } else if (up > right && up > down) {
                    ansTop.add(1);
                    checkMax = up;
                } else if (down > up && down > right) {
                    ansTop.add(-1);
                    checkMax = down;
                }

                System.out.println("CheckMax: " + checkMax);
                
                if (row == 0 && checkMax == up) {
                    row = grid.size() - 1;
                } else if (row != 0 && checkMax == up) {
                    row--;
                }

                if (row == grid.size() - 1 && checkMax == down) {
                    row = 0;
                } else if (row != grid.size() - 1 && checkMax == down) {
                    row++;
                }

                System.out.println(up);
                System.out.println(right);
                System.out.println(down);
                System.out.println("");


                col++;
                System.out.println(ansTop);
            }











        return ansTop;
    }
}
