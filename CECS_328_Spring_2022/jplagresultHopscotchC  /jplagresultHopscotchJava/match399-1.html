    public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {
        ArrayList<Integer> topCurrentPath = new ArrayList<Integer>();
        ArrayList<Integer> botCurrentPath = new ArrayList<Integer>();
        ArrayList<Integer> greatestCurrentPath = new ArrayList<Integer>();

        int spotTop = 0;
        int highestNumTop = 0;
        for(int i = 0; i < 
grid.get(0).size(); i++) {
            if(highestNumTop < grid.get(0).get(i)) {
                highestNumTop = grid.get(0).get(i);
                spotTop = i;
            }
        }

        int spotBot = 0;
        int highestNumBot = 0;
        for(int i = 0; i < grid.get(grid.size()-1).size(); i++) {
            if(highestNumBot < grid.get(grid.size()-1).get(i)) {
                highestNumBot = grid.get(grid.size()-1).get(i);
                spotBot = i;
            }
        }

        int topCurrentValue = grid.get(0).get(spotTop);
        int botCurrentValue = grid.get(grid.size()-1).get(spotBot);

        topCurrentPath.add(1);
        topCurrentPath.add(spotTop);

        botCurrentPath.add(0);
        botCurrentPath.add(spotBot);

        findEnd(grid, topCurrentPath, 0, spotTop, topCurrentValue);
        findEnd(grid, botCurrentPath, grid.size()-1, spotBot, botCurrentValue);

        if(topCurrentPath.get(topCurrentPath.size()-1) >= botCurrentPath.get(botCurrentPath.size()-1)) {
            topCurrentPath.remove(topCurrentPath.size()-1);
            greatestCurrentPath = topCurrentPath; 
        }
        else {
            botCurrentPath.remove(botCurrentPath.size()-1);
            greatestCurrentPath = botCurrentPath; 
        }
        return greatestCurrentPath;
    }
    
    public static ArrayList<Integer> findEnd(ArrayList<ArrayList<Integer>> grid, ArrayList<Integer> currentPath, int currentRow, int currentColumn, int currentValue) {
        int maxRow = grid.size()-1; 
        int maxColumn = grid.get(0).size()-1; 
        
int upValue = 0;
        int straightValue = 0;
        int downValue = 0;
        int upRow = 0;
        int upColumn = 0;
        int straightRow = 0;
        int straightColumn = 0;
        int downRow = 0;
        int downColumn = 0;

        for(int i = 0; i < grid.get(0).size()-2; i++) {
            
            
if((maxColumn - (currentColumn + 1) >= 0)) {
                upColumn = currentColumn + 1;
            }
            else {
                upColumn = 0;
            }
            if((maxRow - (currentRow - 1) <= maxRow)) {
                upRow = currentRow - 1;
            }
            else {
                upRow = maxRow;
            }
            upValue = grid.get(upRow).get(upColumn);

            
            straightRow = currentRow;
            
if((maxColumn - (currentColumn + 1) >= 0)) {
                straightColumn = currentColumn + 1;
            }
            else {
                straightColumn = 0;
            }
            straightValue = grid.get(straightRow).get(straightColumn);

            
            
if((maxColumn - (currentColumn + 1) >= 0)) {
                downColumn = currentColumn + 1;
            }
            else {
                downColumn = 0;
            }
            if((maxRow - (currentRow + 1) >= 0)) {
                downRow = currentRow + 1;
            }
            else {
                downRow = 0;
            }
            downValue = grid.get(downRow).get(downColumn);

            
            if(upValue >= straightValue  && upValue >= downValue) {
                currentRow = upRow;
                currentColumn = upColumn;
                currentPath.add(1);
                currentValue += grid.get(currentRow).get(currentColumn);
            }
            else if(straightValue >= upValue && straightValue >= downValue) {
                currentRow = straightRow;
                currentColumn = straightColumn;
                currentPath.add(0);
                currentValue += grid.get(currentRow).get(currentColumn);
            }
            else {
                currentRow = downRow;
                currentColumn = downColumn;
                currentPath.add(-1);
                currentValue += grid.get(currentRow).get(currentColumn);
            }
        }

        
        if(currentRow == 1) {
            
if((maxColumn - (currentColumn + 1) >= 0)) {
                upColumn = currentColumn + 1;
            }
            else {
                upColumn = 0;
            }
            if((maxRow - (currentRow - 1) <= maxRow)) {
                upRow = currentRow - 1;
            }
            else {
                upRow = maxRow;
            }
            upValue = grid.get(upRow).get(upColumn);
            currentRow = upRow;
            currentColumn = upColumn;
            
currentPath.add(1);
            currentValue += grid.get(currentRow).get(currentColumn);
        }
        
        else if(currentRow == grid.size()-2) {
            if((maxColumn - (currentColumn + 1) >= 0)) {
                downColumn = currentColumn + 1;
            }
            else {
                downColumn = 0;
            }
            if((maxRow - (currentRow + 1) >= 0)) {
                downRow = currentRow + 1;
            }
            else {
                downRow = 0;
            }
            downValue = grid.get(downRow).get(downColumn);
            currentRow = downRow;
            currentColumn = downColumn;
            currentPath.add(-1);
            currentValue += grid.get(currentRow).get(currentColumn);
        }
        
        else if(currentRow == 0 || currentRow == grid.size()-1) {
            straightRow = currentRow;
            if((maxColumn - (currentColumn + 1) >= 0)) {
                straightColumn = currentColumn + 1;
            }
            else {
                straightColumn = 0;
            }
            straightValue = grid.get(straightRow).get(straightColumn);
            currentRow = straightRow;
                currentColumn = straightColumn;
                currentPath.add(0);
                currentValue += grid.get(currentRow).get(currentColumn);
        }
        
        else {
            if((maxColumn - (currentColumn + 1) >= 0)) {
                upColumn = currentColumn + 1;
            }
            else {
                upColumn = 0;
            }
            if((maxRow - (currentRow - 1) <= maxRow)) {
                upRow = currentRow - 1;
            }
            else {
                upRow = maxRow;
            }
            upValue = grid.get(upRow).get(upColumn);

            
            straightRow = currentRow;
            if((maxColumn - (currentColumn + 1) >= 0)) {
                straightColumn = currentColumn + 1;
            }
            else {
                straightColumn = 0;
            }
            straightValue = grid.get(straightRow).get(straightColumn);

            
            if((maxColumn - (currentColumn + 1) >= 0)) {
                downColumn = currentColumn + 1;
            }
            else {
                downColumn = 0;
            }
            if((maxRow - (currentRow + 1) >= 0)) {
                downRow = currentRow + 1;
            }
            else {
                downRow = 0;
            }
            downValue = grid.get(downRow).get(downColumn);
        }
        currentPath.add(currentValue);
        return currentPath;
    }
