import java.util.ArrayList;

public class StudentSolver {
	public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>>grid){
			
		
		ArrayList<Integer> answers = new ArrayList<Integer>();
		
		ArrayList<Integer> path = new ArrayList<Integer>();
		
		
		int daSize = 0;
		
		
		int greatest = 0;
		
		
		int daRow = 0;
		int daCol = 0;
		
		
		for(ArrayList<Integer> row : grid) {
				System.out.println(row);
		}
		
		
	   /* 
	    * STARTING POINT
	    */
		
		int top = 0;	
		int tIndex = 0;
		ArrayList<Integer> topRow = grid.get(0);
		for (int i = 0; i < topRow.size(); i++) {
			int n = topRow.get(i);
			if(n > top) {
				top = n;	
				tIndex = i;
			}
		}
		

		
		int bott = 0; 
		int bIndex = 0;
		ArrayList<Integer> bottRow = grid.get(grid.size()-1);
		for (int i = 0; i < bottRow.size(); i++) {
			int n = bottRow.get(i);
			if(n > bott) {
				bott = n;
				bIndex = i;
			}
		}
		
		
		if (top >= bot
t) {
			greatest = top;
		}else {
			greatest = (bott*-1);
		}
	
		
		if(greatest >= 0) {
			answers.add(1); 
			answers.add(tIndex);
			daRow = 0;
			daCol = tIndex;
		} else {
			answers.add(0); 
			answers.add(bIndex);
			daRow = (grid.size()-1);
			daCol = bIndex;
		}
		
		path.add(grid.get(daRow).get(daCol));
		
			
	   		
		int right = 0;
		int up = 0;
		int uRow = 0; 
		int down = 0;
		int dRow = 0; 
		
		
		
		daSize = grid.get(0).size();
		while(daSize > 1) {	
			
			ArrayList<Integer> atRow = grid.get(daRow);

			
			
			System.out.println("at " + atRow.get(daCol));


			daCol+= 1;
			
			
			if(daCol < atRow.size()) {
				right = atRow.get(daCol);
			} else {
				daCol = 0;
				right = atRow.get(daCol);
			} 
			System.out.print("right "+ right+", ");
			
			
			
			if(daRow > 0) {
				uRow = daRow - 1;
				ArrayList<Integer> upRow = grid.get(uRow);
				up = upRow.get(daCol);
			} else {
				uRow = (grid.size()-1);
				ArrayList<Integer> upRow = grid.get(uRow);
				up = upRow.get(daCol);			
			} System.out.print("up "+ up+", ");
			
			
			
			
			if(daRow < (grid.size()-1)) {
				dRow = daRow + 1;
				ArrayList<Integer> downRow = grid.get(dRow);
				down = downRow.get(daCol);
			} else {
				dRow = 0;
				ArrayList<Integer> downRow = grid.get(dRow);
				down = downRow.get(daCol);
			} System.out.print("down " + down+ "\n");
			
			
			
			if (right >= up && right >= down) {
				answers.add(0);
				
				System.out.println("move right");
				
			} else if (up >= right && up >= down) {
				answers.add(1);
				daRow = uRow;
				System.out.println("move up");
			} else {
				answers.add(-1);
				daRow = dRow;
				System.out.println("move down");
			}
			
			
			if (daRow == 0) {
				System.out.println("at top");
				path.add(grid.get(daRow).get(daCol));
			} else if(daRow == (grid.size()-1)) {
				System.out.println("at bottom");
				path.add((grid.get(daRow).get(daCol))*-1);
			} else {
				System.out.println("at middle");
				path.add(0);
			}
			
				
			


			
			System.out.println("\n");
			for(ArrayList<Integer> row : grid) {
					System.out.println(row);
			}
			
			daSize -=1;
		}
			
		
		
			
		
		System.out.println("\npath: " +path);
		
		System.out.println("RETURN: " +answers);
		return answers;
	}	
}
