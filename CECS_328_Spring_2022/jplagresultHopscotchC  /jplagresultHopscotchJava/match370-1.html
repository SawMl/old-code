import java.lang.reflect.Array;

public class StudentSolver {

	public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>>grid)
	 {
		ArrayList<Integer> answer = null;
		ArrayList<ArrayList<Integer>> solutiona = new ArrayList<ArrayList<Integer>>();
		
		firstCol(grid,solutiona,largestCol(grid));
		for(int n = 0; n<grid.get(0).size()-1 ;n++) {
		
		solutiona = addSol(grid,solutiona);
		}
		for(int i = 0; i<solutiona.size();i++) {
			if(solutiona.get(i).get(2) != 0 && solutiona.get(i).get(2)!=grid.size()-1) {
				solutiona.remove(i);
				i--;
			}
		}
		answer = solutiona.get(findSol(solutiona,grid));
		answer.remove(2);
		answer.remove(2);
		
		return answer;
		
	 }
	public static int findSol(ArrayList<ArrayList<Integer>>solution,ArrayList<ArrayList<Integer>>grid) {
		int largest = 0;
		int place =  0;
		for(int i = 0; i < solution.size();i++) {
			int total = 0;
			int row = solution.get(i).get(0);
			int col = solution.get(i).get(1);
			if(row == 1)
				row =0;
			else 
				row = grid.size()-1;
			 total += grid.get(row).get(col);
			for(int x = 4;x<solution.get(i).size();x++) {
				col++;
				if (col ==grid.get(0).size())
					col = 0;
				if(solution.get(i).get(x) == 1) {
					row = row -1;
					if (row ==-1)
						row = grid.size()-1;
				
				}
				else if (solution.get(i).get(x) == -1)
					row = (row +1)%grid.size();
				total +=grid.get(row).get(col);
				
			}
			if(total>largest) {
				largest = total;
				place = i;
			}
		}
		return place;
	}
	public static int largestCol(ArrayList<ArrayList<Integer>>grid) {
		int biggest = 0;
		int place = 0;
		for(int c = 0; c<grid.get(0).size();c++) {
			
		int total = 0;
		for (int row = 0; row < grid.size();row++) {
			total += grid.get(row).get(c);
		}
		if (total>biggest) {
			biggest = total;
			place = c;
		}
		}
		return place;
	}
	public static ArrayList<ArrayList<Integer>> addSol(ArrayList<ArrayList<Integer>>grid,ArrayList<ArrayList<Integer>>solution) {
		
		ArrayList<ArrayList<Integer>> answer = new ArrayList<ArrayList<Integer>>();
			int counter= 0;
			for(int i = 0; i<solution.size();i++) {
			
			boolean flagged[] = new boolean[3];
			int cr = solution.get(i).get(2);
			int cc = solution.get(i).get(3);
			int num = grid.get(cr).get(cc);
			

















			for(int z = 0; z<3;z++) {
				if (!flagged[z]) {
					
					ArrayList<Integer>add = new ArrayList<Integer>(solution.get(i));
					addNumber(add,z,grid);
					answer.add(add);
				}
			}
			}
			return answer;
	}
	public static void addNumber(ArrayList<Integer>solutions,int num,ArrayList<ArrayList<Integer>>grid) {
		num = num -1;
		int r = grid.size();
		int c = grid.get(0).size();
		solutions.add(num); 
		
		int row = solutions.get(2) - num;
		if (row == -1) {
			row = r-1;
		}
		else if(row ==r)
			row = 0;
			
		int col = solutions.get(3)+1;
		if(col == c)
			col = 0;
			
		solutions.set(
2, row);
		solutions.set(3, col);
	}
	public static void firstCol(ArrayList<ArrayList<Integer>>grid,ArrayList<ArrayList<Integer>>solutions,int col) {
		
		solutions.add(new ArrayList<Integer>(Arrays.asList(1,col,0,col)));
		solutions.add(new ArrayList<Integer>(Arrays.asList(0,col,grid.size()-1,col)));
		
	}
	public static void print(ArrayList<ArrayList<Integer>>grid) {
		for(int row = 0; row <grid.size();row ++) {
			for(int col = 0; col< grid.get(row).size();col++) {
				System.out.print(grid.get(row).get(col));
			}
			System.out.println();
		}
	}











}
