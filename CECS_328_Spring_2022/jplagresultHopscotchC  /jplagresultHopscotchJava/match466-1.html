import java.util.ArrayList;

public class StudentSolver{
  
  public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>>grid){
    ArrayList<Integer> answer = new ArrayList<Integer>();
    int position_number = 0;
    int number_index = 0;
    int movement = -1;
    int limit_of_movements = grid.get(0).size()-1;
    int top_num = 0;
    int bot_num = 0;
    for (int i = 0; i< grid.get(0).size(); i++){
      top_num += grid.get(0).get(i);
      bot_num += grid.get(grid.size()-1).get(i);      
    }
    
    if (top_num > bot_num){
      answer.add(0);
      position_number = grid.size()-1;
    }else{
      answer.add(1);
    }
    int max_val = 0;
    for( int i = 0; i < grid.get(0).size(); i++){     
      if (grid.get(position_number).get(i) > max_val){
        max_val = grid.get(position_number).get(i);
        number_index = i;
      }
    }

    answer.add(number_index);
    
    while (limit_of_movements != 0){    
      int right_down = 0; /*grid.get(position_number+1).get(number_index+1);*/
      int right = 0; /*grid.get(position_number).get(number_index+1);*/
      int right_up = 0;
      
int maximum_row = grid.size()-1;
      int maximum_collumn = grid.get(0).size()-1;
      int row_ind = 0;
      int col_ind = 0;
      int row_ind_2 = 0;
      int col_ind_2 = 0;
      int row_ind_3 = 0;
      int col_ind_3 = 0;      
      int greater = 0;
      /*if statement for right down to check if its out of bounds*/
      if (position_number == maximum_row && number_index == maximum_collumn){
        right_down = grid.get(0).get(0);
        
row_ind = 0;
        col_ind = 0;          
      }else if (position_number == maximum_row){
        row_ind = 0;
        col_ind = number_index+1;            
        right_down = grid.get(0).get(number_index+1);
      } else if(number_index == maximum_collumn){
        row_ind = position_number+1;
        col_ind = 0;            
        right_down = grid.get(position_number+1).get(0);
      }else{
        row_ind = position_number+1;
        col_ind = number_index+1;            
        right_down = grid.get(position_number+1).get(number_index+1);
      }
      /*if statement right to check if it hits out of bounds*/
      if(number_index == maximum_collumn){
        row_ind_2 = position_number;
        col_ind_2 = 0;    
        right = grid.get(position_number).get(0);
      }else{
        row_ind_2 = position_number;
        col_ind_2 = number_index+1;  
        right = grid.get(position_number).get(number_index+1);
      }
      /*compares right and right down to see which is the higher value*/
      if (right > right_down){
        greater = right;
      }else if(right == right_down){
        greater = right_down;
      }else{
        greater = right_down;
      }
      /*checks if the right up position is out of bounds*/

      if (position_number == 0 && number_index == maximum_collumn){
        row_ind_3 = maximum_row;
        col_ind_3 = 0;          
        right_up = grid.get(maximum_row).get(0);
      }else if (position_number == 0){
        row_ind_3 = maximum_row;
        col_ind_3 = number_index+1;                    
        right_up = grid.get(maximum_row).get(number_index+1);
      } else if(number_index == maximum_collumn){
        row_ind_3 = position_number-1;
        col_ind_3 = 0;          
        right_up = grid.get(position_number-1).get(0);
      }else{
        row_ind_3 = position_number-1;
        col_ind_3 = number_index+1;          
        right_up = grid.get(position_number-1).get(number_index+1);
      }
      
      /*compares greater and the right_up value. */
      if(greater>right_up){
        if(greater == right){
          position_number = row_ind_2;
          number_index = col_ind_2;
          answer.add(0);
          limit_of_movements -=1;
        }else if (greater == right_down){
          position_number = row_ind;
          number_index = col_ind;
          answer.add(-1);
          limit_of_movements -=1;          
        }
      }else if(greater == right_up){
        int closer_pt = 0;
        int closer_pt2 = 0;       
        if(greater == right){
          closer_pt =  row_ind_2;
          closer_pt2 = row_ind_3;
          if(closer_pt < closer_pt2){
            position_number = row_ind_2;
            number_index = col_ind_2;
            answer.add(0);
            limit_of_movements -=1;            
          }else{
            position_number = row_ind_3;
            
number_index = col_ind_3;
            answer.add(1);
            limit_of_movements -=1;              
          }
        }else if (greater == right_down){
          closer_pt =  maximum_row - row_ind;
          closer_pt2 = row_ind_3;
          if(closer_pt <closer_pt2){
            position_number = row_ind;
            number_index = col_ind;
            answer.add(-1);
            limit_of_movements -=1;            
          }else{
            position_number = row_ind_3;
            
number_index = col_ind_3;
            answer.add(1);
            limit_of_movements -=1;     
          }    
        }        
      }
      else{
        position_number = row_ind_3;
        number_index = col_ind_3;
        answer.add(1);
        limit_of_movements -=1;        
       }
      }

    return answer;    
    }
  }
