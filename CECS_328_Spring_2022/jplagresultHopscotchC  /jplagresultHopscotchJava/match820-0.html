import java.util.ArrayList;

public class StudentSolver{

    public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {
    	int hori = grid.get(0).size();
    	int vert = grid.size();
    	int highest = 0;
    	int startcol = 0;
    	int topbot = 0;
    	ArrayList<Integer> yeah = new ArrayList<Integer>(hori+1);
    	
    	for (int ii = 0; ii<hori; ii++)
    	{
    		if (grid.get(0).get(ii) > highest)
    		{
    			startcol = ii;
    			topbot = 1;
    		}
    	}
    	for (int ii = 0; ii<hori; ii++)
    	{
    		if (grid.get(vert-1).get(ii) > highest)
    		{
    			startcol = ii;
    			topbot = 0;
    		}
    	}
    	yeah.add(topbot);
    	yeah.add(startcol);
    	if (topbot == 0)
    		topbot = vert-1;
    	else
    		topbot = 0;
    	int check = 0;
    	for (check=check; check<hori; check++) {
    		int jjmod = (startcol+1)%hori;
			int tbmod = (topbot+1)%vert;
    		if (check < hori/2 ) {
    			
	    		int above = grid.get(tbmod).get(jjmod);
	    		int next = grid.get(tbmod).get(jjmod);
	    		int below = grid.get(tbmod).get(jjmod);
	    		if (whatdo(above, next, below) == 1)
	    		{
	    			yeah.add(1);
	    			topbot = (topbot+1)%vert;
	    		}
	    		else if (whatdo(above,next,below) == 0)
	    		{
	    			yeah.add(0);
	    		}
	    		else
	    		{
	    			yeah.add(-1);
	    			topbot = (topbot-1)%vert;
	    		}
	    		
    	}
    		
    		else if (check >= hori/2){
    			if (topbot == 0 || topbot == vert-1)
    			{
    				yeah.add(0);
    			}
    			if (vert-topbot-1 <= vert/2)
    			{
    				yeah.add(-1);
    				topbot = (topbot-1)%vert;
    			}
    			else
    			{
    				yeah.add(1);
    				topbot = (topbot+1)%vert;
    			}
    		}
    		startcol++;
    		
    	}
    	ArrayList<Integer> test1 = new ArrayList<Integer>(hori+1);
    	return test1;
    }
    
    public static int whatdo(int above, int next, int below)
    {
    	if (above > next) {
    		if (above > below)
    			return 1;
    	}
    	else if (below > next)
    	{
    		return -1;
    	}

    	return 0;
    	}
}
