    public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {
        ArrayList<Integer> mystery = new ArrayList<>();
        int highest = MIN_VALUE; 
        Point p = new Point(); 
        int score = 0;
        int ROWS = grid.size();
        int COLS = grid.get(0).size();

        
        
        
        for (int i = 0; i < ROWS; i = grid.size() - 1) {
            for (int j = 0; j < COLS; j++) {
                if (grid.get(i).get(j) > highest) {
                    highest = grid.get(i).get(j);
                    p.setLocation(i, j);
                }
            }
        }
        if (p.y == 0) {
            mystery.add(1);
            mystery.add(p.x);
        } else {
            mystery.add(0);
            mystery.add(p.x);
        }

        
        
        

        Point h = new Point(); 
        int cols = 0;
        int direction = 0;

        /*
         * 0 - right
         * 1 - right up
         * -1 - right down
         */
        while (cols < ROWS) {
            highest = MIN_VALUE;
            
            if (p.x == ROWS - 1) {
                if (grid.get(0).get(p.y) > highest) {
                    highest = grid.get(0).get(p.y);
                    h.setLocation(0, p.y);
                    direction = 0;
                }
            } else if (grid.get(p.x + 1).get(p.y) > highest) {
                highest = grid.get(p.x + 1).get(p.y);
                h.setLocation(p.x + 1, p.y);
                direction = 0;
            }

            
            if (p.x == ROWS - 1 && p.y == 0) {
                if (grid.get(0).get(COLS - 1) > highest) {
                    highest = grid.get(0).get(COLS - 1);
                    h.setLocation(0, COLS - 1);
                    direction = 1;
                }
            } else if (p.y == 0) {
                if (grid.get(p.x + 1).get(COLS - 1) > highest) {
                    highest = grid.get(p.x + 1).get(COLS - 1);
                    h.setLocation(p.x + 1, COLS - 1);
                    direction = 1;
                }
            } else if (p.x == ROWS - 1) {
                if (grid.get(0).get(p.y - 1) > highest) {
                    highest = grid.get(0).get(p.y - 1);
                    h.setLocation(0, p.y - 1);
                    direction = 1;
                }
            } else if (grid.get(p.x + 1).get(p.y - 1) > highest) {
                highest = grid.get(p.x + 1).get(p.y - 1);
                h.setLocation(p.x + 1, p.y - 1);
                direction = 1;
            }

            
            if (p.x == ROWS - 1 && p.y == COLS - 1) {
                if (grid.get(0).get(0) > highest) {
                    highest = grid.get(0).get(0);
                    h.setLocation(0, 0);
                    direction = -1;
                }
            }
            else if (p.y == COLS - 1) {
                if (grid.get(p.x + 1).get(0) > highest) {
                    highest = grid.get(p.x + 1).get(0);
                    h.setLocation(p.x + 1, 0);
                    direction = -1;
                }
            }
            else if (p.x == ROWS - 1) {
                if (grid.get(0).get(p.y + 1) > highest) {
                    highest = grid.get(0).get(p.y + 1);
                    h.setLocation(0, p.y + 1);
                    direction = -1;
                }
            }
            else if (grid.get(p.x + 1).get(p.y + 1) > highest) {
                highest = grid.get(p.x + 1).get(p.y + 1);
                h.setLocation(p.x + 1, p.y + 1);
                direction = -1;
            }

            p.setLocation(h);
            mystery.add(direction);
            cols++;
        }

        return mystery;
    }
}
