import java.util.*;
public class StudentSolver{
    public static void main(String[] args){
        ArrayList<ArrayList<Integer>>grid = new ArrayList<ArrayList<Integer>>();
        Random rand = new Random(); 
        int m = 5, n = 6;
        for (int i = 0; i < m; i++) {
            ArrayList<Integer> row = new ArrayList<Integer>();
        for (int j = 0; j < n; j++) {
            int num = rand.nextInt(10); 
            row.add(num);
                
}
            grid.add(row);
            System.out.println(row);
            }
        System.out.println(solve(grid));
    }

    public static ArrayList<Integer>solve(ArrayList<ArrayList<Integer>>grid){
        ArrayList<Integer> optimal = new ArrayList<Integer>();
        int frow = Collections.max(grid.get(0));
        int lrow = Collections.max(grid.get(grid.size()-1));
        int next = grid.get(0).indexOf(frow);
        int nextl = grid.get(grid.size()-1).indexOf(lrow);
        int value = 0;
        int index = 0;
        int row = 0;
        if(frow> lrow){
            optimal.add(1);   
            optimal.add(grid.get(0).indexOf(frow));
            if(next<grid.get(0).size()-1)
{
            if(grid.get(0).get(next+1)>=grid.get(1).get(next+1) && grid.get(0).get(next+1)>=grid.get(grid.size()-1).get(next+1)){
                optimal.add(0);
                value = grid.get(0).get(next+1);
                index = grid.get(0).indexOf(value);
                row = 0;
                
}
            else if(grid.get(1).get(next+1)> grid.get(0).get(next+1) && grid.get(1).get(next+1)> grid.get(grid.size()-1).get(next+1)){
                optimal.add(-1);
                value = grid.get(1).get(next+1);
                index = grid.get(1).indexOf(value);
                row = 1;
                }
            else{
                optimal.add(1);
                value = grid.get(grid.size()-1).get(next+1);
                index = grid.get(grid.size()-1).indexOf(value);
                row = grid.size()-1;
                } 
            }
        }
        /*  
        for (int i = 0; i < grid.size(); i++){
            for (int j = 0; j < grid.get(0).size(); j++){
            }
        }   
        */

        else if(lrow>=frow){
            optimal.add(0);
            optimal.add(grid.get(grid.size()-1).indexOf(lrow));
            if(nextl<grid.get(0).size()-1){
            if(
grid.get(grid.size()-1).get(nextl+1) >= grid.get(grid.size()-2).get(nextl+1) && grid.get(grid.size()-1).get(nextl+1) >= grid.get(0).get(nextl+1)){
                optimal.add(0);
                value = grid.get(grid.size()-1).get(nextl+1);
                index = grid.get(grid.size()-1).indexOf(value);
                row = 
grid.size()-1;
                }
            else if(grid.get(grid.size()-2).get(nextl+1)> grid.get(grid.size()-1).get(nextl+1) && grid.get(grid.size()-2).get(nextl+1) >  grid.get(0).get(nextl+1)){
                optimal.add(1);
                value = grid.get(grid.size()-2).get(nextl+1);
                index = grid.get(grid.size()-2).indexOf(value);
                row = grid.size()-2;
                }
            else{
                optimal.add(-1);
                value = grid.get(0).get(nextl+1);
                index = grid.get(0).indexOf(value);
                row = 0;
                }
            }
          }

        if(grid.get(0).contains(value)==false && 
grid.get(grid.size()-1).contains(value)==false){
            if(grid.get(row).get(index+1) >= grid.get(row+1).get(index+1) && grid.get(row).get(index+1)>= grid.get(row-1).get(index+1)){
                
optimal.add(0);
                }
            else if(grid.get(row+1).get(index+1)>grid.get(row).get(index+1) &&  grid.get(row+1).get(index+1)>
grid.get(row-1).get(index+1)){
                optimal.add(-1);
                }
            else{
                optimal.add(1);
                }
            }
        
        else if(grid.get(0).contains(value) && index == grid.get(0).indexOf(value)){
            
if(grid.get(0).get(index+1)>= grid.get(1).get(index+1) && grid.get(0).get(index+1) >= grid.get(grid.size()-1).get(index+1)){
                
optimal.add(0);
                }
            else if(grid.get(1).get(index+1)>= grid.get(0).get(index+1) && grid.get(1).get(index+1) >= 
grid.get(grid.size()-1).get(index+1)){
                optimal.add(-1);
                }
            else{
                optimal.add(1);
            }
        }
        
        

        return optimal;
    }
}
