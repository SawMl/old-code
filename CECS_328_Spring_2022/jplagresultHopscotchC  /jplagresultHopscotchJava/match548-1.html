import java.util.ArrayList;

class StudentSolver {

  public static int move(ArrayList<ArrayList<Integer>> grid, int direction, int row, int col) {
    if (direction == -1) {
        row = row + 1;
        col = col + 1;
        if (row == grid.size()) {
          row = 0;
        }
        if (col == grid.get(0).size()) {
          col = 0;
        }
        return grid.get(row).get(col);
      }
      if (direction == 0) {
        col = col + 1;
        if (col == grid.get(0).size()) {
          col = 0;
        }
        return grid.get(row).get(col);
      }
      if (direction == 1) {
        row = row - 1;
        col = col + 1;
        if (row < 0) {
          row = grid.size() - 1;
        }
        if (col == grid.get(0).size()) {
          col = 0;
        }
        return grid.get(row).get(col);
      
}
    return 0;
  }

  public static ArrayList<Integer> answer(ArrayList<ArrayList<Integer>> grid, int score, int steps, int row, int col, ArrayList<Integer> ans) {
    for(int i = -1; i <= 1; i++) {
      if(steps != 0) {
        score = score + move(grid, i, row, col);
        ans.add(i);
        answer(grid, score, steps-1, row, col, ans);
      
}
    }
    return ans;
  }

  public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {

    ArrayList<Integer> output = new ArrayList<Integer>();
    for(int i = 0; i <= 1; i++) {
      output.add(i); 
      for(int j = 0; j < grid.get(0).size(); j++) {
        output.add(j); 
        int step = 0;
        int total = grid.get(i).get(j);
        if(i == 0) {
          total = grid.get(grid.size() - 1).get(j);
        }
      }
    }
    return output;
    
  } 

}
