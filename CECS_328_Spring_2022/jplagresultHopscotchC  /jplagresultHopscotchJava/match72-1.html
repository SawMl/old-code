import java.util.ArrayList;
import static java.lang.Math.max;

class StudentSolver {
  public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>>
grid) {
    ArrayList<Integer> answer = new ArrayList<Integer>();
    
    int row = 0;
    int column = 0;
    int random = (int) (Math.random() * (1 - 0 + 1)) + 0;
    int max = -1;

    if (random == 0) {
      for (int i = 0; i < grid.get(0).size(); i++) {
        max = max(max, grid.get(0).get(i));
        if (max == grid.get(0).get(i)) {
          row = 1;
          column = i;
        }
      } 
    }
    else {
      for (int i = 0; i < grid.get(grid.size() - 1).get(i) ; i++) {
        max = max(max, grid.get(grid.size() - 1).get(i));
        if (max == grid.get(grid.size() - 1).get(i)) {
          row = 0;
          column = i;
        }
      }
    }

    answer.add(row);
    answer.add(column);

    if (row == 1) {
      row = 0;
    }
    else {
      row = grid.size() - 1;
    }
    
    int upright = 0;
    int right = 0;
    int downright = 0;
    
    for (int i = 0; i < grid.get(0).size() - 1; i++) {
      if (row == 0 && column != grid.get(0).size() - 1) { 
        upright = grid.get(grid.size() - 1).get(column + 1);
        right = grid.get(row).get(column + 1);
        downright = grid.get(row + 1).get(column + 1);
        max = max(max(upright, right), downright);
        if (max == upright) {
          answer.add(1);
          row = grid.size() - 1;
          column++;
        }
        else 
if (max == right) {
          answer.add(0);
          column++;
        }
        else {
          answer.add(-1);
          row++;
          column++;
        }
      }
      else if (row == grid.size() - 1 && column != grid.get(0).size() - 1) { 
        upright = grid.get(row - 1).get(column + 1);
        right = grid.get(row).get(column + 1);
        downright = grid.get(0).get(column + 1);
        max = max(max(upright, right), downright);
        if (max == upright) {
          answer.add(1);
          row--;
          column++;
        }
        else if (max == right) {
          answer.add(0);
          column++;
        }
        else {
          answer.add(-1);
          row = 0;
          column++;
        }
      }
      else if (column == grid.get(0).size() - 1 && row == 0) { 
        upright = grid.get(grid.size() - 1).get(0);
        right = grid.get(row).get(0);
        downright = grid.get(row + 1).get(0);
        max = max(max(upright, right), downright);
        if (max == upright) {
          answer.add(1);
          row = grid.size() - 1;
          column = 0;
        }
        else if (max == right) {
          answer.add(0);
          column = 0;
        }
        else {
          answer.add(-1);
          row++;
          column = 0;
        }
      }
      else if (column == grid.get(0).size() - 1 && row == grid.size() - 1) { 
        upright = grid.get(row - 1).get(0);
        right = grid.get(row).get(0);
        downright = grid.get(0).get(0);
        max = max(max(upright, right), downright);
        if (max == upright) {
          answer.add(1);
          row--;
          column = 0;
        }
        else if (max == right) {
          answer.add(0);
          column = 0;
        }
        else {
          answer.add(-1);
          row = 0;
          column = 0;
        }
      }
      else if (column == grid.get(0).size() - 1 && (row != 0 || row != grid.size() - 1)) { 
        upright = grid.get(row - 1).get(0);
        right = grid.get(row).get(0);
        downright = grid.get(row + 1).get(0);
        max = max(max(upright, right), downright);
        if (max == upright) {
          answer.add(1);
          row--;
          column = 0;
        }
        else if (max == right) {
          answer.add(0);
          column = 0;
        }
        else {
          answer.add(-1);
          row++;
          column = 0;
        }
      }
      else {
        upright = grid.get(row - 1).get(column + 1);
        right = grid.get(row).get(column + 1);
        downright = grid.get(row + 1).get(column + 1);
        max = max(
max(upright, right), downright);
        if (max == upright) {
          answer.add(1);
          row--;
          column++;
        }
        else if (max == right) {
          answer.add(0);
          column++;
        }
        else {
          answer.add(-1);
          row++;
          column++;
        }
      }
    }
    return answer;
  }
}
