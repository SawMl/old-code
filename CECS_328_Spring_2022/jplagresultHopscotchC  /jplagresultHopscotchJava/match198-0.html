import java.util.*;

class StudentSolver{
  public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid){

    
    int gridSize = grid.size();

    ArrayList<Integer> userArr = new ArrayList<Integer>();

      for (int i = 0; i < gridSize; i++){
        for (int j = 0; j < gridSize; j++){
          if(grid[i][j] < grid[i+1][j]){
            userArr.add(0);
          }
          else if(grid[i][j] < grid[i+1][j+1]){
            userArr.add(1);
          }
          else if(grid[i][j] < grid[i+1][j-1]){
            userArr.add(-1);
          }
        }
      }
      
      int choice1 = rand.nextInt(2);
      userArr.add(choice1);
  
      
      int choice2 = rand.nextInt(gridSize + 1);
      userArr.add(choice2);  

      for (int k = 2; k < gridSize; k++){
        int choices = rand.nextInt(2);
        userArr.add(choices);
      }
      
        
        
        
      return userArr;
    }
  }
