import java.util.*;
import java.awt.*;
public class Solution 
{
	int score;
	ArrayList<Integer> moves;
	int startRow;
	ArrayList<Point> positions;
	int startIndex;
	Solution(ArrayList<Integer> moves, int score, int startRow, ArrayList<Point> positions)
	{
		this.score = score;
		this.moves = moves;
		this.startRow = startRow;
		this.positions = positions;
		if (moves.size() > 1)
		{
			this.startIndex = moves.get(1);
		}
		else
		{
			this.startIndex = 0;
		}
	}
	ArrayList<Integer> getMoves(Solution solution)
	{
		return solution.moves;
	}
	int getScore(Solution solution)
	{
		return solution.score;
	}
	int getStartRow(Solution solution)
	{
		return solution.startRow;
	}
	int getStartIndex(Solution solution)
	{
		return solution.startIndex;
	}
	void addToMoves(Solution solution, int move)
	{
		solution.moves.add(move);
	}
	void removeFromMoves(Solution solution)
	{
		solution.moves.remove(solution.moves.size() - 1);
	}
	void addToPositions(Solution solution, Point currentPosition, int step)
	{
		solution.positions.add(currentPosition);
	}
	int getRow(Solution solution, int index)
	{
		return solution.positions.get(index).x;
	}
	int getColumn(Solution solution, int index)
	{
		return solution.positions.get(index).y;
	}
}

			ArrayList<Integer> possibleSolution = new ArrayList<>();

					possibleSolution.add(0);
				}
				else if (i / (int) Math.pow(3, j) % 3 == 1)

				}
				else
				{
					possibleSolution.add(-1);
				}
			}
