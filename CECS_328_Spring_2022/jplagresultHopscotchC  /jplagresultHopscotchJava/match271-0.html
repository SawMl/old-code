import java.util.*;

class StudentSolver {
  public static int rows, cols = 0;
  
  public static void main(String[] args) {
    ArrayList<ArrayList<Integer>> grid = new ArrayList<ArrayList<Integer>> ();
	  Scanner in = new Scanner(System.in);
	    		
		System.out.print("Enter number of rows (m): ");
		rows = getInteger();
		System.out.print("Enter number of columns (n): ");
		cols = getInteger();

    
    for (int r=0; r<rows; r++){
      grid.add(new ArrayList<Integer> (cols));
      for (int c=0; c<cols; c++){
        System.out.print("Enter element for row " + r + ", column " + c + ": ");
        grid.get(r).add(c, getInteger());
      }
    }    

    
    for (int r=0; r<rows; r++){
      grid.add(new ArrayList<Integer> (cols));
      for (int c=0; c<cols; c++){
        if (c != cols-1) {
          System.out.print(grid.get(r).get(c) + ", ");
        } else {
          System.out.print(grid.get(r).get(c));
        }
      }
      System.out.println();
    }

    solve(grid);
		
  }

  public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {
    Random randomNumberGen = new Random();
    ArrayList<Integer> answer = new ArrayList<Integer> ();
    
		int entry = randomNumberGen.nextInt(2); 
    int currentRow = 0;

    if (entry == 1) { 
      currentRow = 0;
    } else if (entry == 0) { 
      currentRow = rows-1;
    }
    answer.add(entry); 
		
		int currentColumn = randomNumberGen.nextInt(cols); 
		answer.add(currentColumn); 

    int score = grid.get(currentRow).get(currentColumn);
    System.out.println("Starting column = " + currentColumn);
    System.out.println("Starting score = " + score);	

    int moves = cols-1; 
    int max = 0;
    int moveRight, moveRightUp, moveRightDown = 0;

    while (moves != 0) {
      
      
      if ((currentColumn+1) % cols == 0) { 
        currentColumn = 0;
        
      } else {
        currentColumn++;
        
      }

      moveRight = grid.get(currentRow).get(currentColumn);      
      max = moveRight;

      
      
      int tempRowUp, tempRowDown = 0;
      if ((currentRow-1) < 0) {
        
        
        tempRowUp = rows-1;
        moveRightUp = grid.get(tempRowUp).get(currentColumn);
      } else {
        
        
        tempRowUp = currentRow-1;
        moveRightUp = grid.get(tempRowUp).get(currentColumn);
      }
      
      
      if ((currentRow+1) % rows == 0) {
        
        tempRowDown = 0;
        moveRightDown = grid.get(tempRowDown).get(currentColumn);
      }
      else {
        
        tempRowDown = currentRow+1;
        moveRightDown = grid.get(tempRowDown).get(currentColumn);
      }

      max = moveRight;
      if (moveRightUp > max) {
        max = moveRightUp;
      }
      if (moveRightDown > max) {
        max = moveRightDown;
      }

      if (max == moveRight) {
        answer.add(0);
      } else if (max == moveRightUp) {
        answer.add(1);
        currentRow = tempRowUp;
        
      } else if (max == moveRightDown) {
        answer.add(-1);
        currentRow = tempRowDown;
      }

      score = score + max; 
      moves--; 
    }

    System.out.println(answer);
    System.out.println(score);

    return answer;
  }

  public static int getInteger() {
    Scanner in = new Scanner(System.in);
    boolean isValid = false;
    int userIn, num = 0;
    
    while (isValid == false) {
      if (in.hasNextInt()) {
        userIn = in.nextInt();
        if (userIn > 0) {
          isValid = true;
          num = userIn;
        } else {
          System.out.println("Input out of range! Please enter a positive integer.");
        }
        
      } else {
        in.next(); 
        System.out.println("Invalid input. Please try again.");
      }
    }

    return num;
  }

  
    
  
}
