    public static int updateCell(int col, int cols)
    {
        col++;
        if(col==cols)
        {
            col = 0;
        }
        return col;
    }

    public static int moveUpRow(int row, int rows)
    {
        row--;
        if(row == -1)
        {
            row = rows-1;
        }
        return row;
    }

    public static int moveDownRow(int row, int rows)
    {
        row++;
        if(row == rows)
        {
            row = 0;
        }
        return row;
    }
}
