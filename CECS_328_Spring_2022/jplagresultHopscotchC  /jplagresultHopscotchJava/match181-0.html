	 public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>>grid) 
	 {
		
			int M = grid.size();

			for(int i = 0; i < N - 1; i++)
			{
				if(grid.get(0).get(i) + grid.get(0).get(i + 1) > highestValue)
				{
					highestValue = grid.get(0).get(i) + grid.get(0).get(i + 1);
					computedStartEndValues.clear();
					computedStartEndValues.add(0);
					computedStartEndValues.add(i);
					computedStartEndValues.add(0);
					computedStartEndValues.add(i + 1);
				}
				
				if(grid.get(M - 1).get(i) + grid.get(M - 1).get(i + 1) > highestValue)
				{
					highestValue = grid.get(M - 1).get(i) + grid.get(M - 1).get(i + 1);
					computedStartEndValues.clear();
					computedStartEndValues.add(M - 1);
					computedStartEndValues.add(i);
					computedStartEndValues.add(M - 1);
					computedStartEndValues.add(i + 1);
				}
				
				if(grid.get(0).get(i) + grid.get(M - 1).get(i + 1) > highestValue)
				{
					highestValue = (grid.get(0).get(i) + grid.get(M - 1).get(i + 1));
					computedStartEndValues.clear();
					computedStartEndValues.add(0);
					computedStartEndValues.add(i);
					computedStartEndValues.add(M - 1);
					computedStartEndValues.add(i + 1);
				}
			
				if(i == N - 2)
				{
					if(grid.get(0).get(0) + grid.get(0).get(N - 1) > highestValue)
					{
						highestValue = grid.get(0).get(0) + grid.get(0).get(N - 1);
						computedStartEndValues.clear();
						computedStartEndValues.add(0);
						computedStartEndValues.add(0);
						computedStartEndValues.add(0);
						computedStartEndValues.add(N - 1);
					}
					
					if(grid.get(0).get(i + 1) + grid.get(M - 1).get(0) > highestValue)
					{
						highestValue = grid.get(0).get(i + 1) + grid.get(M - 1).get(0);
						computedStartEndValues.clear();
						computedStartEndValues.add(0);
						computedStartEndValues.add(i + 1);
						computedStartEndValues.add(M - 1);
						computedStartEndValues.add(0);
					}
				}
			}
			
			
			
			if(computedStartEndValues.get(0) == 0)
			{
				optimalHopscotchSolution.add(1);
				startRow = 0;
			}
			else
			{
				optimalHopscotchSolution.add(0);
				startRow = M - 1;
			}
			
			
			optimalHopscotchSolution.add(computedStartEndValues.get(3));
			startCol = computedStartEndValues.get(3);
			
			
			
			
		
					
			
			
			ArrayList<Integer> hopscotchNextValueInformation = new ArrayList<Integer>();
			int row = 0;
			int col = 0;
			int currentTotalValue = 0;
			
			for(int j = 0; j < N*2; j++)
			{
				currentReturnSolution.clear();
				if(j < N) 
				{
					row = 0;
					col = j;
					currentReturnSolution.add(1);
					currentReturnSolution.add(col);
				}
				else if(j > N/2)
				{
					row = M - 1;
					col = j - N;
					currentReturnSolution.add(0);
					currentReturnSolution.add(col);
				}
				int totalValue = grid.get(row).get(col);
				
					
				for(int i = 0; i < movesAllowed; i++)
				{	



























				hopscotchNextValueInformation = findPath(grid, col, row);
				
				
				row = hopscotchNextValueInformation.get(1);
				col = hopscotchNextValueInformation.get(2);
				totalValue += hopscotchNextValueInformation.get(0);
				}
			
			
			
			if(totalValue > currentTotalValue)
			{
				currentTotalValue = totalValue;
				finalReturnSolution.clear();

				for(int i = 0; i < currentReturnSolution.size(); i++)
				{
					finalReturnSolution.add(currentReturnSolution.get(i));
				}
			}
			
			
			





			}
			





			
			return finalReturnSolution;
		
	 }

		public static int[] findLargest(int right, int rightUp, int rightDown)
		{
			int[] hopscotchInformation = new int [2];
			
			
			if( right > rightUp)
			{
				if (right > rightDown)
				{
					hopscotchInformation[0] = 0;
					hopscotchInformation[1] = right;
				}
				else
				{
					hopscotchInformation[0] = -1; 
					hopscotchInformation[1] = rightDown;
				}
			}
			else if( rightUp > rightDown )
			{
				hopscotchInformation[0] = 1;
				hopscotchInformation[1] = rightUp;
			}
			else
			{
				hopscotchInformation[0] = -1;
				hopscotchInformation[1] = rightDown;
			}
			
			
			return hopscotchInformation;
		}
