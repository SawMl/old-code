import java.util.ArrayList;

public class StudentSolver {

    public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid){
        ArrayList<Integer> top = new ArrayList<>();
        ArrayList<Integer> bot = new ArrayList<>();
        top = startTop(grid);
        int sumTop = sumT(grid);
        bot = startBot(grid);
        int sumBot = sumB(grid);

        if (sumTop >= sumBot){
            return top;
        }else
            return bot;
    }

    public static int sumT(ArrayList<ArrayList<Integer>> grid){
        ArrayList<Integer> num = new ArrayList<>();
        ArrayList<Integer> top = new ArrayList<>();
        int temp = 0;
        int sum = 0;

        
        for (int i = 0; i < 1; i++){
            for (int j = 0; j < grid.get(i).size(); j++){
                top.add(grid.get(i).get(j));
            }
        }
        
        for(int i = 0; i < top.size(); i++){
            if(temp < top.get(i)){
                temp = top.get(i);
            }
        }

        sum +=temp;
        num.add(top.indexOf(temp));
        int col = top.indexOf(temp);
        int right = col;
        int upr = col;
        int botr = col;
        int row = 0;


        for(int i =0; i < top.size() -1 ; i++){

            if ((row == 0) && (col + 1 >= top.size())){

                upr = grid.get(grid.size() - 1).get(0);
                right = grid.get(row).get(0);
                botr = grid.get(1).get(0);

                if (upr >= right && upr >= botr) {
                    num.add(1);
                    row = grid.size() - 1;
                    col = 0;
                    sum += grid.get(row).get(col);
                } else if (right >= upr && right >= botr) {
                    num.add(0);
                    col = 0;
                    sum += grid.get(row).get(col);

                } else {
                    num.add(-1);
                    row++;
                    col = 0;
                    sum += grid.get(row).get(col);

                }
            }
            else if (row == 0){
                upr = grid.get(grid.size() - 1).get(col + 1);
                right = grid.get(row).get(col + 1);
                botr = grid.get(1).get(col + 1);

                if(upr >= right && upr >= botr){
                    num.add(1);
                    row = grid.size() - 1;
                    col++;
                    sum += grid.get(row).get(col);

                }else if(right >= upr && right >= botr){
                    num.add(0);
                    col++;
                    sum += grid.get(row).get(col);

                }else {
                    num.add(-1);
                    row++;
                    col++;
                    sum += grid.get(row).get(col);

                }
            } else if((row == grid.size() - 1) && (col + 1) >= top.size()){

                
upr = grid.get(row -1).get(0);
                right = grid.get(row).get(0);
                botr = grid.get(0).get(0);

                if(upr >= right && upr >= botr){
                    num.add(1);
                    row = row -1;
                    col = 0;
                    sum += grid.get(row).get(col);

                }else if( right >= upr && right >= botr){
                    num.add(0);
                    col = 0;
                    sum += grid.get(row).get(col);

                }else{
                    num.add(-1);
                    row = 0;
                    col = 0;
                    sum += grid.get(row).get(col);

                }

            } else if (row == grid.size() -1){
                upr = grid.get(row - 1).get(col + 1);
                right = grid.get(row).get(col + 1);
                botr = grid.get(0).get(col + 1);


                if(upr >= right && upr >= botr){
                    num.add(1);
                    row = row -1;
                    col++;
                    sum += grid.get(row).get(col);

                }else if( right >= upr && right >= botr){
                    num.add(0);
                    col++;
                    sum += grid.get(row).get(col);

                }else {
                    num.add(-1);
                    row = 0;
                    col++;
                    sum += grid.get(row).get(col);

                }

            } else{
                upr = grid.get(row - 1).get(col + 1);
                right = grid.get(row).get(col + 1);
                botr = grid.get(row + 1).get(col + 1);

                if((upr == right) || (upr == botr) || (right == botr)){
                    if(upr == botr){
                        if(((grid.size() - 1) - row) < row  ){
                            num.add(-1);
                            row++; col++;
                            sum += grid.get(row).get(col);

                        }else{
                            num.add(1);
                            row--; col++;
                            sum += grid.get(row).get(col);

                        }

                    }else{
                        num.add(0);
                        col++;
                        sum += grid.get(row).get(col);

                    }

                } else if(upr >= right && upr >= botr){
                    num.add(1);
                    row = row -1;
                    col++;
                    sum += grid.get(row).get(col);

                }else if( right >= upr && right >= botr){
                    num.add(0);
                    col++;
                    sum += grid.get(row).get(col);

                }else{
                    num.add(-1);
                    row++;
                    col++;
                    sum += grid.get(row).get(col);

                }

            
}


        }
        return sum;
    }
