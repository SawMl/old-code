import java.util.ArrayList;
import java.util.Arrays;
import java.util.Scanner;


public class StudentSolver {
    static Scanner in = new Scanner(System.in);
  
    public static void main(String[] args) {
        
        ArrayList<ArrayList<Integer>> grid = new ArrayList(new ArrayList<>());
        
        ArrayList<Integer> col = new ArrayList<>(Arrays.asList(3,4,1,2,8,6));
        grid.add(col);
        col = new ArrayList<>(Arrays.asList(6,1,8,2,7,4));
        grid.add(col);
        col = new ArrayList<>(Arrays.asList(5,4,3,9,9,5));
        grid.add(col);
        col = new ArrayList<>(Arrays.asList(5,9,8,3,2,6));
        grid.add(col);
        col = new ArrayList<>(Arrays.asList(8,7,2,9,6,4));
        grid.add(col);
      
        int highestScore = 0;
        
        while(true){
            
            col = solve(grid);
            
            int score = 0;
            for(int i=0; i<col.size(); i++){
                score = score+col.get(i);
            }
            
            System.out.println("Score: "+score);
            
            if(score>highestScore)
                highestScore = score;
            
            System.out.println("Highest Score: "+highestScore);
            
            System.out.print("Play again? Yes(Y) No(N): ");
            String input = in.next();
            if(input.equalsIgnoreCase("N"))
                break;
        }
    }
  
    
    public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid){
        
        ArrayList<Integer> answer = new ArrayList<>();
        
        ArrayList<Integer> indexCol = new ArrayList<>();
        
        ArrayList<Integer> path = new ArrayList<>();
        int rows = grid.size();
        int cols = grid.get(0).size();
        
        for(int i=0; i<rows; i++){
            for (int j=0; j<cols; j++){
                System.out.print(grid.get(i).get(j)+" ");
            }
            System.out.println();
        }
        
        System.out.print("Enter start position top (1) or bottom (0): ");
        int top = in.nextInt();
        path.add(top);
        int row = 0;
        if(top==0)
            row = rows-1;
        else
            row=0;
        System.out.print("Enter start colum index: ");
        int col = in.nextInt();
        path.add(col);
        
        while(true){
            answer.add(grid.get(row).get(col));
            indexCol.add(col);
            int flag=1;
            for(int i=0; i<cols; i++){
                if(!indexCol.contains(i))
                    flag=0;
            }
            if(flag==1){
                System.out.println("One revolution is done.");
                break;
            }
            System.out.print("Enter move: rigth(0)or up-right(1) or down-right(-1): ");
            int move = in.nextInt();
            path.add(move);
            switch(move){
                case 0:
                    col = updateCell(col, cols);
                    break;
                case 1:
                    col = updateCell(col, cols);
                    row--;
                    if(row==-1)
                        row=rows-1;
                    break;
                case -1:
                    col = updateCell(col, cols);
                    row = updateCell(row, rows);
                    break;
            }
       
        }
        
        System.out.print("Path: ");
        for(int i=0; i<path.size();i++){
            System.out.print(path.get(i)+" ");
        }
        System.out.println();
        return answer;
    }

    
    public static int updateCell(int col, int cols) {
        col++;
        if(col==cols)
            col=0;
        return col;
    }
}
