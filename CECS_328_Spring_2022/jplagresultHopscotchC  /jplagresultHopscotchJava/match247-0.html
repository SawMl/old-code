import java.lang.Math;
import java.util.ArrayList;

public class StudentSolver {
    public static void main(String[] args) {
        int[] arr = { 9, 4, 1, 2, 9, 6, 6, 1, 8, 9, 7, 4, 5, 4, 3, 9, 9, 5, 5, 9, 8, 3, 2, 6, 8, 7, 2, 9, 6, 4, 1, 9, 9, 9, 9, 8, 1, 9, 9, 9, 9, 1, 1,0, 0, 0, 0, 0, 0, 0 };
        ArrayList<ArrayList<Integer>> grid = new ArrayList<ArrayList<Integer>>();
        int i = 0;
        for (int j = 0; j < 3; j++) {
            ArrayList<Integer> row = new ArrayList<Integer>();
            for (int k = 0; k < 10; k++) {
                row.add(arr[i]);
                i++;
            }
            grid.add(row);
        }
        for (ArrayList<Integer> arrayList : grid) {
            System.out.println(arrayList);
        }
        
        System.out.println(solve(grid));

    }

    public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {

        
        
        
        
        
        
        
        
        
        
        ArrayList<Integer> solution = new ArrayList<Integer>();

        

        
        for (int i = 0; i < grid.get(0).size(); i++) {
            ArrayList<Integer> tempSolution = new ArrayList<Integer>();
            tempSolution.add(0); 
            tempSolution.add(0); 
            tempSolution.add(i); 
            tempSolution = recursiveSolving(grid, tempSolution, grid.size() - 1, i, null);
            if (solution.size() == 0 || tempSolution.get(0) > solution.get(0)) {
                solution = tempSolution;
            }
        }

        
        for (int i = 0; i < grid.get(0).size(); i++) {

            ArrayList<Integer> tempSolution = new ArrayList<Integer>();
            tempSolution.add(0); 
            tempSolution.add(1); 
            tempSolution.add(i); 
            tempSolution = recursiveSolving(grid, tempSolution, 0, i, null);
            if (solution.size() == 0 || tempSolution.get(0) > solution.get(0)) {
                solution = tempSolution;
            }
        }

        System.out.println(solution);
        solution.remove(0);
        return solution;
    }

    public static ArrayList<Integer> recursiveSolving(ArrayList<ArrayList<Integer>> grid,
            ArrayList<Integer> scoreAndPath, Integer rowPosition, Integer columnPosition, Integer direction) {

        scoreAndPath.set(0, scoreAndPath.get(0) + grid.get(rowPosition).get(columnPosition));
        if (direction != null) {
            scoreAndPath.add(direction);
        }
        if (scoreAndPath.size() == grid.get(0).size() + 2) { 
            if (rowPosition != 0 && rowPosition != grid.size() - 1) {
                scoreAndPath.set(0, -1); 
            }

            return scoreAndPath;
        }

        ArrayList<Integer> maxPlusOne = recursiveSolving(grid, cloneList(scoreAndPath),
                decrementRow(grid, rowPosition),
                incrementCol(grid, columnPosition), 1);
        ArrayList<Integer> maxPlusZero = recursiveSolving(grid, cloneList(scoreAndPath), rowPosition,
                incrementCol(grid, columnPosition), 0);
        ArrayList<Integer> maxMinusOne = recursiveSolving(grid, cloneList(scoreAndPath),
                incrementRow(grid, rowPosition),
                incrementCol(grid, columnPosition), -1);

        ArrayList<Integer> solution = maxPlusOne;
        if (maxPlusZero.get(0) > solution.get(0)) {
            solution = maxPlusZero;
        }
        if (maxMinusOne.get(0) > solution.get(0)) {
            solution = maxMinusOne;
        }
        return solution;

    }

    public static Integer incrementRow(ArrayList<ArrayList<Integer>> grid, Integer rowPosition) {
        if (rowPosition == grid.size() - 1) {
            return 0;
        } else {
            return rowPosition + 1;
        }
    }

    public static Integer decrementRow(ArrayList<ArrayList<Integer>> grid, Integer rowPosition) {
        if (rowPosition == 0) {
            return grid.size() - 1;
        } else {
            return rowPosition - 1;
        }
    }

    public static Integer incrementCol(ArrayList<ArrayList<Integer>> grid, Integer colPosition) {
        if (colPosition == grid.get(0).size() - 1) {
            return 0;
        } else {
            return colPosition + 1;
        }
    }

    public static ArrayList<Integer> cloneList(ArrayList<Integer> list) {
        ArrayList<Integer> clonedList = new ArrayList<Integer>();
        for (Integer element : list) {
            clonedList.add(element);
        }
        return clonedList;
    }
}
