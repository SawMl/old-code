					if(currCol == grid.get(0).size()-1) { 
						currCol = 0; 
						if(currRow == 0) { 
							if(grid.get(currRow).get(currCol) >= grid.get(grid.size()-1).get(currCol)) { 
								if(grid.get(currRow).get(currCol) >= grid.get(currRow+1).get(currCol)) {

									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(0); 
								}
							}

								if(grid.get(currRow-1).get(currCol) >= grid.get(0).get(currCol)) {
									currRow -= 1;
									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(1); 
								}
							}
							else if(grid.get(0).get(currCol) >= grid.get(currRow).get(currCol)) {
								if(grid.get(0).get(currCol) >= grid.get(currRow-1).get(currCol)) {
									currRow = 0;
									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(-1); 
								}
							}
						}
					}
					else {
						currCol += 1;
						if(currRow == 0) { 
							if(grid.get(currRow).get(currCol) >= grid.get(grid.size()-1).get(currCol)) { 
								if(grid.get(currRow).get(currCol) >= grid.get(currRow+1).get(currCol)) 
								{
									sum += grid.get(currRow).get(currCol); 
									moves.get(start).add(0); 
								}
							}
							else if (grid.get(grid.size()-1).get(currCol) >= grid.get(currRow).get(currCol)) {
								if(grid.get(grid.size()-1).get(currCol) >= grid.get(currRow+1).get(currCol)) {
									sum += grid.get(grid.size()-1).get(currCol);
									currRow = grid.size()-1; 
									moves.get(start).add(1); 
								}
							}
							else if (grid.get(currRow+1).get(currCol) >= grid.get(currRow).get(currCol)) {
								if(grid.get(currRow+1).get(currCol) >= grid.get(grid.size()-1).get(currCol)) {
									sum += grid.get(grid.size()-1).get(currCol);
									currRow += 1; 
									moves.get(start).add(-1); 
								}
							}
						}
						else if(currRow == grid.size()-1) {
							if(grid.get(currRow).get(currCol) >= grid.get(currRow-1).get(currCol)) {
								if(grid.get(currRow).get(currCol) >= grid.get(0).get(currCol)) {
									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(0); 
								}
							}
							else if(grid.get(currRow-1).get(currCol) >= grid.get(currRow).get(currCol)) {
								if(grid.get(currRow-1).get(currCol) >= grid.get(0).get(currCol)) {
									currRow -= 1;
									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(1); 
								}
							}
							else if(grid.get(0).get(currCol) >= grid.get(currRow).get(currCol)) {
								if(grid.get(0).get(currCol) >= grid.get(currRow-1).get(currCol)) {
									currRow = 0;
									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(-1); 
								}
							}
						}
						else 
							if(grid.get(currRow).get(currCol) >= grid.get(currRow-1).get(currCol))
							{
								if(grid.get(currRow).get(currCol) >= grid.get(currRow+1).get(currCol)) {
									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(0); 
								}
							}
							else if(grid.get(currRow-1).get(currCol) >= grid.get(currRow).get(currCol))
							{
								if(grid.get(currRow-1).get(currCol) >= grid.get(currRow+1).get(currCol))
								{
									currRow -= 1;
									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(1); 
								}
							}
							else if(grid.get(currRow+1).get(currCol) >= grid.get(currRow).get(currCol))
							{
								if(grid.get(currRow+1).get(currCol) >= grid.get(currRow-1).get(currCol))
								{
									currRow += 1;
									sum += grid.get(currRow).get(currCol);
									moves.get(start).add(-1); 
								}
							}
					
					}} sumBottom.add(sum); 
			start1++;
			}		}
		
		int max = 0;
		int pos = 0;
		int enter = 0;
		
		for(int i = 0; i < sumTop.size(); i++)
		{
			if(sumTop.get(i) > max) {
				max = sumTop.get(i); 
				pos = i; 
				enter = 1;
			}
		}
		for(int i = 0; i < sumBottom.size(); i++)
		{
			if(sumBottom.get(i) > max) {
				max = sumBottom.get(i);
				pos = i;
				enter = 0;
			}
		}
		
		ArrayList<Integer> res = new ArrayList<Integer>();
		
		
		
		
		
		res.add(0);
		res.add(2);
		res.add(0);
		res.add(1);
		res.add(-1);
