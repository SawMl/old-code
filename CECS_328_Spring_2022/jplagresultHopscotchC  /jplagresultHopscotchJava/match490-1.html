	 public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid) {
		 
		 
		 
		 ArrayList<Integer> arr = new ArrayList<Integer>();
		 arr.add(0);
		 
		 
		 
		 int max = 0;
		 int xPos = 0;
		 int yPos = 0;
		 int remainingMoves = grid.get(0).size() - 1;
		 
		 
		 for (int i = 0; i < grid.get(0).size(); i++) {
	        	if(grid.get(0).get(i) > max) {
	        		max = grid.get(0).get(i);
	        		xPos = i;
	        		yPos = 0;
	        		arr.set(0, 1);
		        }
		    }
		 
		 for(int i = 0; i < grid.get(grid.
size() - 1).size(); i++) {
			 if(grid.get(grid.size() - 1).get(i) > max) {
				 max = grid.get(grid.size() - 1).get(i);
				 yPos = grid.size() - 1;
				 xPos = i;
				 arr.set(0, 0);
			 }
		 }
		 arr.add(yPos);
		 
		 
		 
		 
		 
		 
		 
		while(remainingMoves > 0) {
			 
			 ArrayList<Integer> potNextMovesArr = new ArrayList<Integer>();
			 
			 
			 if(xPos + 1 > grid.get(0).size() - 2) { 
				 
				 int xTemp = 0;
				 potNextMovesArr.add(grid.get(xTemp).get(yPos));
			 }
			 else { 
				 int xTemp = xPos + 1;
				 potNextMovesArr.add(grid.get(xTemp).get(yPos));
			 }
			 System.out.println("potNextMovesArr pos for right: " + potNextMovesArr.get(0));
			 
			 
			 if((xPos + 1 > grid.get(0).size() - 2) && (yPos - 1 < 0)) { 
				 int xTemp = 0;
				 int yTemp = grid.size() - 1;
				 potNextMovesArr.add(grid.get(yTemp).get(xTemp));
			 }
			 else if(xPos + 1 > grid.get(0).size() - 2) { 
				 int xTemp = 0;
				 int yTemp = yPos - 2;
				 potNextMovesArr.add(grid.get(xTemp).get(yTemp));
			 }
			 else if(yPos - 1 < 0) { 
				 int yTemp = grid.size() - 1;
				 int xTemp = xPos + 1;
				 potNextMovesArr.add(grid.get(xTemp).get(yTemp));
			 }
			 else { 
				 int xTemp = xPos + 1;
				 int yTemp = yPos - 2;
			 	 
				 potNextMovesArr.add(grid.get(xTemp).get(yTemp));
			 }
			 System.out.println("potNextMovesArr pos for up-right: " + potNextMovesArr.get(1));
			 
			 
			 
			if((xPos + 1 > grid.get(0).size() - 2) && yPos + 1 > (grid.size() - 1)) { 
				 int xTemp = 0;
				 int yTemp = 0;
				 potNextMovesArr.add(grid.get(xTemp).get(yTemp));
				 
			 }
			 else if(xPos + 1 > grid.get(0).size() - 2) {
				 int xTemp = 0;
				 int yTemp = yPos + 1;
				 potNextMovesArr.add(grid.get(yTemp).get(xTemp));
				 System.out.println("potNextMovesArr pos for down-right: " + potNextMovesArr.get(2));
			 }
			 else if(yPos + 1 > grid.size() - 1) { 
				 int xTemp = xPos + 1;
				 int yTemp = 0;
				 potNextMovesArr.add(grid.get(xTemp).get(yTemp));
				 System.out.println("potNextMovesArr pos for down-right: " + potNextMovesArr.get(2));
			 }
			 else { 
				 int xTemp = xPos + 1;
				 int yTemp = yPos + 1;
				 potNextMovesArr.add(grid.get(xTemp).get(yTemp));
				 System.out.println("potNextMovesArr pos for down-right: " + potNextMovesArr.get(2));
			 }
			 
			 
			 
			 int tempMax = 0;
			 for(int i = 0; i < potNextMovesArr.size(); i++) {
				 if(potNextMovesArr.get(i) > tempMax) {
					 tempMax = potNextMovesArr.get(i);
				 }
			 }
			 
			 
			 if(tempMax == potNextMovesArr.get(0)) { 
				 arr.add(0);
				 if(xPos + 1 > grid.get(0).size() - 2) { 
					 
					 xPos = 0; 
				 }
				 else { 
					 xPos++;
				 }
			 }
			 else if(tempMax == potNextMovesArr.get(1)) { 
				 arr.add(1);
				 if((xPos + 1 > grid.get(0).size() - 2) && (yPos - 1 < 0)) { 
					 xPos = 0;
					 yPos = grid.size() - 1;
				 }
				 else if(xPos + 1 > grid.get(0).size() - 2) { 
					 xPos = 0;
					 yPos = yPos - 2;
				 }
				 else if(yPos - 1 < 0) { 
					 yPos= grid.size() - 1;
					 xPos = xPos + 1;
				 }
				 else { 
					 xPos = xPos + 1;
					 yPos = yPos - 2;
				 }

			 }
			 else if(tempMax == potNextMovesArr.get(2)) { 
				 arr.add(-1);
				 if((xPos + 1 > grid.get(0).size() - 2) && yPos + 1 > (grid.size() - 1)) { 
					 xPos = 0;
					 yPos = 0;		 
				 }
				 else if(xPos + 1 > grid.get(0).size() - 2) {
					 xPos = 0;
					 yPos = yPos + 1;
				 }
				 else if(yPos + 1 > grid.size() - 1) { 
					 xPos = xPos + 1;
					 yPos = 0;
				 }
				 else { 
					 xPos = xPos + 1;
					 yPos = yPos + 1;
				 }	 
			 }
			 
			 

			 remainingMoves--;
		 }
			 System.out.println("arr: " + arr);
		return arr;
