import java.awt.Point;
import java.util.ArrayList;

public class Position{
  private Point p = new Point();
  private int x;
  private int y;

  public Position(int x, int y){
    this.x = x;
    this.y = y;
  }

  public Point goRight(ArrayList<ArrayList<Integer>> grid, Point p){
    p.translate(1, 0);
    int x = (int)p.getX();
    int y = (int)p.getY();
    if(x == grid.get(0).size()){
      p.setLocation(0, y);
    }
    return p;

  }

  public Point goUp(ArrayList<ArrayList<Integer>> grid, Point p){
    p.translate(0, -1);
    int x = (int)p.getX();
    int y = (int)p.getY();
    if(y == -1){
      p.setLocation(x, grid.size()-1);
    }
    return p;
  }

  public Point goDown(ArrayList<ArrayList<Integer>> grid, Point p){
    p.translate(0, 1);
    int x = (int)p.getX();
    int y = (int)p.getY();
    if(y == grid.size()){
      p.setLocation(x, 0);
    }
    return p;
  }

  public Point goLeft(ArrayList<ArrayList<Integer>> grid, Point p){
    p.translate(-1, 0);
    int x = (int)p.getX();
    int y = (int)p.getY();
    if(x == -1){
      p.setLocation(grid.get(0).size()-1, y);
    }
    return p;
  }
}
