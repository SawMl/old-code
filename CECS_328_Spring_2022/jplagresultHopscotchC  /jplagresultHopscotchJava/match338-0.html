    public static ArrayList<Integer> solve(ArrayList<ArrayList<Integer>> grid)
    {
        
        
        
        
        
        ArrayList<Integer> biggestpath = new ArrayList<Integer>();
        int biggestsum = 0;

        Integer moves[] = { -1, 0, 1};
        Integer moves2[] = {0,1};

        for (Integer i = 0; i < 10000000; i++)
        {
            ArrayList<Integer> currentpath = new ArrayList<Integer>();
            Random ran1 = new Random(); 
            int random1 = ran1.nextInt(2);
            currentpath.add(moves2[random1]);
            Random ran2 = new Random();
            int random2 = ran2.nextInt(grid.size()-1);
            currentpath.add(random2);

            for (Integer j = 0; j < grid.get(0).size()-1; j++)
            {
                Random ran3 = new Random();
                int n = ran3.nextInt(3);
                currentpath.add(moves[n]);
            }

            int row = 0;
            if (currentpath.get(0) == 0)
            {
                if (grid.size() != 0)
                {
                    row = grid.size()-1;
                }
                else
                {
                    row = 0;
                }
                
            }
            else if (currentpath.get(0) == 1)
            {
                row = 0;
            }
            int col = currentpath.get(1);
            int currentsum = grid.get(row).get(col);
            for (Integer j = 0; j < currentpath.size()-1; j++)
            {
                if (currentpath.get(j) == 0)
                        {
                            col = col+1;
                        }
                        else if (currentpath.get(j) == -1)
                        {
                            row = row-1;
                            col = col+1;
                        }
                        else if (currentpath.get(j) == 1)
                        {
                            row = row+1;
                            col = col+1;
                        }
                        if (row > grid.size()-1)
                        {
                            row = 0;
                        }
                        if (row < 0)
                        {
                            row = grid.size()-1;
                        }
                        if (col > grid.get(0).size()-1)
                        {
                            col = 0;
                        }
                        if (col < 0)
                        {
                            col = grid.get(0).size()-1;
                        }
                        currentsum = currentsum + grid.get(row).get(col);
                        }
                        if (row == 0 || row == grid.size()-1)
                                {
                                    if (biggestsum < currentsum)
                                    {
                                        biggestsum = currentsum;
                                        biggestpath = currentpath;
                                    }
                                }
            
        }
        System.out.print(biggestsum);
        return biggestpath;
    }
